## 🔢 一、数值计算与浮点体系（深度 B + 安全 G）

### 1.1 IEEE 754 的二进制级解码（深度）

**双精度（double）的 64 位布局**：

```
[S: 1bit][Exponent: 11bit][Mantissa: 52bit]
     63        62-52            51-0
```

- **指数偏移**：实际指数 = 存储值 - 1023（bias）
- **隐藏位**：规格化数隐含前导 1（1.xxx），非规格化数（subnormal）隐含 0
- **特殊值**：
  - `0x7FF0000000000000` = +Inf
  - `0x7FF8000000000001` = NaN（Quiet NaN，最高位 mantissa 为 1）
  - `0x0000000000000000` = +0

**位操作实战**：手动实现 `isnan`（不依赖标准库）

```c
#include <stdint.h>
#include <stdbool.h>

bool my_isnan(double x) {
    union { double d; uint64_t u; } pun = { .d = x };
    // 指数全 1 且尾数非 0 -> NaN
    return ((pun.u >> 52) & 0x7FF) == 0x7FF && (pun.u & 0xFFFFFFFFFFFFF) != 0;
}
```

### 1.2 浮点异常与 FENV（被忽视的标准库）

**C99 `<fenv.h>` 的精确控制**：

```c
#include <fenv.h>
#include <math.h>

void precise_calculation() {
    // 保存当前环境
    fenv_t old_env;
    feholdexcept(&old_env);  // 清除异常标志并建立非停止模式

    // 启用特定陷阱（硬件异常）
    feclearexcept(FE_ALL_EXCEPT);

    // 检查标志

        // 发生了精度舍入
    }

    // 恢复环境
    feupdateenv(&old_env);
}
```

### 1.3 数值稳定性算法（实战 C）

**灾难性抵消（Catastrophic Cancellation）**：

```c
// 错误：当 a ≈ b 时，sqrt 内两相近数相减丢失精度
double bad_distance(double x1, double y1, double x2, double y2) {
    return sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1));
}

// 安全：Kahan 求和算法（补偿求和）
double kahan_sum(const double *arr, size_t n) {
    double sum = 0.0;
    double c = 0.0;  // 补偿变量
    for (size_t i = 0; i < n; i++) {
        double y = arr[i] - c;  // 减去之前的误差
        double t = sum + y;     // 新的和
        c = (t - sum) - y;      // 计算新的误差（可能为负）
        sum = t;
    }
    return sum;
}
```

## ⚡ 二、异步信号与并发控制（深度 B + 安全 G）

### 2.1 信号处理的异步安全性（AS-Safe）

**唯一 AS-Safe 的函数**：`write()`（不是 `printf`！）、`_exit()`、`sem_post()`、`kill()` 等。
**致命陷阱**：在信号处理函数中调用 `malloc` 或 `printf` 会导致死锁（如果信号打断的正是 `malloc` 临界区）。

**安全信号处理模式（现代 Linux）**：

```c
#include <signal.h>
#include <unistd.h>
#include <stdatomic.h>

// 使用原子标志而非处理函数内执行复杂逻辑
static volatile sig_atomic_t signal_received = 0;
// C11 原子更优
static _Atomic(int) graceful_shutdown = 0;

void safe_handler(int sig) {
    // 唯一操作：设置标志
    atomic_store_explicit(&graceful_shutdown, 1, memory_order_relaxed);
    // 或写 pipe 通知主循环（Self-Pipe Trick）
    char byte = 1;
    write(signal_pipe[1], &byte, 1);  // AS-Safe
}

int main() {
    // 使用 sigaction 替代 signal()
    struct sigaction sa = {
        .sa_handler = safe_handler,
        .sa_flags = SA_RESTART  // 自动重启被中断的系统调用
    };
    sigemptyset(&sa.sa_mask);
    sigaction(SIGTERM, &sa, NULL);

    // 主事件循环
    while (!atomic_load(&graceful_shutdown)) {
        // 处理业务...
    }
}
```

### 2.2 实时信号与队列（POSIX 扩展）

**标准信号（SIGINT）会合并丢失，实时信号（SIGRTMIN+）排队**：

```c
union sigval value = { .sival_ptr = malloc(sizeof(request)) };
// 发送带数据的实时信号
sigqueue(target_pid, SIGRTMIN, value);

// 接收端使用 SA_SIGINFO
void advanced_handler(int sig, siginfo_t *info, void *context) {
    request *req = (request*)info->si_value.sival_ptr;
    // 处理请求...
```

### 2.3 `setjmp`/`longjmp` 的栈展开与资源泄漏（安全 G）

**问题**：`longjmp` 跳出时不会调用栈上对象的析构（C++ 会，但 C 不会），导致资源泄漏。

```c
    siglongjmp(jump_buffer, 1);  // 恢复信号掩码，防止竞态
}

int main() {

    if (sigsetjmp(jump_buffer, 1) == 0) {  // 保存信号掩码
        can_jump = 1;
        alarm(3);  // 3 秒后发送 SIGALRM
        risky_blocking_operation();
```

### 6.2 无锁数据结构（C11 原子）

**无锁队列（Michael-Scott Queue 简化版）**：

```c
typedef struct node {
    _Atomic(struct node*) next;
    int value;
} node_t;

typedef struct {
    _Atomic(node_t*) head;
    _Atomic(node_t*) tail;
} lock_free_queue;

void enqueue(lock_free_queue *q, int val) {
    node_t *new_node = malloc(sizeof(node_t));
    atomic_init(&new_node->next, NULL);
    new_node->value = val;

    node_t *tail = atomic_load_explicit(&q->tail, memory_order_relaxed);
    node_t *null = NULL;

    // CAS 循环直到成功
    while (!atomic_compare_exchange_weak_explicit(
        &tail->next, &null, new_node,
        memory_order_release, memory_order_relaxed)) {
        tail = atomic_load_explicit(&q->tail, memory_order_acquire);
        null = NULL;
    }
    atomic_compare_exchange_strong_explicit(
        &q->tail, &tail, new_node,
        memory_order_release, memory_order_relaxed);
}
```

## 🧠 一、内存分配器设计与实现（深度 B + 实战 C）

### 1.1 工业级分配器架构（超越 malloc）

**arena/bump allocator（编译器/游戏首选）**：

```c
typedef struct {
    uint8_t *buffer;
    size_t size;
    size_t used;
    size_t align;  // 通常为 8 或 16
} Arena;

void *arena_alloc(Arena *a, size_t size) {
    // 对齐计算：(addr + (align-1)) & ~(align-1)
    uintptr_t addr = (uintptr_t)(a->buffer + a->used);
    uintptr_t aligned = (addr + (a->align - 1)) & ~(a->align - 1);
    size_t offset = aligned - (uintptr_t)a->buffer;

    if (offset + size > a->size) return NULL;  // OOM

    void *ptr = a->buffer + offset;
    a->used = offset + size;

    // 零成本标记：调试模式下记录分配信息
    #ifndef NDEBUG
    printf("Arena alloc: %p, size %zu\n", ptr, size);
    #endif
    return ptr;
}

// 一键回滚（比 free 所有个体快 1000 倍）
void arena_reset(Arena *a) { a->used = 0; }
```

**pool allocator（固定大小对象，无碎片）**：

```c
typedef struct PoolChunk {
    struct PoolChunk *next;
} PoolChunk;

typedef struct {
    size_t obj_size;
    size_t chunk_capacity;
    uint8_t *memory;           // 大块内存
    PoolChunk *free_list;      // 空闲链表
} Pool;

void pool_init(Pool *p, size_t obj_size, size_t count) {
    p->obj_size = obj_size > sizeof(PoolChunk) ? obj_size : sizeof(PoolChunk);
    p->memory = malloc(p->obj_size * count);

    // 初始化空闲链表（头插法）
    p->free_list = NULL;
    for (size_t i = 0; i < count; i++) {
        PoolChunk *chunk = (PoolChunk*)(p->memory + i * p->obj_size);
        chunk->next = p->free_list;
        p->free_list = chunk;
    }
}

void *pool_alloc(Pool *p) {
    if (!p->free_list) return NULL;  // 池耗尽
    PoolChunk *chunk = p->free_list;
    p->free_list = chunk->next;
    return chunk;
}

void pool_free(Pool *p, void *ptr) {
    PoolChunk *chunk = ptr;
    chunk->next = p->free_list;
    p->free_list = chunk;
}
```

```ld
    RAM (rwx)  : ORIGIN = 0x20000000, LENGTH = 128K

        _bss_end = .;
```

### 4.2 无锁环形缓冲区（Lock-Free Ring Buffer）

**单生产者单消费者（SPSC）无锁队列，适用于中断上下文**：

```c
typedef struct {
    uint8_t *buffer;
    size_t capacity;
    _Atomic(size_t) head;  // 仅生产者写入
    _Atomic(size_t) tail;  // 仅消费者写入
} RingBuffer;

bool rb_write(RingBuffer *rb, const uint8_t *data, size_t len) {
    size_t current_tail = atomic_load_explicit(&rb->tail, memory_order_relaxed);
    size_t next_tail = (current_tail + len) % rb->capacity;

    // 检查满（预留 1 字节区分空/满）
    if (next_tail == atomic_load_explicit(&rb->head, memory_order_acquire))
        return false;  // 满

    // 写入数据（分两段处理回绕）
    size_t first_part = rb->capacity - current_tail;
    if (len <= first_part) {
        memcpy(rb->buffer + current_tail, data, len);
    } else {
        memcpy(rb->buffer + current_tail, data, first_part);
        memcpy(rb->buffer, data + first_part, len - first_part);
    }

    // 内存屏障：确保数据写入后才更新 tail
    atomic_store_explicit(&rb->tail, next_tail, memory_order_release);
    return true;
}
```

```c
size_t safe_truncate(const char *str, size_t max_bytes) {
    size_t len = 0;
    while (len < max_bytes && str[len]) {
        if ((str[len] & 0x80) == 0) len++;  // ASCII
        else if ((str[len] & 0xe0) == 0xc0) len += 2;  // 2-byte
        else if ((str[len] & 0xf0) == 0xe0) len += 3;  // 3-byte
```

```c
typedef struct {
    // MemTable：跳表（Skip List）或 B+ 树，内存中
    SkipList *memtable;
    size_t memtable_size;

    // SSTable：不可变有序文件，分层（Level 0-6）
    SSTable **levels[7];

void lsm_put(LSMDB *db, const char *key, const char *value) {
    // 先写 WAL（顺序 IO，fsync）
    wal_append(db->wal_fd, key, value);

    // 再写 MemTable（O(log n)）
    skiplist_insert(db->memtable, key, value);
    db->memtable_size += strlen(key) + strlen(value);

    // 触发刷盘（Minor Compaction）
    if (db->memtable_size > MEMTABLE_THRESHOLD) {
        SSTable *new_sst = sstable_create_from_memtable(db->memtable);
        level0_add(db, new_sst);
        skiplist_clear(db->memtable);
        db->memtable_size = 0;
        ftruncate(db->wal_fd, 0);  // 清空 WAL
    }
}
```

## 🗜️ 五、压缩算法实现（深度 B）

### 5.1 LZ77 滑动窗口压缩（ZIP/gzip 核心）

**基于哈希链的字符串匹配**：

```c
#define WINDOW_SIZE 32768  // 32KB 滑动窗口
#define LOOKAHEAD 258      // 最大匹配长度

typedef struct {
    uint8_t *window;
    int pos;
    // 哈希表加速匹配：hash -> 链表头
    int *head;
    int *prev;  // 冲突链
} LZ77Encoder;

void lz77_compress(LZ77Encoder *enc, const uint8_t *in, size_t len) {
    for (size_t i = 0; i < len; ) {
        // 哈希计算：3 字节 hash
        uint32_t hash = (in[i] << 16) | (in[i+1] << 8) | in[i+2];

        // 在窗口中查找最长匹配
        int best_len = 0, best_dist = 0;
        for (int j = enc->head[hash]; j != -1; j = enc->prev[j & (WINDOW_SIZE-1)]) {
            int dist = enc->pos - j;
            if (dist > WINDOW_SIZE) break;

            int match_len = compare(in+i, enc->window + (j & (WINDOW_SIZE-1)), LOOKAHEAD);
            if (match_len > best_len) {
                best_len = match_len;
                best_dist = dist;
            }
        }

        if (best_len >= 3) {
            emit_match(best_dist, best_len);  // 输出 (距离, 长度) 对
            i += best_len;
        } else {
            emit_literal(in[i]);  // 输出原字节
            i++;
        }

        // 更新哈希链
        enc->prev[enc->pos & (WINDOW_SIZE-1)] = enc->head[hash];
        enc->head[hash] = enc->pos;
        enc->window[enc->pos & (WINDOW_SIZE-1)] = in[i];
        enc->pos++;
    }
}
```

```c
// 计算默克尔根
void compute_merkle_root(uint8_t **tx_hashes, int n, uint8_t *root) {
    if (n == 0) return;
    if (n == 1) {
        memcpy(root, tx_hashes[0], HASH_SIZE);
        return;
    }

    uint8_t **next_level = malloc(((n + 1) / 2) * sizeof(uint8_t*));

    for (int i = 0; i < n; i += 2) {
        uint8_t *combined = malloc(HASH_SIZE * 2);
        memcpy(combined, tx_hashes[i], HASH_SIZE);

        if (i + 1 < n) {
            memcpy(combined + HASH_SIZE, tx_hashes[i+1], HASH_SIZE);
        } else {
            memcpy(combined + HASH_SIZE, tx_hashes[i], HASH_SIZE);  // 奇数复制最后一个
        }

        next_level[i/2] = malloc(HASH_SIZE);
        sha256(combined, HASH_SIZE * 2, next_level[i/2]);
        free(combined);
    }
```

```c
    DFU_STATE_IDLE,
    DFU_STATE_DNLOAD_SYNC,
    DFU_STATE_MANIFEST,
    DFU_STATE_ERROR
} DFUState;

void dfu_handle_request(USB_SETUP_PACKET *req, uint8_t *data) {
    switch (req->bRequest) {
        case DFU_DETACH:

            status.bwPollTimeout = flash_busy() ? 100 : 0;
            break;
    }
}
```

```c
    .rflags = 0x2,  // 保留位

ioctl(vcpu_fd, KVM_SET_REGS, &regs);

// 运行（VM Entry/Exit 循环）
while (1) {
    ioctl(vcpu_fd, KVM_RUN, 0);
    struct kvm_run *run = mmap(NULL, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, vcpu_fd, 0);

    switch (run->exit_reason) {
        case KVM_EXIT_IO:

                     run->io.size, run->io.direction);

        case KVM_EXIT_MMIO:
            // 处理内存映射 IO（如虚拟设备）
            break;
        case KVM_EXIT_SHUTDOWN:
```

```c
int xdp_drop_icmp(struct xdp_md *ctx) {
    void *data_end = (void *)(long)ctx->data_end;
    void *data = (void *)(long)ctx->data;

    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end) return XDP_DROP;

    if (eth->h_proto != __constant_htons(ETH_P_IP)) return XDP_PASS;

    struct iphdr *ip = data + sizeof(*eth);

        return XDP_DROP;
    }

    // 重定向到另一网卡（负载均衡）

    return XDP_PASS;
}
```
