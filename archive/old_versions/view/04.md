```c
void __attribute__((section(".locked_text"))) abs_control_loop(void) {
    // 物理操作 1：禁用中断（CLI，防止调度器抢占）
```

```c
// ELF：SYMBOL TABLE: 0000000000000000  0  NOTYPE  GLOBAL  DEFAULT  UND  global_counter
```

## 🏗️ 二、声明与定义（Declarations & Definitions）：物理存储分配

### 2.1 存储类别说明符（Storage Class Specifiers）的物理含义

| 语法机制 | 物理语义 | 物理存储位置 | 生命周期 |
|---------|---------|-------------|---------|
| `auto` | 栈帧自动分配 | 栈（RSP/EBP 相对寻址） | 块作用域（函数调用期间） |
| `static`（函数内） | 静态存储，内部链接 | `.data`（初始化）或 `.bss`（零初始化） | 程序生命周期 |

| `extern` | 外部链接，声明不分配 | 无分配，仅符号引用 | 依赖定义处 |
| `_Thread_local` | 线程私有存储（TLS） | `.tdata` 段，通过 FS/GS 段寄存器访问 | 线程生命周期 |
| `register` | 建议寄存器分配（C89），现代编译器忽略但禁止取地址 | CPU 寄存器（RAX, RBX...）或栈（若溢出） | 块作用域 |

**物理深度解析**：

```c
// 物理：TLS（Thread-Local Storage）实现机制
_Thread_local int thread_id;

// x86-64 Linux 物理实现：
// 访问通过 %fs 段寄存器（用户态 TLS 基址）
// mov %fs:0x10, %eax  （加载 thread_id）

// 初始化：动态链接器在创建 pthread 时，为每个线程分配 TLS 块（Block）
// 物理：clone() 系统调用设置 FS 基址（arch_prctl(ARCH_SET_FS)）
```

### 2.2 类型修饰符（Type Qualifiers）的物理控制

**`const`：只读权限的物理实现**

```c
const int ro_data = 100;
// 物理：放入 .rodata 段，页表标记为只读（PTE_U 清除或只读）
// 物理后果：写入触发 #PF（Page Fault），内核发送 SIGSEGV
```

**`volatile`：禁用优化的物理观测**

```c
volatile uint32_t *status_reg = (void*)0x40000000;
uint32_t val = *status_reg;  // 物理：必须生成 load 指令，即使刚读过
// 物理语义：对应内存映射 IO（MMIO），设备寄存器状态可能由外部硬件改变
// 编译器屏障：防止指令重排越过序列点
```

**`restrict`：内存别名分析的物理优化**

```c
void add(int *restrict a, int *restrict b, int n) {
    for(int i=0; i<n; i++) a[i] += b[i];
}
// 物理语义：承诺 a 和 b 指向的内存区域不相交（无别名）
// 物理优化：编译器可生成 SIMD 向量化代码（AVX-512），无运行时别名检查
// 物理后果：若违反，UB 导致错误向量化（缓存行冲突）
```

## 🧮 三、类型系统（Type System）：物理数据编码

### 3.1 整数类型的物理位模式（Integer Representation）

**语法**：`signed/unsigned char/short/int/long/long long`

**物理语义**：

- **补码（Two's Complement）**：现代计算机物理实现的标准，利用溢出回绕特性
- **字节序（Endianness）**：物理存储地址与数值权重的映射关系
  - Little Endian（x86）：低地址存低位（LSB），符合硬件加法器从低位开始的物理流程
  - Big Endian（网络/MIPS）：高地址存低位，符合人类阅读习惯

```c
uint32_t x = 0x12345678;
// Little Endian 物理内存（地址递增）：
// 0x78 0x56 0x34 0x12

// 物理操作：字节提取（Bit-field 或 Union）
uint8_t low_byte = *(uint8_t*)&x;  // 物理：读取最低地址字节，得 0x78（LE）
```

### 3.2 浮点类型的物理电路语义（IEEE 754）

**语法**：`float`（binary32）、`double`（binary64）、`long double`（80-bit x87 或 128-bit IEEE 754-2008）

**物理语义映射**：

```
符号位 Sign（S）| 指数 Exponent（E）| 尾数 Mantissa（M）
物理电路：
  S → 符号处理单元（异或门处理乘法符号）
  E → 指数加法器（带偏置 127/1023 减法）
  M → 尾数乘法器（Booth 编码 Wallace Tree）

特殊值物理意义：
  E=全0, M=0：±0（零检测电路）
  E=全1, M=0：±Inf（溢出检测）
  E=全1, M≠0：NaN（Not a Number，异常传播）
```

**物理控制**：

```c
// FPU 控制字（x87 FPUCW）的 C 接口
fesetround(FE_UPWARD);  // 物理：设置浮点单元舍入模式（向正无穷）
// 物理影响：RC（Rounding Control）字段（第 10-11 位）
// 00=就近，01=向负无穷，10=向正无穷，11=向零截断
```

### 3.3 数组的物理连续性与缓存语义（Array as Physical Contiguity）

**语法**：`T arr[N]`

**物理语义**：

- **空间局部性（Spatial Locality）**：元素物理连续，缓存行（64B）可预取多个元素
- **基址+偏移寻址**：`arr[i]` 物理实现为 `*(arr + i*sizeof(T))`，利用地址生成单元（AGU）

**物理优化（定制能力）**：

```c
// 物理：行优先 vs 列优先的缓存效应
int matrix[1024][1024];

// 缓存友好（空间局部性）：
for(int i=0; i<1024; i++)
    for(int j=0; j<1024; j++)
        matrix[i][j]++;  // 物理：顺序访问，L1 D-cache 命中率高

// 缓存不友好（跨步访问）：
for(int j=0; j<1024; j++)
    for(int i=0; i<1024; i++)
        matrix[i][j]++;  // 物理：每次跳过 4KB（1024*4），频繁 cache miss
```

### 3.4 结构体的物理布局与对齐（Struct Layout as Physical Packing）

**语法**：`struct { ... }`

**物理语义**：

- **内存对齐（Alignment）**：满足硬件总线宽度（32/64/128-bit）的访问效率
- **填充（Padding）**：编译器插入的空字节，满足成员对齐要求

```c
struct Bad {
    char c;   // offset 0
    // 3 bytes padding（物理浪费，满足 int 的 4-byte 对齐）
    int i;    // offset 4
    short s;  // offset 8
    // 2 bytes padding（满足结构体整体对齐）
}; // size = 12

struct Good {
    int i;    // offset 0
    short s;  // offset 4
    char c;   // offset 6
    char pad; // offset 7（显式填充）
}; // size = 8（物理内存节省 33%）

// 物理控制：
struct __attribute__((packed)) Unaligned {
    char c;
    int i;  // 物理：可能跨越两个 32-bit 字，需要两次内存访问（性能惩罚）
};
```

### 3.5 位域（Bit-fields）：物理寄存器精确映射

**语法**：`unsigned field : N;`

**物理语义**：将多个逻辑变量打包到同一物理内存单元（通常是 32/64-bit 字），精确匹配硬件寄存器布局。

```c
// 物理：GPIO 控制寄存器（32-bit）
struct GPIO_CR {
    unsigned mode   : 2;  // bit 0-1（物理引脚模式）
    unsigned speed  : 2;  // bit 2-3（物理驱动速度）
    unsigned pupd   : 2;  // bit 4-5（物理上拉/下拉）
    unsigned : 26;        // bit 6-31（保留，物理上必须写 0）
};

// 物理访问：
struct GPIO_CR *cr = (void*)0x40020000;
cr->mode = 0x3;  // 物理：生成 32-bit 写，仅修改 bit 0-1，保留其他位
// 实际物理：读-修改-写（Read-Modify-Write）或 位带（Bit-banding，如果硬件支持）
```

**物理陷阱**：位域的 signedness 未指定（int vs unsigned），影响符号扩展；跨边界位域（>32bit）的物理布局实现定义。

### 3.6 柔性数组成员（FAM）：物理内存的单一分配

**语法**：`struct { ... T arr[]; };`

**物理语义**：变长结构体，最后成员不占空间（size=0），允许运行时确定数组大小，**物理上连续分配**。

```c
struct Packet {
    uint32_t header;
    uint16_t len;
    uint8_t  payload[];  // 物理：不占用 sizeof(struct Packet) 的空间
};

// 物理分配（单次 malloc，缓存友好）：
struct Packet *p = malloc(sizeof(*p) + data_len);
// 物理布局：[header(4)][len(2)][payload(data_len)] 连续无间隙

// 对比指针版本（非连续，二次寻址）：
struct BadPacket {
    uint32_t header;
    uint8_t  *payload;  // 物理：8 字节指针 + 堆上另一块内存
};
```

## 🎯 四、指针与内存寻址（Pointers & Addressing）

### 4.1 指针类型的物理语义（Pointer as Physical Address）

**语法**：`T *p`

**物理语义**：

- **虚拟地址**：进程地址空间中的索引（48/57/64-bit）
- **页表遍历**：硬件 MMU 将虚拟地址翻译为物理地址（DRAM 行/列）
- **寻址模式**：`*(p + i)` 利用 CPU 地址生成单元（AGU）的**缩放索引寻址**（scaled indexed addressing）

```c
int *p = arr;
*(p + 4);  // 物理：p + 4*sizeof(int)，AGU 自动计算，无需额外指令（x86 LEA）
```

### 4.2 函数指针的物理实现（Function Pointer as Code Address）

**语法**：`void (*fp)(int)`

**物理语义**：

- **代码段地址**：指向 `.text` 段的机器码入口点
- **间接跳转**：`call *%rax`（x86），修改 PC（程序计数器）为函数地址
- **PLT/GOT**：动态链接时，函数指针指向过程链接表（Procedure Linkage Table），通过全局偏移表（GOT）解析实际地址

```c
void (*fp)(void) = &target_func;
fp();  // 物理：间接调用（Indirect Call），分支目标缓冲区（BTB）预测目标地址
```

### 4.3 void*与字节级物理访问（void* as Untyped Memory）

**语法**：`void *`

**物理语义**：**原始字节序列**的句柄，对应物理存储的最低抽象（无类型解释）。`memcpy`、`memset` 操作的是物理字节。

```c
void *memcpy(void *dest, const void *src, size_t n);
// 物理：逐字节复制，或利用 SIMD（AVX-512）批量复制（64 bytes/指令）
// 物理约束：若 dest/src 重叠，UB（应使用 memmove）
```

## ⚙️ 五、表达式与运算符（Expressions & Operators）：物理计算单元映射

### 5.1 算术运算符的物理电路（Arithmetic as ALU Operations）

**语法**：`+ - * / %`

**物理语义映射**：

- **加法/减法**：ALU 整数加法器（行波进位或超前进位），设置标志寄存器（ZF/SF/CF/OF）
- **乘法**： booth 算法乘法器（多周期）或 SIMD 并行乘法（单周期 4/8 整数）
- **除法**：慢速（20-40 周期），SRT 算法或牛顿迭代；现代 CPU 支持硬件除法但延迟高
- **取模**：通常作为除法副产物（余数在 EDX/RDX 寄存器）

```c
int a = 10, b = 3;
int c = a / b;  // 物理：IDIV 指令，商在 EAX，余数在 EDX
int d = a % b;  // 物理：直接读取 EDX，无需第二次除法
```

### 5.2 位运算符的物理门级操作（Bitwise as Logic Gates）

**语法**：`& | ^ ~ << >>`

**物理语义**：

- **AND/OR/XOR/NOT**：直接对应 CMOS 逻辑门（与门、或门、异或门、非门），单周期延迟
- **移位**：桶形移位器（Barrel Shifter），物理上通过多级多路选择器实现，对数级延迟（log2(n)）

```c
uint32_t x = 1;
x << 5;   // 物理：桶形移位器，选择第 5 位输出线，相当于乘以 32（2^5）
x >> 5;   // 物理：逻辑右移（无符号）或算术右移（有符号，符号位扩展）
```

### 5.3 逻辑运算符的物理短路（Logical Operators as Short-circuit Control）

**语法**：`&& || !`

**物理语义**：

- **短路求值**：物理上对应**条件跳转**（conditional branch），而非位运算
- **分支预测**：CPU 预测短路路径，预测失败时刷新流水线（15-20 周期惩罚）

```c
if (p != NULL && *p > 0)  // 物理：先测试 p，若为假则跳转（跳过 *p），避免 #PF
```

## 🔄 六、控制流（Control Flow）：物理程序计数器操作

### 6.1 条件语句的物理分支（If-Else as Conditional Jump）

**语法**：`if (cond) { ... } else { ... }`

**物理语义**：

- **比较指令**：`cmp` 设置标志寄存器
- **条件跳转**：`je`（jump if equal）、`jg`（jump if greater）等，修改 PC（程序计数器）
- **分支预测单元（BPU）**：历史缓冲区预测跳转方向，推测执行（Speculative Execution）

```c
if (x > 0) {
    // 物理：若 x>0 预测正确，顺序执行；否则刷新流水线，跳转到 else
} else {
    // 物理：冷路径代码可能预取但不确定执行
}
```

### 6.2 循环的物理优化（Loops as Hardware Loop Support）

**语法**：`for`, `while`, `do-while`

**物理语义**：

- **循环展开（Unrolling）**：编译器优化，减少分支指令数量，增加指令级并行（ILP）
- **硬件循环缓冲（Loop Buffer）**：ARM/Intel 的 LSD（Loop Stream Detector），小循环缓存在微操作缓存，不经过取指单元

```c
// 物理优化：编译器展开 4 次，减少 75% 的分支指令
for(int i=0; i<100; i++) sum += a[i];
// 生成：
// sum += a[0]; sum += a[1]; sum += a[2]; sum += a[3]; i+=4;
// 分支指令从 100 次减至 25 次
```

### 6.3 Switch 的物理跳转表（Switch as Jump Table）

**语法**：`switch (x) { case 1: ... case 2: ... }`

**物理语义**：

- **稀疏值**：转换为 if-else 链（顺序比较）
- **稠密值**：**跳转表（Jump Table）**，物理上是一个函数指针数组，通过基址+偏移直接跳转（O(1)）

```c
switch(opcode) {
    case 0: add(); break;
    case 1: sub(); break;
    case 2: mul(); break;
}
// 物理（稠密时）：
// jmp [jump_table + opcode * 8]  （8 字节地址）
// jump_table: .quad add, sub, mul...
```

### 6.4 Goto 与标签的物理无条件跳转（Goto as Unconditional Jump）

**语法**：`goto label;`

**物理语义**：`jmp` 指令，直接修改 PC，无返回地址压栈。用于错误处理、状态机实现。

```c
error:
    cleanup();  // 物理：直接跳转，无调用开销（不修改栈）
```

## 🧩 七、函数机制（Function Mechanism）：调用约定的物理实现

### 7.1 函数调用的物理栈帧（Function Call as Stack Frame Manipulation）

**语法**：`T func(Args...) { ... return val; }`

**物理语义**（System V AMD64 ABI）：

```
调用者（Caller）：
  1. 参数 1-6 放入寄存器（RDI, RSI, RDX, RCX, R8, R9），剩余压栈
  2. CALL 指令：压栈返回地址（RIP），跳转到函数
  3. 被调用者返回后，调用者清理栈参数（Caller cleanup）

被调用者（Callee）：
  1. PUSH RBP（保存旧栈帧基址）
  2. MOV RSP, RBP（建立新栈帧）
  3. SUB RSP, N（为局部变量分配栈空间，物理上只是调整 RSP 寄存器）
  4. 执行函数体
  5. MOV RBP, RSP（恢复栈）
  6. POP RBP
  7. RET（从栈弹出返回地址到 RIP）
```

**物理定制**：`__attribute__((naked))` 完全控制汇编序言/尾声。

### 7.2 内联函数的物理消除（Inline as Zero-Cost Abstraction）

**语法**：`static inline`

**物理语义**：编译器将函数体直接插入调用点，**消除 call/ret 指令**，消除栈帧建立开销。物理上就是复制代码。

```c
static inline int max(int a, int b) { return a > b ? a : b; }
// 物理：调用处直接生成 cmp/cmov 指令序列，无函数调用开销
```

### 7.3 可变参数列表的物理栈遍历（Variadic Functions as Stack Walking）

**语法**：`...` 与 `va_list`

**物理语义**：通过栈地址递增访问参数，依赖调用约定（参数从右向左压栈）。

```c
void printf(const char *fmt, ...) {
    va_list ap;
    va_start(ap, fmt);  // 物理：ap = &fmt + sizeof(fmt)，指向第一个变参的栈地址
    int x = va_arg(ap, int);  // 物理：*(int*)ap; ap += sizeof(int)
}
```

## 🧵 八、并发与同步（Concurrency）：物理时间的协调

### 8.1 信号量的物理原子性（Semaphore as Atomic Counter）

**语法**：`sem_wait`, `sem_post`（POSIX）或 `_Atomic`（C11）

**物理语义**：

- **原子操作**：`lock xadd`（x86）或 `ldrex/strex`（ARM），总线锁定或独占监视器
- **阻塞**：操作系统将线程状态改为 `TASK_INTERRUPTIBLE`，从运行队列移除，切换上下文（保存/恢复寄存器）

### 8.2 内存屏障的物理排序（Memory Barriers as Execution Ordering）

**语法**：`atomic_thread_fence(memory_order_seq_cst)`

**物理语义**：

- **CPU 屏障**：
  - `mfence`：确保所有 load/store 全局可见（刷新 Store Buffer）
  - `sfence`：Store Fence，确保写操作顺序
  - `lfence`：Load Fence，确保读操作顺序

**物理意义**：在乱序执行（Out-of-Order Execution）的 CPU 上，强制内存操作按程序顺序提交到内存子系统。

## 🎭 十、未定义行为（UB）的物理本质（UB as Physical Indeterminacy）

**核心洞察**：C 的 UB 不是"错误"，而是**物理系统的非确定性**在语言层面的体现。

| UB 类型 | 物理语义 | 物理后果 |
|---------|---------|---------|
| **有符号溢出** | 物理 ALU 溢出标志（OF）被忽略 | 补码回绕（x86）或陷阱（DSP） |
| **空指针解引用** | 访问页表未映射地址（PTE Present=0） | #PF（Page Fault），SIGSEGV |
| **数组越界** | 访问未分配物理页或相邻变量 | 数据损坏、安全漏洞、崩溃 |
| **数据竞争** | 多核缓存不一致（MESI 状态冲突） | 读取撕裂（Tear）、不可重复读 |

## 🔬 一、编译器 IR（中间表示）的物理语义（The Physics of Compiler IR）

### 1.1 SSA（静态单赋值）与物理寄存器生命的同构

**LLVM IR 示例**：

```llvm
%add = add i32 %a, %b
%mul = mul i32 %add, %c
```

**物理语义映射**：

```
SSA 形式（逻辑层）：
  %add ──(虚拟寄存器)──► 唯一赋值，无限存活期

物理实现（硅片层）：
  %add ──(物理寄存器分配)──► EAX (live range: [add指令, mul指令])
  %mul ──(物理寄存器分配)──► EAX (reuse，因为 %add 已死)

关键洞察：SSA 的 "def-use 链" 精确对应物理寄存器的 "生命期（Live Range）"
```

**寄存器分配即图着色（Chaitin 算法）的物理意义**：

- **干扰图（Interference Graph）的节点**：SSA 变量（虚拟值）
- **边**：两变量同时存活（ overlapping live ranges ），对应物理寄存器冲突
- **颜色**：物理寄存器（RAX, RBX, RCX... 有限集合）
- **着色失败（Spill）**：变量溢出到栈（物理：从寄存器降级到内存，访问延迟 +3 周期）

### 1.2 指令调度（Instruction Scheduling）的物理时间

**LLVM 调度单元与 CPU 流水线**：

```c
// C 源代码
a = b + c;
d = a * e;

// LLVM IR（未调度）
%1 = load i32* %b
%2 = load i32* %c
%3 = add i32 %1, %2
store i32 %3, i32* %a
%4 = load i32* %e      // 可以乱序提前！
%5 = mul i32 %3, %4

// 物理调度（针对 x86 Skylake 端口约束）
// 端口 0,1,5,6 可执行 ALU 操作
// 负载（Load）使用端口 2,3
Cycle 0: movl (%rbx), %eax    ; 端口 2, 加载 b
Cycle 0: movl (%rcx), %edx    ; 端口 3, 加载 c（并行）
Cycle 1: addl %edx, %eax      ; 端口 0, 加法（依赖加载结果）
Cycle 1: movl (%r8), %ecx     ; 端口 2, 加载 e（乱序，与加法并行）
Cycle 2: imull %ecx, %eax     ; 端口 1, 乘法（3 周期延迟）
```

**物理约束的数学表达**：

- **数据依赖图（DAG）**：边表示 " happens-before "（物理因果律）
- **资源约束**：每个周期每个功能单元（F0, F1...）只能执行一条指令
- **目标**：最小化总周期数（ makespan ）= 物理执行时间

### 1.3 向量化（Vectorization）的物理数据并行

**SLP（Superword Level Parallelism）的物理意义**：

```c
// C 标量代码
a[0] = b[0] + c[0];
a[1] = b[1] + c[1];
a[2] = b[2] + c[2];
a[3] = b[3] + c[3];

// LLVM 向量化后（IR 层）
%vec.load.b = load <4 x i32>, <4 x i32>* %b
%vec.load.c = load <4 x i32>, <4 x i32>* %c
%vec.add = add <4 x i32> %vec.load.b, %vec.load.c
store <4 x i32> %vec.add, <4 x i32>* %a

// 物理实现（AVX-512）
vmovdqa (%rbx), %zmm0      ; 512-bit 加载（64 字节）
vmovdqa (%rcx), %zmm1      ; 512-bit 加载
vpaddd  %zmm1, %zmm0, %zmm0 ; 16 个整数并行加（物理：单周期，16 ALU 并行）
vmovdqa %zmm0, (%rax)      ; 512-bit 存储
```

**物理洞察**：向量化是**空间并行性**（Spatial Parallelism）的显式表达，利用 SIMD 单元的物理数据通路宽度（512-bit ZMM 寄存器 = 16 个 lane）。

## 🔗 三、位置无关代码（PIC/PIE）的数学拓扑（The Topology of Position Independent Code）

### 3.1 虚拟地址的纤维丛结构（Virtual Address as Fiber Bundle）

**数学结构**：

- **基空间（Base Space）**：共享库的**逻辑偏移**（0x0 → 符号）
- **纤维（Fiber）**：加载时的**实际虚拟地址**（由 ASLR 随机化）
- **截面（Section）**：运行时链接器建立的**映射函数** `f: 逻辑偏移 → 虚拟地址`

**C 与汇编的对应**：

```c
// C：全局变量访问（PIC 模式）
extern int global;
void f() { global = 1; }

// 汇编（位置无关）：
// 1. 通过 GOT（Global Offset Table）间接寻址
movq global@GOTPCREL(%rip), %rax  ; RIP-relative 加载 GOT 条目地址
movl $1, (%rax)                   ; 通过指针写入

// 物理意义：代码段（.text）是只读且位置无关的，
// 实际地址 = 加载基址 + 编译时偏移（Relative addressing 的物理实现）
```

### 3.2 延迟绑定（Lazy Binding）的物理优化

**PLT（Procedure Linkage Table）的状态机**：

```
状态 0（未解析）：
  PLT[n] → 跳转到 PLT[0] → 调用 _dl_runtime_resolve
  物理：首次调用触发动态链接器，解析符号，回填 GOT

状态 1（已解析）：
  PLT[n] → 直接跳转到目标函数（通过 GOT 条目）
  物理：后续调用无解析开销，直接跳转

数学：这是从 "符号名（字符串）" 到 "物理地址（整数）" 的惰性求值（Lazy Evaluation）
```

## ⚠️ 四、异常处理的物理栈展开（Exception Handling as Physical Stack Unwinding）

### 4.1 setjmp/longjmp 的物理寄存器保存

**物理状态捕获**：

```c
typedef struct __jmp_buf {
    uint64_t regs[8];   // RBX, RBP, R12-R15（被调用者保存寄存器）
    uint64_t rsp;       // 栈指针（物理内存位置）
    uint64_t rip;       // 指令指针（下一条指令地址）
    uint64_t mxcsr;     // SSE 控制/状态寄存器（物理：浮点异常标志）
    uint64_t fp_regs[8]; // x87 FPU / XMM 寄存器（物理：浮点栈）
} jmp_buf[1];

int setjmp(jmp_buf env) {
    // 物理：将当前 CPU 寄存器状态保存到内存（栈或静态区）
    asm volatile(
        "movq %%rbx, (%rdi)\n\t"
        "movq %%rbp, 8(%rdi)\n\t"
        "movq %%r12, 16(%rdi)\n\t"
        // ...
        "leaq 8(%%rsp), %%rax\n\t"  // 保存调用者的 RSP
        "movq %%rax, 64(%rdi)\n\t"
        "movq (%%rsp), %%rax\n\t"   // 保存返回地址（RIP）
        "movq %%rax, 72(%rdi)\n\t"
    );
}
```

**物理恢复**：

```c
void longjmp(jmp_buf env, int val) {
    // 物理：从内存恢复寄存器，直接跳转到 saved RIP
    // 不经过正常函数返回路径（绕过栈帧析构）
    asm volatile(
        "movq (%rdi), %%rbx\n\t"
        "movq 64(%rdi), %%rsp\n\t"  // 直接切换栈指针！
        "movq 72(%rdi), %%r8\n\t"   // 目标 RIP
        "movl %%esi, %%eax\n\t"     // 返回值
        "jmp *%%r8\n\t"             // 非 ret 跳转，直接修改 PC
    );
}
```

**物理后果**：

- **栈切换**：`longjmp` 可以跨栈帧（从深层调用跳回早期栈帧），物理上丢弃中间栈帧（不调用析构函数，C 中无自动析构，但 C++ 中这是灾难）。
- **寄存器恢复**：浮点状态（MXCSR）恢复，物理上清除异常标志。

```c
    .event = "INSTRUCTIONS_RETIRED",      // 物理：指令退役（Retirement）计数
    .event = "L1D_CACHE_MISSES",          // 物理：L1 数据缓存未命中（行填充）
    .event = "BRANCH_MISSES_PREDICTED",   // 物理：分支预测失败（流水线刷新）
    .event = "CYCLE_ACTIVITY.STALLS_L2_MISS" // 物理：因 L2 未命中导致的停顿周期
});

// C 代码执行...

// 物理：读取 MSR（Model Specific Register）获取计数值
```

**物理洞察**：

- **L1_MISS** 高 → 缓存不友好访问模式（跨步、随机访问）
- **BRANCH_MISS** 高 → 难以预测的控制流（建议改用 cmov 或查找表）
- **UOPS_ISSUED.ANY** vs **INSTRUCTIONS_RETIRED** → 微操作融合效率

## 🔄 六、一致性协议与 C 内存模型的物理实现（Coherence Protocols）

### 6.1 MESI 状态机在 C 代码中的显式触发

**C 代码触发状态转换**：

```c
// 初始：缓存行在 Shared 状态（S）
_Atomic(int) shared_var;

// Core 0：读取（保持 S）
int x = atomic_load_explicit(&shared_var, memory_order_relaxed);
// 物理：总线事务 BusRd，其他核心持有副本，状态保持 S

// Core 1：写入（触发状态转换）
atomic_store_explicit(&shared_var, 42, memory_order_release);
// 物理：
// 1. Core 1 发送 BusRdX（或 BusUpgr）总线请求
// 2. Core 0 的缓存行从 S → I（Invalid）
// 3. Core 1 从 M（Modified）获取所有权
// 4. 后续 Core 0 读取将触发 BusRd，数据从 Core 1 写回（Writeback）
```

### 6.2 伪共享（False Sharing）的物理惩罚

