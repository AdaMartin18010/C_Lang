# 微程序控制器 (Microprogrammed Control Unit)

## 1. 概述

微程序控制器是CPU控制单元的另一种实现方式，与硬布线控制不同，它使用存储在控制存储器(Control Store)中的微程序来生成控制信号。
每个机器指令对应一段微程序，由若干条微指令组成。

### 3.2 Verilog实现

```verilog
// ============================================================================
// 控制存储器 (Control Store) - Verilog实现
// ============================================================================

module control_store (
    input  wire        clk,
    input  wire [7:0]  address,        // 微地址 (256个微指令)
    output reg  [31:0] microinstruction // 微指令输出
);

    //==========================================================================
    // 微指令格式定义
    //==========================================================================
    // [31:24] - 控制字段 (8位控制信号)
    // [23:20] - 条件选择 (4位条件码)
    // [19:16] - 下地址控制 (4位)
    // [15:0]  - 下地址/立即数 (16位)

    // 控制信号位定义
    localparam CTRL_MEM_READ  = 0;    // 存储器读
    localparam CTRL_MEM_WRITE = 1;    // 存储器写
    localparam CTRL_IR_WRITE  = 2;    // IR写使能
    localparam CTRL_PC_WRITE  = 3;    // PC写使能
    localparam CTRL_REG_WRITE = 4;    // 寄存器写
    localparam CTRL_ALU_SRC   = 5;    // ALU操作数来源
    localparam CTRL_BRANCH    = 6;    // 分支指令
    localparam CTRL_JUMP      = 7;    // 跳转指令

    //==========================================================================
    // 微地址定义
    //==========================================================================
    // 取指序列
    localparam [7:0]
        IF1 = 8'h00,   // PC → 地址总线
        IF2 = 8'h01,   // 读存储器
        IF3 = 8'h02,   // IR ← 数据, PC+4

        // R-Type 微程序
        R_ID  = 8'h10,   // 译码/读寄存器
        R_EX  = 8'h11,   // 执行
        R_WB  = 8'h12,   // 写回

        // I-Type ALU 微程序
        I_ID  = 8'h20,   // 译码/读寄存器
        I_EX  = 8'h21,   // 执行
        I_WB  = 8'h22,   // 写回

        // Load 微程序
        L_ID  = 8'h30,   // 译码
        L_EX  = 8'h31,   // 地址计算
        L_RD  = 8'h32,   // 读存储器
        L_WB  = 8'h33,   // 写回

        // Store 微程序
        S_ID  = 8'h40,   // 译码
        S_EX  = 8'h41,   // 地址计算
        S_WR  = 8'h42,   // 写存储器

        // Branch 微程序
        B_ID  = 8'h50,   // 译码
        B_EX  = 8'h51,   // 比较/分支

        // Jump 微程序
        J_ID  = 8'h60,   // 译码
        J_EX  = 8'h61,   // 跳转执行
        J_WB  = 8'h62,   // 链接寄存器写回

        // 返回取指
        RET_IF = 8'hFF;  // 返回取指

    //==========================================================================
    // 控制存储器 (ROM)
    //==========================================================================
    reg [31:0] control_memory [0:255];

    //==========================================================================
    // 初始化微程序
    //==========================================================================
    initial begin
        // 初始化所有微指令为NOP
        integer i;
        for (i = 0; i < 256; i = i + 1) begin
            control_memory[i] = 32'h0000_00FF;  // NOP, 跳转到FF
        end

        //---------------------------------------------------------------------
        // 取指序列 (所有指令共享)
        //---------------------------------------------------------------------
        // IF1: PC → 地址总线, PC+4
        // 控制: PC输出到地址总线, ALU做PC+4
        control_memory[IF1] = {
            8'b0000_1000,   // [31:24] ctrl: PC_WRITE=1
            4'b0000,        // [23:20] cond: 无条件
            4'b0001,        // [19:16] next: 顺序执行
            8'h00, IF2      // [15:0]  next_addr: IF2
        };

        // IF2: 读存储器
        control_memory[IF2] = {
            8'b0000_0111,   // [31:24] ctrl: MEM_READ=1, IR_WRITE=1, PC_WRITE=1
            4'b0000,        // [23:20] cond: 无条件
            4'b0001,        // [19:16] next: 顺序执行
            8'h00, IF3      // [15:0]  next_addr: IF3
        };

        // IF3: 指令译码, 根据Opcode跳转
        control_memory[IF3] = {
            8'b0000_0000,   // [31:24] ctrl: 无控制信号
            4'b1111,        // [23:20] cond: 使用映射ROM
            4'b0010,        // [19:16] next: 映射跳转
            8'h00, 8'h00    // [15:0]  由映射ROM提供
        };

        //---------------------------------------------------------------------
        // R-Type 微程序
        //---------------------------------------------------------------------
        // R_ID: 读寄存器 rs1, rs2
        control_memory[R_ID] = {
            8'b0000_0000,   // 读寄存器是默认操作
            4'b0000,        // 无条件
            4'b0001,        // 顺序
            8'h00, R_EX
        };

        // R_EX: ALU运算 (R-type)
        control_memory[R_EX] = {
            8'b0000_0000,   // ALU操作由funct控制
            4'b0000,
            4'b0001,
            8'h00, R_WB
        };

        // R_WB: 写回寄存器 rd
        control_memory[R_WB] = {
            8'b0001_0000,   // REG_WRITE=1
            4'b0000,
            4'b0011,        // 跳转到取指
            8'h00, IF1
        };

        //---------------------------------------------------------------------
        // I-Type ALU 微程序
        //---------------------------------------------------------------------
        // I_ID: 读寄存器 rs1, 生成立即数
        control_memory[I_ID] = {
            8'b0010_0000,   // ALU_SRC=1 (选择立即数)
            4'b0000,
            4'b0001,
            8'h00, I_EX
        };

        // I_EX: ALU立即数运算
        control_memory[I_EX] = {
            8'b0010_0000,   // ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, I_WB
        };

        // I_WB: 写回寄存器
        control_memory[I_WB] = {
            8'b0011_0000,   // REG_WRITE=1, ALU_SRC=1
            4'b0000,
            4'b0011,
            8'h00, IF1
        };

        //---------------------------------------------------------------------
        // Load 微程序
        //---------------------------------------------------------------------
        // L_ID: 读基址寄存器
        control_memory[L_ID] = {
            8'b0010_0000,   // ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, L_EX
        };

        // L_EX: 地址计算 (基址 + 偏移)
        control_memory[L_EX] = {
            8'b0010_0000,   // ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, L_RD
        };

        // L_RD: 读存储器
        control_memory[L_RD] = {
            8'b0000_0001,   // MEM_READ=1
            4'b0000,
            4'b0001,
            8'h00, L_WB
        };

        // L_WB: 写回寄存器
        control_memory[L_WB] = {
            8'b0001_0001,   // REG_WRITE=1, MEM_READ=1 (数据已读出)
            4'b0000,
            4'b0011,
            8'h00, IF1
        };

        //---------------------------------------------------------------------
        // Store 微程序
        //---------------------------------------------------------------------
        // S_ID: 读基址和数据寄存器
        control_memory[S_ID] = {
            8'b0010_0000,   // ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, S_EX
        };

        // S_EX: 地址计算
        control_memory[S_EX] = {
            8'b0010_0000,   // ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, S_WR
        };

        // S_WR: 写存储器
        control_memory[S_WR] = {
            8'b0000_0010,   // MEM_WRITE=1
            4'b0000,
            4'b0011,
            8'h00, IF1
        };

        //---------------------------------------------------------------------
        // Branch 微程序
        //---------------------------------------------------------------------
        // B_ID: 读比较寄存器
        control_memory[B_ID] = {
            8'b0000_0000,
            4'b0000,
            4'b0001,
            8'h00, B_EX
        };

        // B_EX: 比较并分支
        control_memory[B_EX] = {
            8'b0100_0000,   // BRANCH=1
            4'b0001,        // 条件: ALU zero标志
            4'b0100,        // 条件跳转
            8'h00, IF1      // 不跳转时返回取指
        };

        //---------------------------------------------------------------------
        // Jump 微程序
        //---------------------------------------------------------------------
        // J_ID: 准备跳转
        control_memory[J_ID] = {
            8'b1010_0000,   // JUMP=1, ALU_SRC=1
            4'b0000,
            4'b0001,
            8'h00, J_EX
        };

        // J_EX: 执行跳转
        control_memory[J_EX] = {
            8'b1000_1000,   // JUMP=1, PC_WRITE=1
            4'b0000,
            4'b0001,
            8'h00, J_WB
        };

        // J_WB: 链接寄存器写回
        control_memory[J_WB] = {
            8'b1001_0000,   // REG_WRITE=1, JUMP=1
            4'b0000,
            4'b0011,
            8'h00, IF1
        };

        //---------------------------------------------------------------------
        // 返回取指
        //---------------------------------------------------------------------
        control_memory[RET_IF] = {
            8'b0000_0000,
            4'b0000,
            4'b0011,
            8'h00, IF1
        };
    end

    //==========================================================================
    // 读取微指令 (同步读)
    //==========================================================================
    always @(posedge clk) begin
        microinstruction <= control_memory[address];
    end

endmodule
```

### 4.2 具体微指令格式设计

```verilog
// ============================================================================
// 微指令格式定义与译码模块
// ============================================================================

module microinstruction_decoder (
    input  wire [31:0] microinstruction,

    // 控制信号输出
    output wire [3:0]  alu_op,         // ALU操作码
    output wire        alu_src_a,      // ALU A操作数来源
    output wire [1:0]  alu_src_b,      // ALU B操作数来源
    output wire        reg_write,      // 寄存器写使能
    output wire [1:0]  reg_src,        // 寄存器数据来源
    output wire        mem_read,       // 存储器读
    output wire        mem_write,      // 存储器写
    output wire [1:0]  mem_size,       // 存储器访问大小
    output wire        ir_write,       // IR写使能
    output wire        pc_write,       // PC写使能
    output wire [1:0]  pc_src,         // PC来源
    output wire        branch,         // 分支指令
    output wire        jump,           // 跳转指令

    // 时序控制
    output wire [3:0]  condition,      // 分支条件
    output wire [1:0]  next_ctrl,      // 下地址控制
    output wire [7:0]  next_addr       // 下地址
);

    //==========================================================================
    // 微指令字段提取
    //==========================================================================
    // [31:24] - 直接控制字段
    // [23:20] - ALU控制
    // [19:16] - 寄存器控制
    // [15:12] - 存储器控制
    // [11:8]  - PC控制
    // [7:4]   - 时序控制
    // [3:0]   - 条件/下地址

    wire [7:0]  ctrl_direct = microinstruction[31:24];
    wire [3:0]  alu_ctrl    = microinstruction[23:20];
    wire [3:0]  reg_ctrl    = microinstruction[19:16];
    wire [3:0]  mem_ctrl    = microinstruction[15:12];
    wire [3:0]  pc_ctrl     = microinstruction[11:8];
    wire [3:0]  seq_ctrl    = microinstruction[7:4];
    wire [3:0]  cond_addr   = microinstruction[3:0];

    //==========================================================================
    // 直接控制信号译码
    //==========================================================================
    assign reg_write  = ctrl_direct[7];
    assign mem_read   = ctrl_direct[6];
    assign mem_write  = ctrl_direct[5];
    assign ir_write   = ctrl_direct[4];
    assign pc_write   = ctrl_direct[3];
    assign alu_src_a  = ctrl_direct[2];      // 0=PC, 1=寄存器A
    assign branch     = ctrl_direct[1];
    assign jump       = ctrl_direct[0];

    //==========================================================================
    // ALU控制译码
    //==========================================================================
    assign alu_op = alu_ctrl;
    // alu_ctrl编码:
    // 0000 = ADD
    // 0001 = SUB
    // 0010 = AND
    // 0011 = OR
    // 0100 = XOR
    // 0101 = SLL
    // 0110 = SRL
    // 0111 = SRA
    // 1000 = SLT
    // 1001 = SLTU
    // 1111 = NOP

    // ALU B操作数来源
    assign alu_src_b = alu_ctrl[3:2];
    // 00 = 寄存器B
    // 01 = 常数4
    // 10 = 立即数
    // 11 = 0

    //==========================================================================
    // 寄存器控制译码
    //==========================================================================
    assign reg_src = reg_ctrl[1:0];
    // 00 = ALU结果
    // 01 = 存储器数据
    // 10 = PC+4
    // 11 = 立即数

    //==========================================================================
    // 存储器控制译码
    //==========================================================================
    assign mem_size = mem_ctrl[1:0];
    // 00 = 字节
    // 01 = 半字
    // 10 = 字
    // 11 = 双字

    //==========================================================================
    // PC控制译码
    //==========================================================================
    assign pc_src = pc_ctrl[1:0];
    // 00 = ALU结果 (PC+4)
    // 01 = ALUOut (分支目标)
    // 10 = 跳转目标
    // 11 = ALU结果 (JALR)

    //==========================================================================
    // 时序控制译码
    //==========================================================================
    assign next_ctrl = seq_ctrl[1:0];
    // 00 = 停止
    // 01 = 顺序执行 (μPC+1)
    // 10 = 条件跳转
    // 11 = 无条件跳转

    assign condition = seq_ctrl[3:2];
    // 条件码 (当next_ctrl=10时使用)
    // 00 = 无条件
    // 01 = ALU zero
    // 10 = ALU sign
    // 11 = 溢出

    //==========================================================================
    // 下地址
    //==========================================================================
    assign next_addr = {4'b0, cond_addr};  // 扩展到8位

endmodule
```

### 5.2 下地址生成器Verilog实现

```verilog
// ============================================================================
// 下地址生成器 (Next Address Generator)
// ============================================================================

module next_address_generator (
    input  wire        clk,
    input  wire        reset,

    // 当前微指令信息
    input  wire [1:0]  next_ctrl,      // 下地址控制
    input  wire [3:0]  condition,      // 分支条件
    input  wire [7:0]  next_addr,      // 微指令中的下地址

    // 状态输入
    input  wire [6:0]  opcode,         // 机器指令操作码
    input  wire        alu_zero,       // ALU零标志
    input  wire        alu_sign,       // ALU符号标志
    input  wire        alu_overflow,   // ALU溢出标志
    input  wire        alu_carry,      // ALU进位标志

    // 当前微地址
    input  wire [7:0]  current_uaddr,

    // 输出下地址
    output reg  [7:0]  next_uaddr,

    // 调试输出
    output wire [7:0]  mapped_addr
);

    //==========================================================================
    // 映射ROM: Opcode → 微程序入口地址
    //==========================================================================
    reg [7:0] mapping_rom [0:127];  // 128个条目 (7位opcode)

    initial begin
        // 初始化映射表
        mapping_rom[7'b0110011] = 8'h10;  // R-Type
        mapping_rom[7'b0010011] = 8'h20;  // I-Type ALU
        mapping_rom[7'b0000011] = 8'h30;  // Load
        mapping_rom[7'b0100011] = 8'h40;  // Store
        mapping_rom[7'b1100011] = 8'h50;  // Branch
        mapping_rom[7'b1101111] = 8'h60;  // JAL
        mapping_rom[7'b1100111] = 8'h65;  // JALR
        mapping_rom[7'b0110111] = 8'h70;  // LUI
        mapping_rom[7'b0010111] = 8'h75;  // AUIPC
        // 其他默认为0xFF (非法指令处理)
    end

    // 读取映射地址
    assign mapped_addr = mapping_rom[opcode];

    //==========================================================================
    // 条件判断
    //==========================================================================
    reg condition_met;

    always @(*) begin
        case (condition)
            4'b0000: condition_met = 1'b1;           // 无条件
            4'b0001: condition_met = alu_zero;       // ALU结果为0
            4'b0010: condition_met = ~alu_zero;      // ALU结果非0
            4'b0011: condition_met = alu_sign;       // 结果为负
            4'b0100: condition_met = ~alu_sign;      // 结果非负
            4'b0101: condition_met = alu_overflow;   // 溢出
            4'b0110: condition_met = alu_carry;      // 有进位
            4'b0111: condition_met = ~alu_carry;     // 无进位
            4'b1000: condition_met = alu_zero & ~alu_sign;  // 零且非负
            4'b1001: condition_met = alu_zero | alu_sign;   // 零或负
            default: condition_met = 1'b0;
        endcase
    end

    //==========================================================================
    // 下地址生成逻辑
    //==========================================================================
    always @(*) begin
        case (next_ctrl)
            2'b00: begin  // 停止/保持
                next_uaddr = current_uaddr;
            end

            2'b01: begin  // 顺序执行 (μPC + 1)
                next_uaddr = current_uaddr + 8'b1;
            end

            2'b10: begin  // 条件跳转
                if (condition_met) begin
                    next_uaddr = next_addr;
                end else begin
                    next_uaddr = current_uaddr + 8'b1;
                end
            end

            2'b11: begin  // 无条件跳转 / 映射跳转
                next_uaddr = next_addr;
            end

            default: begin
                next_uaddr = 8'h00;  // 返回取指
            end
        endcase
    end

endmodule
```

// ============================================================================
// 完整微程序控制器 - 多周期实现
// ============================================================================

module microprogrammed_control_unit (
    input  wire        clk,
    input  wire        reset,

    // 指令和数据接口
    input  wire [31:0] instruction,
    input  wire [31:0] alu_result,
    input  wire        alu_zero,
    input  wire        alu_sign,

    // 控制信号输出
    output wire [3:0]  alu_control,
    output wire        alu_src_a,
    output wire [1:0]  alu_src_b,
    output wire        reg_write,
    output wire [1:0]  reg_src,
    output wire        mem_read,
    output wire        mem_write,
    output wire        ir_write,
    output wire        pc_write,
    output wire [1:0]  pc_src,
    output wire        branch,
    output wire        jump,

    // 调试输出
    output wire [7:0]  debug_uaddr,
    output wire [31:0] debug_uir,
    output wire [7:0]  debug_next_uaddr
);

    //==========================================================================
    // 内部信号
    //==========================================================================
    wire [31:0] microinstruction;
    wire [7:0]  next_uaddr;
    wire [7:0]  mapped_addr;

    reg  [7:0]  u_pc;           // 微程序计数器
    reg  [7:0]  current_uaddr;  // 当前微地址

    // 微指令字段
    wire [1:0]  next_ctrl;
    wire [3:0]  condition;
    wire [7:0]  u_next_addr;

    //==========================================================================
    // 控制存储器实例化
    //==========================================================================
    control_store u_control_store (
        .clk(clk),
        .address(current_uaddr),
        .microinstruction(microinstruction)
    );

    //==========================================================================
    // 微指令译码
    //==========================================================================
    assign next_ctrl  = microinstruction[7:6];
    assign condition  = microinstruction[11:8];
    assign u_next_addr = microinstruction[7:0];

    //==========================================================================
    // 微指令到控制信号译码
    //==========================================================================
    assign alu_control = microinstruction[23:20];
    assign alu_src_a   = microinstruction[26];       // 简化映射
    assign alu_src_b   = microinstruction[25:24];
    assign reg_write   = microinstruction[31];
    assign reg_src     = microinstruction[17:16];
    assign mem_read    = microinstruction[30];
    assign mem_write   = microinstruction[29];
    assign ir_write    = microinstruction[28];
    assign pc_write    = microinstruction[27];
    assign pc_src      = microinstruction[19:18];
    assign branch      = microinstruction[25];
    assign jump        = microinstruction[24];

    //==========================================================================
    // 下地址生成器实例化
    //==========================================================================
    next_address_generator u_nag (
        .clk(clk),
        .reset(reset),
        .next_ctrl(next_ctrl),
        .condition(condition),
        .next_addr(u_next_addr),
        .opcode(instruction[6:0]),
        .alu_zero(alu_zero),
        .alu_sign(alu_sign),
        .alu_overflow(1'b0),
        .alu_carry(1'b0),
        .current_uaddr(current_uaddr),
        .next_uaddr(next_uaddr),
        .mapped_addr(mapped_addr)
    );

    //==========================================================================
    // 微程序计数器 (时序逻辑)
    //==========================================================================
    always @(posedge clk or posedge reset) begin
        if (reset) begin
            u_pc <= 8'h00;  // 从取指开始
            current_uaddr <= 8'h00;
        end else begin
            current_uaddr <= next_uaddr;
            u_pc <= next_uaddr;
        end
    end

    //==========================================================================
    // 调试输出
    //==========================================================================
    assign debug_uaddr = current_uaddr;
    assign debug_uir = microinstruction;
    assign debug_next_uaddr = next_uaddr;

endmodule

## 9. 总结

微程序控制器通过存储在控制存储器中的微程序来生成控制信号，主要特点包括：

1. **灵活性**: 修改微程序即可改变指令行为，无需改动硬件
2. **规整性**: 控制存储器结构规整，易于设计和验证
3. **可扩展性**: 增加新指令只需添加微程序
4. **适合CISC**: 复杂指令可分解为简单微操作序列

与硬布线控制相比，微程序控制在性能上有所牺牲，但在设计灵活性、可维护性和支持复杂指令集方面具有明显优势。现代处理器往往采用混合方案，简单指令使用硬布线控制以获得高性能，复杂指令使用微程序控制以保持灵活性。
