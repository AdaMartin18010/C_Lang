## жҰӮиҝ°

дәҢиҝӣеҲ¶з®—жңҜжҳҜи®Ўз®—жңәиҝҗз®—зҡ„еҹәзЎҖгҖӮ
д»ҺжңҖеҹәжң¬зҡ„дҪҚж“ҚдҪңеҲ°еӨҚжқӮзҡ„еӨҡдҪҚиҝҗз®—пјҢзҗҶи§Јиҝҷдәӣеә•еұӮе®һзҺ°еҜ№дәҺжҺҢжҸЎи®Ўз®—жңәдҪ“зі»з»“жһ„е’ҢдјҳеҢ–CиҜӯиЁҖзЁӢеәҸиҮіе…ійҮҚиҰҒгҖӮ

### дёәд»Җд№ҲйңҖиҰҒдәҶи§ЈдәҢиҝӣеҲ¶з®—жңҜпјҹ

```c
// CиҜӯиЁҖдёӯзҡ„еҠ жі•зңӢдјјз®ҖеҚ•
int a = 5, b = 3;
int sum = a + b;  // еә•еұӮжҳҜеҰӮдҪ•е®һзҺ°зҡ„пјҹ

// е®һйҷ…дёҠпјҢзј–иҜ‘еҷЁдјҡе°Ҷе…¶иҪ¬жҚўдёәжңәеҷЁжҢҮд»Ө
// жңҖз»ҲеңЁзЎ¬д»¶дёҠз”ұеҠ жі•еҷЁз”өи·Ҝжү§иЎҢ
```

module half_adder (
    input  wire a,      // иў«еҠ ж•°
    input  wire b,      // еҠ ж•°
    output wire sum,    // е’Ң
    output wire carry   // иҝӣдҪҚиҫ“еҮә
);

    // йҖ»иҫ‘е®һзҺ°
    assign sum   = a ^ b;   // ејӮжҲ–иҝҗз®—еҫ—еҲ°е’Ң
    assign carry = a & b;   // дёҺиҝҗз®—еҫ—еҲ°иҝӣдҪҚ

endmodule

module ripple_carry_adder_4bit (
    input  wire [3:0] a,      // 4дҪҚиў«еҠ ж•°
    input  wire [3:0] b,      // 4дҪҚеҠ ж•°
    input  wire       cin,    // иҝӣдҪҚиҫ“е…Ҙ
    output wire [3:0] sum,    // 4дҪҚе’Ң
    output wire       cout    // иҝӣдҪҚиҫ“еҮә
);

    // еҶ…йғЁиҝӣдҪҚдҝЎеҸ·
    wire c0, c1, c2;

    // зә§иҒ”4дёӘе…ЁеҠ еҷЁ
    full_adder fa0 (
        .a(a[0]),
        .b(b[0]),
        .cin(cin),
        .sum(sum[0]),
        .cout(c0)
    );

    full_adder fa1 (
        .a(a[1]),
        .b(b[1]),
        .cin(c0),
        .sum(sum[1]),
        .cout(c1)
    );

    full_adder fa2 (
        .a(a[2]),
        .b(b[2]),
        .cin(c1),
        .sum(sum[2]),
        .cout(c2)
    );

    full_adder fa3 (
        .a(a[3]),
        .b(b[3]),
        .cin(c2),
        .sum(sum[3]),
        .cout(cout)
    );

endmodule

// ============================================================
// еҸӮж•°еҢ–NдҪҚиЎҢжіўиҝӣдҪҚеҠ жі•еҷЁ
// ============================================================

module ripple_carry_adder_nbit #(
    parameter WIDTH = 32
)(
    input  wire [WIDTH-1:0] a,
    input  wire [WIDTH-1:0] b,
    input  wire             cin,
    output wire [WIDTH-1:0] sum,
    output wire             cout
);

    // еҶ…йғЁиҝӣдҪҚй“ҫ
    wire [WIDTH:0] carry;
    assign carry[0] = cin;
    assign cout = carry[WIDTH];

    // з”ҹжҲҗеӨҡдёӘе…ЁеҠ еҷЁ
    generate
        genvar i;
        for (i = 0; i < WIDTH; i = i + 1) begin : fa_chain
            full_adder fa (
                .a(a[i]),
                .b(b[i]),
                .cin(carry[i]),
                .sum(sum[i]),
                .cout(carry[i+1])
            );
        end
    endgenerate

endmodule

## и¶…еүҚиҝӣдҪҚеҠ жі•еҷЁ

### жҰӮиҝ°

и¶…еүҚиҝӣдҪҚеҠ жі•еҷЁпјҲCarry Lookahead Adder, CLAпјүйҖҡиҝҮе№¶иЎҢи®Ўз®—иҝӣдҪҚдҝЎеҸ·пјҢе…ӢжңҚдәҶиЎҢжіўиҝӣдҪҚеҠ жі•еҷЁзҡ„е»¶иҝҹй—®йўҳгҖӮе…¶ж ёеҝғжҖқжғіжҳҜжҸҗеүҚи®Ўз®—жҜҸдёҖдҪҚзҡ„иҝӣдҪҚпјҢиҖҢдёҚжҳҜзӯүеҫ…еүҚдёҖзә§дј йҖ’гҖӮ

### иҝӣдҪҚз”ҹжҲҗдёҺдј ж’ӯдҝЎеҸ·

```
еҜ№дәҺжҜҸдёҖдҪҚiпјҢе®ҡд№ү:

з”ҹжҲҗдҝЎеҸ· (Generate): Gi = Ai В· Bi
  - еҪ“Aiе’ҢBiйғҪдёә1ж—¶пјҢеҝ…з„¶дә§з”ҹиҝӣдҪҚ

дј ж’ӯдҝЎеҸ· (Propagate): Pi = Ai вҠ• Bi
  - еҪ“Aiе’ҢBiдёӯжңүдёҖдёӘдёә1ж—¶пјҢиҝӣдҪҚдјҡиў«дј йҖ’

иҝӣдҪҚе…¬ејҸ:
C_{i+1} = Gi + Pi В· Ci
```

// 4дҪҚи¶…еүҚиҝӣдҪҚеҠ жі•еҷЁ (Carry Lookahead Adder)
// ============================================================

module carry_lookahead_adder_4bit (
    input  wire [3:0] a,       // 4дҪҚиў«еҠ ж•°
    input  wire [3:0] b,       // 4дҪҚеҠ ж•°
    input  wire       cin,     // иҝӣдҪҚиҫ“е…Ҙ
    output wire [3:0] sum,     // 4дҪҚе’Ң
    output wire       cout,    // иҝӣдҪҚиҫ“еҮә
    output wire       g_block, // еқ—з”ҹжҲҗдҝЎеҸ·
    output wire       p_block  // еқ—дј ж’ӯдҝЎеҸ·
);

    // еҶ…йғЁдҝЎеҸ·
    wire [3:0] g;  // з”ҹжҲҗдҝЎеҸ·
    wire [3:0] p;  // дј ж’ӯдҝЎеҸ·
    wire [4:0] c;  // иҝӣдҪҚй“ҫ (c[0]=cin, c[4]=cout)

    assign c[0] = cin;

    // з”ҹжҲҗе’Ңдј ж’ӯдҝЎеҸ·и®Ўз®— (з¬¬дёҖзә§ - е№¶иЎҢ)
    assign g[0] = a[0] & b[0];
    assign g[1] = a[1] & b[1];
    assign g[2] = a[2] & b[2];
    assign g[3] = a[3] & b[3];

    assign p[0] = a[0] ^ b[0];
    assign p[1] = a[1] ^ b[1];
    assign p[2] = a[2] ^ b[2];
    assign p[3] = a[3] ^ b[3];

    // и¶…еүҚиҝӣдҪҚи®Ўз®— (з¬¬дәҢзә§ - е№¶иЎҢ)
    assign c[1] = g[0] | (p[0] & c[0]);
    assign c[2] = g[1] | (p[1] & g[0]) | (p[1] & p[0] & c[0]);
    assign c[3] = g[2] | (p[2] & g[1]) | (p[2] & p[1] & g[0]) | (p[2] & p[1] & p[0] & c[0]);
    assign c[4] = g[3] | (p[3] & g[2]) | (p[3] & p[2] & g[1]) | (p[3] & p[2] & p[1] & g[0]) |
                  (p[3] & p[2] & p[1] & p[0] & c[0]);

    // е’Ңи®Ўз®— (з¬¬дёүзә§ - е№¶иЎҢ)
    assign sum[0] = p[0] ^ c[0];
    assign sum[1] = p[1] ^ c[1];
    assign sum[2] = p[2] ^ c[2];
    assign sum[3] = p[3] ^ c[3];

    assign cout = c[4];

    // еқ—зә§з”ҹжҲҗе’Ңдј ж’ӯдҝЎеҸ· (з”ЁдәҺзә§иҒ”)
    assign g_block = g[3] | (p[3] & g[2]) | (p[3] & p[2] & g[1]) | (p[3] & p[2] & p[1] & g[0]);
    assign p_block = p[3] & p[2] & p[1] & p[0];

endmodule

### иЎҘз ҒеҹәзЎҖ

еңЁи®Ўз®—жңәдёӯпјҢеҮҸжі•йҖҡиҝҮиЎҘз ҒиҪ¬жҚўдёәеҠ жі•жқҘе®һзҺ°пјҡ

```
A - B = A + (-Bзҡ„иЎҘз Ғ)
```

#### иЎҘз Ғи®Ўз®—

```
Bзҡ„иЎҘз Ғ = ~B + 1  (жҢүдҪҚеҸ–еҸҚеҗҺеҠ 1)

зӨәдҫӢ: и®Ўз®— -5 зҡ„8дҪҚиЎҘз Ғ
еҺҹз Ғ:  0000 0101  (5)
еҸҚз Ғ:  1111 1010  (жҢүдҪҚеҸ–еҸҚ)
иЎҘз Ғ:  1111 1011  (еҠ 1) = 0xFB
```

### жәўеҮәжЈҖжөӢ

```
жңүз¬ҰеҸ·ж•°жәўеҮәжЈҖжөӢ:

жәўеҮәжқЎд»¶ = Cout XOR C[n-1]
(жңҖй«ҳдҪҚиҝӣдҪҚдёҺж¬Ўй«ҳдҪҚиҝӣдҪҚдёҚеҗҢ)

жҲ–иҖ…:
жәўеҮә = (A[n-1] == B[n-1]) && (Sum[n-1] != A[n-1])
(дёӨдёӘеҗҢеҸ·ж•°зӣёеҠ пјҢз»“жһңз¬ҰеҸ·дёҚеҗҢ)

ж— з¬ҰеҸ·ж•°жәўеҮәжЈҖжөӢ:
жәўеҮә = Cout (иҝӣдҪҚиҫ“еҮә)
```

// ============================================================
// 8дҪҚеҠ еҮҸжі•еҷЁ (Adder/Subtractor)
// ============================================================

module adder_subtractor_8bit (
    input  wire [7:0] a,       // ж“ҚдҪңж•°A
    input  wire [7:0] b,       // ж“ҚдҪңж•°B
    input  wire       sub,     // 0=еҠ жі•, 1=еҮҸжі•
    output wire [7:0] result,  // з»“жһң
    output wire       cout,    // иҝӣдҪҚ/еҖҹдҪҚиҫ“еҮә
    output wire       overflow // жәўеҮәж Үеҝ—
);

    wire [7:0] b_xor;
    wire [8:0] carry;

    // BдёҺSubејӮжҲ–пјҡSub=0ж—¶BдёҚеҸҳпјҢSub=1ж—¶BеҸ–еҸҚ
    assign b_xor = b ^ {8{sub}};
    assign carry[0] = sub;  // еҮҸжі•ж—¶еҠ 1(иЎҘз Ғ)

    // дҪҝз”ЁиЎҢжіўиҝӣдҪҚеҠ жі•еҷЁ
    generate
        genvar i;
        for (i = 0; i < 8; i = i + 1) begin : adder_chain
            full_adder fa (
                .a(a[i]),
                .b(b_xor[i]),
                .cin(carry[i]),
                .sum(result[i]),
                .cout(carry[i+1])
            );
        end
    endgenerate

    assign cout = carry[8];

    // жңүз¬ҰеҸ·ж•°жәўеҮәжЈҖжөӢ
    // жәўеҮәеҸ‘з”ҹеңЁпјҡдёӨдёӘжӯЈж•°зӣёеҠ еҫ—иҙҹж•°пјҢжҲ–дёӨдёӘиҙҹж•°зӣёеҠ еҫ—жӯЈж•°
    assign overflow = carry[7] ^ carry[8];

endmodule

`timescale 1ps/1fs

module adder_timing_analysis;

    // еҸӮж•°е®ҡд№ү
    parameter T_XOR = 50;   // XORй—Ёе»¶иҝҹ 50ps
    parameter T_AND = 30;   // ANDй—Ёе»¶иҝҹ 30ps

## жҖ»з»“

жң¬ж–ҮжЎЈиҜҰз»Ҷд»Ӣз»ҚдәҶдәҢиҝӣеҲ¶з®—жңҜзҡ„ж ёеҝғеҶ…е®№пјҢд»ҺжңҖеҹәжң¬зҡ„еҚҠеҠ еҷЁеҲ°еӨҚжқӮзҡ„еӨҡдҪҚи¶…еүҚиҝӣдҪҚеҠ жі•еҷЁгҖӮе…ій”®зҹҘиҜҶзӮ№еҢ…жӢ¬пјҡ

1. **еҚҠеҠ еҷЁдёҺе…ЁеҠ еҷЁ**пјҡжһ„е»әжүҖжңүеҠ жі•еҷЁзҡ„еҹәзЎҖеҚ•е…ғ
2. **иЎҢжіўиҝӣдҪҚеҠ жі•еҷЁ**пјҡз®ҖеҚ•дҪҶе»¶иҝҹйҡҸдҪҚж•°зәҝжҖ§еўһй•ҝ
3. **и¶…еүҚиҝӣдҪҚеҠ жі•еҷЁ**пјҡйҖҡиҝҮе№¶иЎҢи®Ўз®—иҝӣдҪҚе®һзҺ°еёёж•°зә§е»¶иҝҹ
4. **еҮҸжі•еҷЁе®һзҺ°**пјҡеҲ©з”ЁиЎҘз Ғе°ҶеҮҸжі•иҪ¬жҚўдёәеҠ жі•
5. **еӨҡдҪҚеҠ жі•еҷЁжһ„е»ә**пјҡеұӮж¬ЎеҢ–и®ҫи®Ўж–№жі•
6. **е»¶иҝҹеҲҶжһҗ**пјҡзҗҶи§ЈдёҚеҗҢеҠ жі•еҷЁжһ¶жһ„зҡ„жҖ§иғҪзү№еҫҒ
7. **CиҜӯиЁҖжҳ е°„**пјҡзҗҶи§Јй«ҳеұӮд»Јз ҒеҰӮдҪ•жҳ е°„еҲ°еә•еұӮзЎ¬д»¶

иҝҷдәӣзҹҘиҜҶеҜ№дәҺзҗҶи§Ји®Ўз®—жңәдҪ“зі»з»“жһ„гҖҒдјҳеҢ–CиҜӯиЁҖзЁӢеәҸд»ҘеҸҠж•°еӯ—з”өи·Ҝи®ҫи®ЎйғҪиҮіе…ійҮҚиҰҒгҖӮ
