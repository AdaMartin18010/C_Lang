## 概述

浮点单元（FPU）是处理浮点数运算的专用硬件单元。与整数运算不同，浮点运算需要处理符号、指数和尾数三个部分，并需要特殊的舍入和异常处理机制。

## 浮点加法器设计

### 算法流程

```
浮点加法步骤 (A + B):

1. 比较指数，找出较大的指数
2. 对阶: 将较小指数的尾数右移，使其指数与较大的相同
3. 尾数相加 (考虑符号)
4. 规格化: 调整结果使其符合IEEE 754格式
5. 舍入: 根据舍入模式处理多余的位数
6. 处理特殊情况 (溢出、下溢等)

示例: 1.5 + 1.25
  1.5  = 1.1 × 2^0  = 0x3FC00000
  1.25 = 1.01 × 2^0 = 0x3FA00000

  指数相同，直接相加:
    1.10000000000000000000000
  + 1.01000000000000000000000
  ─────────────────────────────
   10.11000000000000000000000

  规格化: 1.011 × 2^1 = 2.75 ✓
```

### Verilog实现

```verilog
// ============================================================
// 单精度浮点加法器
// ============================================================

module fp_adder_single (
    input  wire [31:0] a,           // 输入A
    input  wire [31:0] b,           // 输入B
    output reg  [31:0] result,      // 结果
    output reg         overflow,    // 溢出标志
    output reg         underflow    // 下溢标志
);

    // 分解字段
    wire        sign_a, sign_b;
    wire [7:0]  exp_a, exp_b;
    wire [22:0] frac_a, frac_b;

    assign sign_a = a[31];
    assign exp_a  = a[30:23];
    assign frac_a = a[22:0];

    assign sign_b = b[31];
    assign exp_b  = b[30:23];
    assign frac_b = b[22:0];

    // 内部信号
    reg         sign_res;
    reg [7:0]   exp_res;
    reg [22:0]  frac_res;

    // 尾数处理 (含隐含位和保护位)
    reg [26:0]  mant_a, mant_b;     // {隐含1, 尾数23位, G, R, S}
    reg [27:0]  mant_sum;           // 加法结果 (可能进位)

    // 指数差和移位
    reg [7:0]   exp_diff;
    reg [7:0]   exp_max;
    reg         a_larger;

    // 特殊情况检测
    wire        a_zero, b_zero;
    wire        a_inf, b_inf;
    wire        a_nan, b_nan;

    assign a_zero = (exp_a == 8'b0) && (frac_a == 23'b0);
    assign b_zero = (exp_b == 8'b0) && (frac_b == 23'b0);
    assign a_inf  = (exp_a == 8'hFF) && (frac_a == 23'b0);
    assign b_inf  = (exp_b == 8'hFF) && (frac_b == 23'b0);
    assign a_nan  = (exp_a == 8'hFF) && (frac_a != 23'b0);
    assign b_nan  = (exp_b == 8'hFF) && (frac_b != 23'b0);

    // 主逻辑
    always @(*) begin
        // 默认输出
        result = 32'b0;
        overflow = 1'b0;
        underflow = 1'b0;

        // 处理特殊情况
        if (a_nan || b_nan) begin
            result = 32'h7FC00000;  // 返回NaN
        end else if (a_inf && b_inf && (sign_a != sign_b)) begin
            result = 32'h7FC00000;  // Inf - Inf = NaN
        end else if (a_inf) begin
            result = a;
        end else if (b_inf) begin
            result = b;
        end else if (a_zero) begin
            result = b;
        end else if (b_zero) begin
            result = a;
        end else begin
            // 正常浮点加法流程

            // 比较指数
            if (exp_a >= exp_b) begin
                exp_max = exp_a;
                exp_diff = exp_a - exp_b;
                a_larger = 1'b1;
            end else begin
                exp_max = exp_b;
                exp_diff = exp_b - exp_a;
                a_larger = 1'b0;
            end

            // 准备尾数 (添加隐含位)
            mant_a = {1'b1, frac_a, 3'b0};  // 隐含1 + 尾数 + 3位保护
            mant_b = {1'b1, frac_b, 3'b0};

            // 对阶: 较小指数的尾数右移
            if (a_larger) begin
                mant_b = mant_b >> exp_diff;
            end else begin
                mant_a = mant_a >> exp_diff;
            end

            // 尾数相加或相减
            if (sign_a == sign_b) begin
                // 同号: 直接相加
                mant_sum = {1'b0, mant_a} + {1'b0, mant_b};
                sign_res = sign_a;
            end else begin
                // 异号: 相减 (大减小)
                if (a_larger) begin
                    mant_sum = {1'b0, mant_a} - {1'b0, mant_b};
                    sign_res = sign_a;
                end else begin
                    mant_sum = {1'b0, mant_b} - {1'b0, mant_a};
                    sign_res = sign_b;
                end
            end

            // 规格化
            exp_res = exp_max;
            if (mant_sum[27]) begin
                // 进位，需要右规
                mant_sum = mant_sum >> 1;
                exp_res = exp_res + 1'b1;
            end else begin
                // 左规
                while (!mant_sum[26] && exp_res > 0) begin
                    mant_sum = mant_sum << 1;
                    exp_res = exp_res - 1'b1;
                end
            end

            // 舍入 (向偶数舍入)
            if (mant_sum[2]) begin  // G位为1
                if (mant_sum[1:0] != 2'b0 || mant_sum[3]) begin
                    mant_sum[26:3] = mant_sum[26:3] + 1'b1;
                end
            end

            // 再次规格化 (舍入可能产生进位)
            if (mant_sum[27]) begin
                mant_sum = mant_sum >> 1;
                exp_res = exp_res + 1'b1;
            end

            // 检查溢出和下溢
            if (exp_res >= 8'hFF) begin
                overflow = 1'b1;
                result = {sign_res, 8'hFF, 23'b0};  // Inf
            end else if (exp_res == 8'b0) begin
                underflow = 1'b1;
                result = {sign_res, 31'b0};  // 零或非规格化数
            end else begin
                frac_res = mant_sum[25:3];  // 去掉隐含位和保护位
                result = {sign_res, exp_res, frac_res};
            end
        end
    end

endmodule

## 浮点乘法器设计

### 算法流程

```
浮点乘法步骤 (A × B):

1. 符号计算: S_res = SA XOR SB
2. 指数相加: E_res = EA + EB - 偏移量(127)
3. 尾数相乘: M_res = MA × MB (1.x × 1.y = z.zzzz)
4. 规格化: 如果 M_res >= 2，右移1位，指数+1
5. 舍入: 根据舍入模式处理
6. 处理特殊情况

示例: 1.5 × 2.0
  1.5 = 1.1 × 2^0  = 0x3FC00000
  2.0 = 1.0 × 2^1  = 0x40000000

  符号: 0 XOR 0 = 0 (正)
  指数: 127 + 128 - 127 = 128 → 2^1
  尾数: 1.1 × 1.0 = 1.10

  结果: 1.1 × 2^1 = 3.0 = 0x40400000 ✓
```

// ============================================================
// 单精度浮点乘法器
// ============================================================

module fp_multiplier_single (
    input  wire [31:0] a,           // 输入A
    input  wire [31:0] b,           // 输入B
    output reg  [31:0] result,      // 结果
    output reg         overflow,    // 溢出标志
    output reg         underflow    // 下溢标志
);

    // 分解字段
    wire        sign_a, sign_b;
    wire [7:0]  exp_a, exp_b;
    wire [22:0] frac_a, frac_b;

    assign sign_a = a[31];
    assign exp_a  = a[30:23];
    assign frac_a = a[22:0];

    assign sign_b = b[31];
    assign exp_b  = b[30:23];
    assign frac_b = b[22:0];

    // 内部信号
    reg         sign_res;
    reg [8:0]   exp_res;            // 9位防止溢出
    reg [22:0]  frac_res;

    // 尾数乘法
    reg [47:0]  mant_product;       // 24×24=48位
    reg [23:0]  mant_a, mant_b;     // 含隐含位

    // 特殊情况检测
    wire        a_zero, b_zero;
    wire        a_inf, b_inf;
    wire        a_nan, b_nan;

    assign a_zero = (exp_a == 8'b0) && (frac_a == 23'b0);
    assign b_zero = (exp_b == 8'b0) && (frac_b == 23'b0);
    assign a_inf  = (exp_a == 8'hFF) && (frac_a == 23'b0);
    assign b_inf  = (exp_b == 8'hFF) && (frac_b == 23'b0);
    assign a_nan  = (exp_a == 8'hFF) && (frac_a != 23'b0);
    assign b_nan  = (exp_b == 8'hFF) && (frac_b != 23'b0);

    // 主逻辑
    always @(*) begin
        // 默认输出
        result = 32'b0;
        overflow = 1'b0;
        underflow = 1'b0;

        // 处理特殊情况
        if (a_nan || b_nan) begin
            result = 32'h7FC00000;  // NaN
        end else if ((a_inf && b_zero) || (b_inf && a_zero)) begin
            result = 32'h7FC00000;  // Inf × 0 = NaN
        end else if (a_inf || b_inf) begin
            result = {sign_a ^ sign_b, 8'hFF, 23'b0};  // Inf
        end else if (a_zero || b_zero) begin
            result = {sign_a ^ sign_b, 31'b0};  // 零
        end else begin
            // 正常浮点乘法流程

            // 符号计算
            sign_res = sign_a ^ sign_b;

            // 指数相加 (减去偏移量127)
            exp_res = {1'b0, exp_a} + {1'b0, exp_b} - 9'd127;

            // 尾数准备 (添加隐含位)
            mant_a = {1'b1, frac_a};
            mant_b = {1'b1, frac_b};

            // 尾数乘法 (48位结果)
            mant_product = mant_a * mant_b;

            // 规格化
            if (mant_product[47]) begin
                // 需要右规
                frac_res = mant_product[46:24];
                exp_res = exp_res + 1'b1;
            end else begin
                // 已经是规格化形式
                frac_res = mant_product[45:23];
            end

            // 舍入 (检查舍入位)
            if (mant_product[22]) begin  // 舍入位为1
                frac_res = frac_res + 1'b1;
                // 舍入可能导致进位
                if (frac_res == 23'b0) begin
                    exp_res = exp_res + 1'b1;
                end
            end

            // 溢出和下溢检测
            if (exp_res >= 9'h0FF) begin
                overflow = 1'b1;
                result = {sign_res, 8'hFF, 23'b0};  // Inf
            end else if (exp_res[8] || (exp_res == 9'b0)) begin
                underflow = 1'b1;
                result = {sign_res, 31'b0};  // 零
            end else begin
                result = {sign_res, exp_res[7:0], frac_res};
            end
        end
    end

endmodule

### 舍入模式

```
┌─────────────────────────────────────────────────────────────────────┐
│                    IEEE 754舍入模式                                  │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  1. 向最接近的偶数舍入 (Round to Nearest, Even):                   │
│     默认模式，统计上最公平                                          │
│     例如: 1.5 → 2, 2.5 → 2                                          │
│                                                                     │
│  2. 向零舍入 (Round toward Zero):                                   │
│     直接截断小数部分                                                │
│     例如: 1.9 → 1, -1.9 → -1                                        │
│                                                                     │
│  3. 向正无穷舍入 (Round toward +Infinity):                          │
│     总是向上取整                                                    │
│     例如: 1.1 → 2, -1.9 → -1                                        │
│                                                                     │
│  4. 向负无穷舍入 (Round toward -Infinity):                          │
│     总是向下取整                                                    │
│     例如: 1.1 → 1, -1.9 → -2                                        │
│                                                                     │
│  保护位、舍入位、粘滞位 (G, R, S):                                  │
│  ┌───────────────────────────────────────────────────────────┐     │
│  │   计算结果: 1.01010101 │ 1101 │ 0110...                    │     │

### 异常处理

```verilog
// ============================================================
// 浮点异常检测与处理模块
// ============================================================

module fp_exception_handler (
    input  wire [7:0]  exp_result,
    input  wire [22:0] frac_result,
    input  wire        mant_msb,        // 尾数乘法最高位
    input  wire        sign,
    input  wire        is_nan_a, is_nan_b,
    input  wire        is_inf_a, is_inf_b,
    input  wire        is_zero_a, is_zero_b,
    output reg  [31:0] final_result,
    output reg         invalid_op,      // 无效操作
    output reg         div_by_zero,     // 除零
    output reg         overflow,        // 溢出
    output reg         underflow,       // 下溢
    output reg         inexact          // 不精确
);

    always @(*) begin
        // 默认无异常
        invalid_op = 1'b0;
        div_by_zero = 1'b0;
        overflow = 1'b0;
        underflow = 1'b0;
        inexact = 1'b0;

        // 检测无效操作
        if ((is_inf_a && is_zero_b) || (is_inf_a && is_inf_b)) begin
            invalid_op = 1'b1;
            final_result = 32'h7FC00000;  // qNaN
        end else if (is_nan_a) begin
            final_result = 32'h7FC00000;
        end else if (is_nan_b) begin
            final_result = 32'h7FC00000;
        end
        // 检测溢出
        else if (exp_result >= 8'hFF) begin
            overflow = 1'b1;
            inexact = 1'b1;
            final_result = {sign, 8'hFF, 23'b0};  // Inf
        end
        // 检测下溢
        else if (exp_result == 8'b0) begin
            underflow = 1'b1;
            inexact = 1'b1;
            final_result = {sign, 31'b0};  // 零
        end
        // 正常结果
        else begin
            final_result = {sign, exp_result, frac_result};
            // 检测不精确 (有舍入)
            inexact = mant_msb || (frac_result != 23'b0);
        end
    end

endmodule
```

### 浮点精度问题

```c
#include <stdio.h>
#include <float.h>

int main() {
    // 精度问题演示
    float f1 = 0.1f;
    float f2 = 0.2f;
    float f3 = f1 + f2;

    // 预期: 0.3, 实际: 0.30000001192092896
    printf("0.1 + 0.2 = %.17f\n", f3);
    printf("0.1 + 0.2 == 0.3 ? %s\n", (f3 == 0.3f) ? "true" : "false");

    // 正确比较方法
    float epsilon = 1e-6f;
    printf("近似相等? %s\n", (fabs(f3 - 0.3f) < epsilon) ? "true" : "false");

    // 大数吃小数
    float big = 1e8f;
    float small = 1.0f;
    float sum = big + small;  // 结果还是1e8!
    printf("%f + %f = %f\n", big, small, sum);

    // Kahan求和算法 (减少误差)
    float sum_k = 0.0f;
    float c = 0.0f;  // 补偿
    for (int i = 0; i < 1000; i++) {
        float y = 0.001f - c;
        float t = sum_k + y;
        c = (t - sum_k) - y;
        sum_k = t;
    }
    printf("Kahan sum: %f\n", sum_k);

    return 0;
}
```
