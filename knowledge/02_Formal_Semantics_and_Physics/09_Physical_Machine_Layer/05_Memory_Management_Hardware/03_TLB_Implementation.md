## 1. 概述

TLB (Translation Lookaside Buffer) 是MMU中的关键缓存组件，用于缓存虚拟地址到物理地址的映射关系。由于页表通常存储在内存中，每次地址转换都遍历页表会带来显著的延迟。TLB通过缓存最近的地址转换，大幅减少地址转换时间。

## 2. TLB组织方式

### 3.2 C语言实现LRU替换

```c
/*
 * TLB LRU替换策略的C语言实现
 */

#include <stdio.h>
#include <stdint.h>
#include <stdbool.h>
#include <string.h>

#define TLB_SIZE 64
#define VPN_BITS 36
#define PFN_BITS 40

/*
 * TLB条目
 */
typedef struct {
    uint64_t vpn;        // 虚拟页号 (标签)
    uint64_t pfn;        // 物理页号
    uint16_t asid;       // 地址空间ID
    uint8_t  flags;      // 权限标志
    bool     valid;      // 有效位
    bool     global;     // 全局页
    uint8_t  lru_counter; // LRU计数器
} tlb_entry_t;

/*
 * TLB结构
 */
typedef struct {
    tlb_entry_t entries[TLB_SIZE];
    uint32_t access_count;  // 访问计数，用于LRU更新
} tlb_t;

/*
 * 初始化TLB
 */
void tlb_init(tlb_t *tlb) {
    memset(tlb, 0, sizeof(tlb_t));
    for (int i = 0; i < TLB_SIZE; i++) {
        tlb->entries[i].valid = false;
        tlb->entries[i].lru_counter = 0;
    }
    tlb->access_count = 0;
}

/*
 * 查找TLB条目
 */
bool tlb_lookup(tlb_t *tlb, uint64_t vpn, uint16_t asid,
                uint64_t *pfn, uint8_t *flags) {
    for (int i = 0; i < TLB_SIZE; i++) {
        if (tlb->entries[i].valid &&
            tlb->entries[i].vpn == vpn &&
            (tlb->entries[i].global || tlb->entries[i].asid == asid)) {

            // 命中!
            *pfn = tlb->entries[i].pfn;
            *flags = tlb->entries[i].flags;

            // 更新LRU计数器 (置为最大值)
            tlb->entries[i].lru_counter = 255;

            // 减少其他条目的计数器
            for (int j = 0; j < TLB_SIZE; j++) {
                if (j != i && tlb->entries[j].valid &&
                    tlb->entries[j].lru_counter > 0) {
                    tlb->entries[j].lru_counter--;
                }
            }

            tlb->access_count++;
            return true;
        }
    }

    // 未命中
    return false;
}

/*
 * 插入TLB条目 (使用LRU替换)
 */
void tlb_insert(tlb_t *tlb, uint64_t vpn, uint64_t pfn,
                uint16_t asid, uint8_t flags, bool global) {
    int victim_idx = -1;
    uint8_t min_lru = 255;

    // 查找空闲条目或LRU最小的条目
    for (int i = 0; i < TLB_SIZE; i++) {
        if (!tlb->entries[i].valid) {
            victim_idx = i;
            break;
        }
        if (tlb->entries[i].lru_counter < min_lru) {
            min_lru = tlb->entries[i].lru_counter;
            victim_idx = i;
        }
    }

    // 插入新条目
    tlb->entries[victim_idx].valid = true;
    tlb->entries[victim_idx].vpn = vpn;
    tlb->entries[victim_idx].pfn = pfn;
    tlb->entries[victim_idx].asid = asid;
    tlb->entries[victim_idx].flags = flags;
    tlb->entries[victim_idx].global = global;
    tlb->entries[victim_idx].lru_counter = 255;

    printf("TLB Insert: VPN=0x%lx, PFN=0x%lx, Index=%d (LRU replaced %d)\n",
           vpn, pfn, victim_idx, min_lru);
}

/*
 * 打印TLB状态
 */
void tlb_print(tlb_t *tlb) {
    printf("\n=== TLB Status ===\n");
    int valid_count = 0;
    for (int i = 0; i < TLB_SIZE; i++) {
        if (tlb->entries[i].valid) {
            valid_count++;
            printf("Entry %2d: VPN=0x%010lx, PFN=0x%010lx, ASID=%d, LRU=%d\n",
                   i, tlb->entries[i].vpn, tlb->entries[i].pfn,
                   tlb->entries[i].asid, tlb->entries[i].lru_counter);
        }
    }
    printf("Valid entries: %d/%d\n", valid_count, TLB_SIZE);
}

/*
 * 演示LRU行为
 */
int main() {
    tlb_t tlb;
    uint64_t pfn;
    uint8_t flags;

    tlb_init(&tlb);

    printf("=== TLB LRU Replacement Demo ===\n\n");

    // 插入8个条目
    printf("Inserting 8 entries...\n");
    for (int i = 0; i < 8; i++) {
        tlb_insert(&tlb, 0x1000 + i, 0x2000 + i, 0, 0x07, false);
    }

    // 访问其中几个，更新LRU
    printf("\nAccessing entries 0, 2, 4, 6...\n");
    tlb_lookup(&tlb, 0x1000, 0, &pfn, &flags);
    tlb_lookup(&tlb, 0x1002, 0, &pfn, &flags);
    tlb_lookup(&tlb, 0x1004, 0, &pfn, &flags);
    tlb_lookup(&tlb, 0x1006, 0, &pfn, &flags);

    printf("\nInserting new entry (should replace LRU=0)...\n");

}
```

### 4.2 TLB Shootdown机制

```c
/*
 * Linux TLB Shootdown实现
 */

#include <linux/smp.h>
#include <linux/mm.h>
#include <asm/tlbflush.h>

/*
 * TLB Shootdown核心函数
 */

/*
 * 本地TLB刷新
 */
void local_flush_tlb(void) {
    /*
     * x86: 重新加载CR3或使用INVPCID
     * CR3 reload会刷新所有非全局TLB条目
     */
    native_write_cr3(native_read_cr3());
}

/*
 * 刷新指定虚拟地址的TLB条目 (本地)
 */
void local_flush_tlb_page(unsigned long addr) {
    /*
     * x86: 使用INVLPG指令
     * 仅使指定虚拟地址的TLB条目失效
     */
    asm volatile("invlpg (%0)" :: "r" (addr) : "memory");
}

/*
 * SMP TLB Shootdown
 * 当页表修改后，需要通知其他CPU刷新TLB
 */

// 每个CPU的TLB Shootdown统计
typedef struct {
    unsigned long tlb_flush_count;
    unsigned long tlb_shootdown_count;
} tlb_stats_t;

tlb_stats_t tlb_stats[NR_CPUS];

/*
 * TLB Shootdown IPI处理函数
 * 在目标CPU上执行
 */
void smp_invalidate_interrupt(struct pt_regs *regs) {
    /*
     * 1. 从共享数据结构读取需要刷新的地址
     * 2. 执行本地TLB刷新
     * 3. 通知发起者完成
     */

    struct mm_struct *mm = current->mm;

    if (mm->tlb_flush_all) {
        // 刷新整个TLB
        local_flush_tlb();
        tlb_stats[smp_processor_id()].tlb_flush_count++;
    } else {
        // 刷新指定地址
        for (int i = 0; i < mm->tlb_flush_count; i++) {
            local_flush_tlb_page(mm->tlb_flush_addrs[i]);
        }
        tlb_stats[smp_processor_id()].tlb_shootdown_count++;
    }

    // 确认完成
    ack_APIC_irq();
}

/*
 * 发起TLB Shootdown
 */
void flush_tlb_others(const struct cpumask *cpumask,
                      struct mm_struct *mm,
                      unsigned long addr) {
    /*
     * 1. 设置共享数据结构
     * 2. 向目标CPU发送IPI (处理器间中断)
     * 3. 等待所有目标CPU确认
     */

    // 设置需要刷新的地址
    mm->tlb_flush_all = (addr == 0);  // addr=0表示刷新全部
    if (!mm->tlb_flush_all) {
        mm->tlb_flush_addrs[0] = addr;
        mm->tlb_flush_count = 1;
    }

    // 发送IPI到目标CPU (不包括自己)
    smp_call_function_many(cpumask, smp_invalidate_interrupt, mm, 1);
}

/*
 * 完整TLB刷新流程 (munmap, mprotect等使用)
 */
void flush_tlb_mm_range(struct mm_struct *mm,
                        unsigned long start,
                        unsigned long end,
                        unsigned long stride) {
    int cpu = get_cpu();
    cpumask_t cpumask;

    /*
     * 步骤1: 本地刷新
     */
    if (stride == PAGE_SIZE) {
        // 刷新单个页
        local_flush_tlb_page(start);
    } else {
        // 刷新范围
        for (unsigned long addr = start; addr < end; addr += stride) {
            local_flush_tlb_page(addr);
        }
    }

    /*
     * 步骤2: 检查是否需要向其他CPU发送shootdown
     */
    cpumask_clear(&cpumask);

    // 遍历所有使用该mm的CPU
    for_each_online_cpu(cpu) {
        if (cpu == smp_processor_id())
            continue;

        if (per_cpu(cpu_tlbstate.loaded_mm, cpu) == mm) {
            cpumask_set_cpu(cpu, &cpumask);
        }
    }

    /*
     * 步骤3: 发送IPI进行shootdown
     */
    if (!cpumask_empty(&cpumask)) {
        flush_tlb_others(&cpumask, mm, start);
    }

    put_cpu();
}

/*
 * 优化的批量TLB Shootdown
 * Linux使用延迟TLB Shootdown来减少IPI开销
 */

struct tlb_batch {
    struct mm_struct *mm;
    unsigned long vaddrs[TLB_BATCH_SIZE];
    int nr;
};

DEFINE_PER_CPU(struct tlb_batch, tlb_batch);

/*
 * 批量添加TLB刷新请求
 */
void tlb_batch_add(struct mm_struct *mm, unsigned long vaddr) {
    struct tlb_batch *batch = &get_cpu_var(tlb_batch);

    batch->mm = mm;
    batch->vaddrs[batch->nr++] = vaddr;

    if (batch->nr >= TLB_BATCH_SIZE) {
        // 批量刷新
        tlb_batch_flush(batch);
    }

    put_cpu_var(tlb_batch);
}

/*
 * 执行批量刷新
 */
void tlb_batch_flush(struct tlb_batch *batch) {
    if (batch->nr == 0)
        return;

    // 本地批量刷新
    for (int i = 0; i < batch->nr; i++) {
        local_flush_tlb_page(batch->vaddrs[i]);
    }

    // 发送单次IPI通知其他CPU刷新所有地址
    cpumask_t cpumask;
    // ... 收集需要通知的CPU

    if (!cpumask_empty(&cpumask)) {
        smp_call_function_many(&cpumask, smp_invalidate_interrupt,
                               batch->mm, 1);
    }

    batch->nr = 0;
}
```

```c
/*
 * TLB对指针解引用的影响

void tlb_and_pointer_dereference(void) {
    int array[1000];

    // 第一次访问 array[0] - 可能发生TLB未命中
    // 需要页表遍历，耗时较长
    array[0] = 1;

    // 后续访问 array[1..511] - TLB命中
    // 同一页内的访问，TLB条目已缓存
    for (int i = 1; i < 512; i++) {
        array[i] = i;  // TLB命中，快速
    }

    // 访问 array[512] - 可能TLB未命中 (跨越页边界)
    // 需要访问新的页表项
    array[512] = 512;

    /*
     * 优化提示: 顺序访问连续内存可以最大化TLB命中率
     * 随机访问或大跨度跳跃会导致更多TLB未命中
     */
}

/*
 * 大页与指针性能

void huge_pages_and_pointers(void) {
    /*
     * 标准4KB页:
     * - 1GB内存需要 262,144 个页表项
     * - TLB只能缓存其中少量条目
     * - 大范围随机访问会导致大量TLB未命中
     */

    /*
     * 2MB大页:
     * - 1GB内存只需要 512 个页表项
     * - TLB可以缓存更多映射
     * - 更好的TLB命中率
     */

    /*
     * 1GB大页:
     * - 整个1GB只需要 1 个页表项
     * - 单个TLB条目覆盖整个1GB
     * - 最佳TLB效率
     */

    // 使用大页的代码示例
    // 需要特殊分配: mmap with MAP_HUGETLB
    void *huge_page = mmap(NULL, 2 * 1024 * 1024,
                           PROT_READ | PROT_WRITE,
                           MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB,
                           -1, 0);

    if (huge_page != MAP_FAILED) {
        int *array = (int*)huge_page;

        // 整个2MB范围内的访问都使用同一个TLB条目
        for (int i = 0; i < (2 * 1024 * 1024) / sizeof(int); i++) {
            array[i] = i;  // 所有访问都是TLB命中
        }

        munmap(huge_page, 2 * 1024 * 1024);
    }
}

 * volatile和TLB

    /*
     * volatile告诉编译器:
     * 1. 不要优化掉对此地址的访问
     * 2. 每次访问都要真正读写内存
     *
     * TLB层面:
     * 0xFEBC0000通常是MMIO地址
     * 页表项可能设置了PWT/PCD标志 (禁用缓存)
     * 或者映射为Device Memory类型

    printf("Virtual address: 0x%lx\n", vaddr);

     * 读取 /proc/self/pagemap 可以获取对应的物理页框号
     * 需要root权限
     *
     * 这展示了虚拟地址到物理地址的映射关系
     * 正是TLB缓存的内容
     */

    if (fd >= 0) {
        uint64_t offset = (vaddr / 4096) * sizeof(uint64_t);
        lseek(fd, offset, SEEK_SET);

        uint64_t entry;
        read(fd, &entry, sizeof(entry));

        if (entry & (1ULL << 63)) {
            uint64_t pfn = entry & ((1ULL << 55) - 1);
            uint64_t paddr = (pfn << 12) | (vaddr & 0xFFF);
            printf("Physical address: 0x%lx\n", paddr);
        }

        close(fd);
    }
}
```

```verilog
    input  wire [`VA_BITS-1:0]     lookup_va,
    input  wire [`ASID_BITS-1:0]   lookup_asid,
    output reg                     lookup_hit,
    output reg  [`PA_BITS-1:0]     lookup_pa,
    output reg  [7:0]              lookup_flags,

    // 更新端口 (来自Page Table Walker)
    input  wire                    update_en,
    input  wire [`VA_BITS-1:0]     update_va,
    input  wire [`PA_BITS-1:0]     update_pa,
    input  wire [7:0]              update_flags,
    input  wire [`ASID_BITS-1:0]   update_asid,
    input  wire                    update_global,

    // 失效端口 (TLB Shootdown)
    input  wire                    invalidate_all,
    input  wire                    invalidate_asid_en,
    input  wire [`ASID_BITS-1:0]   invalidate_asid,
    input  wire                    invalidate_va_en,
    input  wire [`VA_BITS-1:0]     invalidate_va,

    // 状态输出
    output wire [7:0]              tlb_hit_count,
    output wire [7:0]              tlb_miss_count
);

    generate
        if (`TLB_WAYS == 1) begin : gen_fully_assoc
            // 全相联TLB
            tlb_fully_associative u_tlb (
                .clk(clk),
                .rst_n(rst_n),
                .lookup_en(lookup_en),
                .lookup_va(lookup_va),
                .lookup_asid(lookup_asid),
                .lookup_hit(lookup_hit),
                .lookup_pa(lookup_pa),
                .lookup_flags(lookup_flags),
                .update_en(update_en),
                .update_va(update_va),
                .update_pa(update_pa),
                .update_flags(update_flags),
                .update_asid(update_asid),
                .update_global(update_global),
                .invalidate_all(invalidate_all),
                .invalidate_asid_en(invalidate_asid_en),
                .invalidate_asid(invalidate_asid),
                .invalidate_va_en(invalidate_va_en),
                .invalidate_va(invalidate_va),
                .hit_count(tlb_hit_count),
                .miss_count(tlb_miss_count)
            );
        end else begin : gen_set_assoc
            // 组相联TLB
            tlb_set_associative #(
                .WAYS(`TLB_WAYS),
                .SETS(`TLB_SETS)
            ) u_tlb (
                .clk(clk),
                .rst_n(rst_n),
                .lookup_en(lookup_en),
                .lookup_va(lookup_va),
                .lookup_asid(lookup_asid),
                .lookup_hit(lookup_hit),
                .lookup_pa(lookup_pa),
                .lookup_flags(lookup_flags),
                .update_en(update_en),
                .update_va(update_va),
                .update_pa(update_pa),
                .update_flags(update_flags),
                .update_asid(update_asid),
                .update_global(update_global),
                .invalidate_all(invalidate_all),
                .invalidate_asid_en(invalidate_asid_en),
                .invalidate_asid(invalidate_asid),
                .invalidate_va_en(invalidate_va_en),
                .invalidate_va(invalidate_va),
                .hit_count(tlb_hit_count),
                .miss_count(tlb_miss_count)
            );
        end
    endgenerate

endmodule

module tlb_fully_associative #(
    parameter ENTRIES = 64,
    parameter ASID_BITS = 12
)(
    input  wire                    clk,
    input  wire                    rst_n,

    // 查找端口
    input  wire                    lookup_en,
    input  wire [`VA_BITS-1:0]     lookup_va,
    input  wire [ASID_BITS-1:0]    lookup_asid,
    output reg                     lookup_hit,
    output reg  [`PA_BITS-1:0]     lookup_pa,
    output reg  [7:0]              lookup_flags,

    // 更新端口
    input  wire                    update_en,
    input  wire [`VA_BITS-1:0]     update_va,
    input  wire [`PA_BITS-1:0]     update_pa,
    input  wire [7:0]              update_flags,
    input  wire [ASID_BITS-1:0]    update_asid,
    input  wire                    update_global,

    // 失效端口
    input  wire                    invalidate_all,
    input  wire                    invalidate_asid_en,
    input  wire [ASID_BITS-1:0]    invalidate_asid,
    input  wire                    invalidate_va_en,
    input  wire [`VA_BITS-1:0]     invalidate_va,

    // 统计
    output reg  [7:0]              hit_count,
    output reg  [7:0]              miss_count
);

    // TLB条目结构
    localparam VPN_BITS = `VPN_BITS;
    localparam PFN_BITS = `PFN_BITS;

    reg [VPN_BITS-1:0]     tlb_vpn     [0:ENTRIES-1];
    reg [PFN_BITS-1:0]     tlb_pfn     [0:ENTRIES-1];
    reg [7:0]              tlb_flags   [0:ENTRIES-1];
    reg [ASID_BITS-1:0]    tlb_asid    [0:ENTRIES-1];
    reg                    tlb_valid   [0:ENTRIES-1];
    reg                    tlb_global  [0:ENTRIES-1];

    // PLRU状态 (树形PLRU，需要ENTRIES-1位)
    reg [ENTRIES-2:0]      plru_bits;

    // 查找结果
    wire [ENTRIES-1:0]     match_vec;
    wire                   hit;
    reg  [5:0]             hit_index;  // log2(64) = 6

    integer i;

    // 匹配逻辑 (组合逻辑)
    genvar g;
    generate
        for (g = 0; g < ENTRIES; g = g + 1) begin : gen_match
            assign match_vec[g] = tlb_valid[g] &&
                                   (tlb_vpn[g] == lookup_va[47:12]) &&
                                   (tlb_global[g] || (tlb_asid[g] == lookup_asid));
        end
    endgenerate

    assign hit = |match_vec;

    // 优先级编码器 (找到第一个匹配的索引)
    always @(*) begin
        hit_index = 0;
        for (i = 0; i < ENTRIES; i = i + 1) begin
            if (match_vec[i])
                hit_index = i[5:0];
        end
    end

    // 查找输出
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            lookup_hit <= 1'b0;
            hit_count <= 0;
            miss_count <= 0;
        end else begin
            lookup_hit <= lookup_en && hit;

            if (lookup_en) begin
                if (hit) begin
                    lookup_pa <= {tlb_pfn[hit_index], lookup_va[11:0]};
                    lookup_flags <= tlb_flags[hit_index];
                    hit_count <= hit_count + 1;
                end else begin
                    miss_count <= miss_count + 1;
                end
            end
        end
    end

    // PLRU更新
    always @(posedge clk) begin
        if (lookup_en && hit) begin
            // 更新PLRU树
            // 这是一个简化的PLRU更新
            // 实际实现需要完整的树形更新逻辑
            // 这里仅示意
        end
    end

    // 查找替换受害者 (使用PLRU)
    reg [5:0] victim_index;
    always @(*) begin
        // 简化的PLRU受害者选择
        // 实际应该遍历PLRU树
        victim_index = 0;
        for (i = 0; i < ENTRIES; i = i + 1) begin
            if (!tlb_valid[i]) begin
                victim_index = i[5:0];
            end
        end
    end

    // 更新和失效逻辑
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (i = 0; i < ENTRIES; i = i + 1) begin
                tlb_valid[i] <= 1'b0;
            end
        end else begin
            // 全局失效
            if (invalidate_all) begin
                for (i = 0; i < ENTRIES; i = i + 1) begin
                    tlb_valid[i] <= 1'b0;
                end
            end
            // ASID选择性失效
            else if (invalidate_asid_en) begin
                for (i = 0; i < ENTRIES; i = i + 1) begin
                    if (!tlb_global[i] && (tlb_asid[i] == invalidate_asid))
                        tlb_valid[i] <= 1'b0;
                end
            end
            // 虚拟地址选择性失效
            else if (invalidate_va_en) begin
                for (i = 0; i < ENTRIES; i = i + 1) begin
                    if (tlb_valid[i] &&
                        (tlb_vpn[i] == invalidate_va[47:12]) &&
                        (tlb_global[i] || (tlb_asid[i] == invalidate_asid)))
                        tlb_valid[i] <= 1'b0;
                end
            end
            // 正常更新
            else if (update_en) begin
                tlb_vpn[victim_index] <= update_va[47:12];
                tlb_pfn[victim_index] <= update_pa[51:12];
                tlb_flags[victim_index] <= update_flags;
                tlb_asid[victim_index] <= update_asid;
                tlb_global[victim_index] <= update_global;
                tlb_valid[victim_index] <= 1'b1;
            end
        end
    end

endmodule

// 模块: 组相联TLB
//////////////////////////////////////////////////////////////////////////////

module tlb_set_associative #(
    parameter WAYS = 4,
    parameter SETS = 16,
    parameter ASID_BITS = 12
)(
    input  wire                    clk,
    input  wire                    rst_n,

    // 查找端口
    input  wire                    lookup_en,
    input  wire [`VA_BITS-1:0]     lookup_va,
    input  wire [ASID_BITS-1:0]    lookup_asid,
    output reg                     lookup_hit,
    output reg  [`PA_BITS-1:0]     lookup_pa,
    output reg  [7:0]              lookup_flags,

    // 更新端口
    input  wire                    update_en,
    input  wire [`VA_BITS-1:0]     update_va,
    input  wire [`PA_BITS-1:0]     update_pa,
    input  wire [7:0]              update_flags,
    input  wire [ASID_BITS-1:0]    update_asid,
    input  wire                    update_global,

    // 失效端口
    input  wire                    invalidate_all,
    input  wire                    invalidate_asid_en,
    input  wire [ASID_BITS-1:0]    invalidate_asid,
    input  wire                    invalidate_va_en,
    input  wire [`VA_BITS-1:0]     invalidate_va,

    // 统计
    output reg  [7:0]              hit_count,
    output reg  [7:0]              miss_count
);

    localparam VPN_BITS = `VPN_BITS;
    localparam PFN_BITS = `PFN_BITS;
    localparam SET_BITS = $clog2(SETS);
    localparam TAG_BITS = VPN_BITS - SET_BITS;

    // 地址分解
    wire [TAG_BITS-1:0] lookup_tag = lookup_va[47:12+SET_BITS];
    wire [SET_BITS-1:0] lookup_set = lookup_va[12+SET_BITS-1:12];

    wire [TAG_BITS-1:0] update_tag = update_va[47:12+SET_BITS];
    wire [SET_BITS-1:0] update_set = update_va[12+SET_BITS-1:12];

    wire [TAG_BITS-1:0] invalidate_tag = invalidate_va[47:12+SET_BITS];
    wire [SET_BITS-1:0] invalidate_set_idx = invalidate_va[12+SET_BITS-1:12];

    // TLB存储数组
    // 使用二维数组: [set][way]
    reg [TAG_BITS-1:0]   tlb_tag   [0:SETS-1][0:WAYS-1];
    reg [PFN_BITS-1:0]   tlb_pfn   [0:SETS-1][0:WAYS-1];
    reg [7:0]            tlb_flags [0:SETS-1][0:WAYS-1];
    reg [ASID_BITS-1:0]  tlb_asid  [0:SETS-1][0:WAYS-1];
    reg                  tlb_valid [0:SETS-1][0:WAYS-1];
    reg                  tlb_global[0:SETS-1][0:WAYS-1];

    // PLRU状态 [set][way-1 bits]
    reg [WAYS-2:0]       plru      [0:SETS-1];

    // 查找逻辑 (组合逻辑)
    wire [WAYS-1:0] match_vec;
    wire            hit;
    reg  [1:0]      hit_way;

    genvar s, w;
    generate
        for (w = 0; w < WAYS; w = w + 1) begin : gen_way_match
            assign match_vec[w] = tlb_valid[lookup_set][w] &&
                                   (tlb_tag[lookup_set][w] == lookup_tag) &&
                                   (tlb_global[lookup_set][w] ||
                                    (tlb_asid[lookup_set][w] == lookup_asid));
        end
    endgenerate

    assign hit = |match_vec;

    // 优先级编码器
    integer wi;
    always @(*) begin
        hit_way = 0;
        for (wi = 0; wi < WAYS; wi = wi + 1) begin
            if (match_vec[wi])
                hit_way = wi[1:0];
        end
    end

    // 查找输出
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            lookup_hit <= 1'b0;
            hit_count <= 0;
            miss_count <= 0;
        end else begin
            lookup_hit <= lookup_en && hit;

            if (lookup_en) begin
                if (hit) begin
                    lookup_pa <= {tlb_pfn[lookup_set][hit_way], lookup_va[11:0]};
                    lookup_flags <= tlb_flags[lookup_set][hit_way];
                    hit_count <= hit_count + 1;
                end else begin
                    miss_count <= miss_count + 1;
                end
            end
        end
    end

    // PLRU更新
    integer s_idx;
    always @(posedge clk) begin
        if (lookup_en && hit) begin
            // 更新命中组的PLRU
            // 简化的伪LRU: 如果命中way 0，设置bit 0
            case (hit_way)
                2'd0: plru[lookup_set] <= {plru[lookup_set][1], 1'b0};
                2'd1: plru[lookup_set] <= {plru[lookup_set][1], 1'b1};
                2'd2: plru[lookup_set] <= {1'b0, plru[lookup_set][0]};
                2'd3: plru[lookup_set] <= {1'b1, plru[lookup_set][0]};
            endcase
        end
    end

    // 受害者选择
    reg [1:0] victim_way;
    always @(*) begin
        // 查找无效条目
        victim_way = 0;
        for (wi = 0; wi < WAYS; wi = wi + 1) begin
            if (!tlb_valid[update_set][wi])
                victim_way = wi[1:0];
        end

        // 否则使用PLRU
        // 简化的PLRU决策
        if (victim_way == 0) begin
            if (plru[update_set][0] == 0)
                victim_way = (plru[update_set][1] == 0) ? 2'd2 : 2'd3;
            else
                victim_way = (plru[update_set][1] == 0) ? 2'd0 : 2'd1;
        end
    end

    // 更新和失效逻辑
    integer si, wi;
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (si = 0; si < SETS; si = si + 1) begin
                for (wi = 0; wi < WAYS; wi = wi + 1) begin
                    tlb_valid[si][wi] <= 1'b0;
                end
            end
        end else begin
            // 全局失效
            if (invalidate_all) begin
                for (si = 0; si < SETS; si = si + 1) begin
                    for (wi = 0; wi < WAYS; wi = wi + 1) begin
                        tlb_valid[si][wi] <= 1'b0;
                    end
                end
            end
            // 正常更新
            else if (update_en) begin
                tlb_tag[update_set][victim_way] <= update_tag;
                tlb_pfn[update_set][victim_way] <= update_pa[51:12];
                tlb_flags[update_set][victim_way] <= update_flags;
                tlb_asid[update_set][victim_way] <= update_asid;
                tlb_global[update_set][victim_way] <= update_global;
                tlb_valid[update_set][victim_way] <= 1'b1;
            end
        end
    end

endmodule

`endif // TLB_V
```
