# MMU架构 (Memory Management Unit Architecture)

## 1. 概述

内存管理单元(MMU, Memory Management Unit)是现代处理器中负责虚拟内存管理的关键硬件组件。它位于CPU核心与内存之间，实现了虚拟地址到物理地址的动态转换，同时提供内存保护、缓存控制等关键功能。

### 1.1 MMU的核心使命

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        MMU的核心使命                                         │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│   ┌──────────────┐         ┌──────────────┐         ┌──────────────┐       │
│   │   虚拟地址    │ ──────> │     MMU      │ ──────> │   物理地址    │       │
│   │  (VA/虚拟)   │  转换   │  (地址翻译)   │  转换   │  (PA/物理)   │       │
│   └──────────────┘         └──────────────┘         └──────────────┘       │
│          │                        │                        │               │
│          ▼                        ▼                        ▼               │
│   ┌──────────────┐         ┌──────────────┐         ┌──────────────┐       │
│   │  进程隔离    │         │  内存保护    │         │  按需分页    │       │
│   │  (Isolation) │         │ (Protection) │         │  (Demand Pg) │       │

└─────────────────────────────────────────────────────────────────────────────┘
```

MMU的三大核心功能：

1. **地址转换**：将虚拟地址(Virtual Address)映射到物理地址(Physical Address)
2. **内存保护**：通过页表项中的权限位控制访问权限（读/写/执行）
3. **缓存管理**：控制内存访问的缓存属性（Cacheable/Non-cacheable）

### 1.2 为什么需要MMU

在没有MMU的系统中，程序直接使用物理地址访问内存：

```c
// 无MMU系统的问题示例
// 程序A和程序B都尝试访问地址0x1000

// 程序A
int* ptr_a = (int*)0x1000;  // 直接使用物理地址
*ptr_a = 100;                // 写入物理内存0x1000

// 程序B
int* ptr_b = (int*)0x1000;  // 同样的物理地址
*ptr_b = 200;                // 覆盖程序A的数据！

// 问题：
// 1. 没有内存隔离 - 程序可以互相破坏
// 2. 无法运行比物理内存大的程序
// 3. 内存碎片无法解决
// 4. 程序无法在内存中移动
```

有了MMU后，每个进程拥有独立的虚拟地址空间：

```c
// 有MMU系统的解决方案
// 程序A和程序B使用相同的虚拟地址，但映射到不同物理地址

// 程序A的页表：VA 0x1000 -> PA 0x100000
// 程序B的页表：VA 0x1000 -> PA 0x200000

int* ptr_a = (int*)0x1000;  // 虚拟地址
*ptr_a = 100;                // 实际写入物理地址0x100000

int* ptr_b = (int*)0x1000;  // 相同的虚拟地址
*ptr_b = 200;                // 实际写入物理地址0x200000，互不干扰！
```

```c
/*
 * 从C语言程序员的角度看地址转换
 * 这些代码展示了虚拟地址如何在MMU作用下映射到物理地址

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/mman.h>

/*
 * 虚拟地址结构 (从软件视角)
 * 注意：这只是概念模型，实际MMU在硬件中实现
 */
typedef struct {
    uint64_t unused  : 16;  // 高位未使用 (符号扩展)
    uint64_t pml4    : 9;   // PML4索引
    uint64_t pdpt    : 9;   // PDPT索引
    uint64_t pd      : 9;   // PD索引
    uint64_t pt      : 9;   // PT索引
    uint64_t offset  : 12;  // 页内偏移
} virtual_addr_t;

/*
 * 页表项结构 (x86-64, 64位)
 */
typedef union {
    uint64_t value;
    struct {
        uint64_t present    : 1;   // P - 存在位
        uint64_t rw         : 1;   // R/W - 读写位
        uint64_t us         : 1;   // U/S - 用户/超级用户
        uint64_t pwt        : 1;   // PWT - 页级写透
        uint64_t pcd        : 1;   // PCD - 页级缓存禁用
        uint64_t accessed   : 1;   // A - 访问位
        uint64_t dirty      : 1;   // D - 脏位
        uint64_t ps         : 1;   // PS - 页大小
        uint64_t global     : 1;   // G - 全局页
        uint64_t ignored    : 3;   // 忽略
        uint64_t pfn        : 40;  // 页框号 (物理地址[51:12])
        uint64_t reserved   : 11;  // 保留
        uint64_t nx         : 1;   // XD/NX - 执行禁用
    } bits;
} page_table_entry_t;

/*
 * 演示：C程序如何"感知"虚拟内存
 */
void demonstrate_virtual_memory(void) {
    int local_var;           // 栈变量
    static int static_var;   // BSS段
    int* heap_var;           // 堆变量

    heap_var = malloc(sizeof(int));

    printf("=== C程序中的虚拟地址 ===\n");
    printf("代码段地址:   %p\n", (void*)demonstrate_virtual_memory);
    printf("栈变量地址:   %p\n", (void*)&local_var);
    printf("静态变量地址: %p\n", (void*)&static_var);
    printf("堆变量地址:   %p\n", (void*)heap_var);

    /*
     * 输出示例 (x86-64 Linux):
     * 代码段地址:   0x55a3b2c0c1a9  (高地址 - 接近0xFFFFFFFF)
     * 栈变量地址:   0x7ffcc2a8f4ac  (高地址 - 向下增长)
     * 静态变量地址: 0x55a3b2c0d024  (中等地址)
     * 堆变量地址:   0x55a3b3e2a2a0  (中等地址 - 向上增长)
     *
     * 注意：这些都是虚拟地址！
     * 实际的物理地址只有MMU知道
     */

    free(heap_var);
}

/*
 * 读取进程的页表信息 (通过/proc/[pid]/pagemap)
 * 需要root权限
 */
int get_physical_address(uintptr_t virtual_addr, uint64_t* physical_addr) {
    int fd = open("/proc/self/pagemap", O_RDONLY);
    if (fd < 0) {
        perror("open pagemap");
        return -1;
    }

    // 计算pagemap中的偏移
    // 每个页对应一个64位条目
    uint64_t page_size = 4096;
    uint64_t vpn = virtual_addr / page_size;
    uint64_t offset = vpn * sizeof(uint64_t);

    if (lseek(fd, offset, SEEK_SET) < 0) {
        perror("lseek");
        close(fd);
        return -1;
    }

    uint64_t entry;
    if (read(fd, &entry, sizeof(entry)) != sizeof(entry)) {
        perror("read");
        close(fd);
        return -1;
    }

    close(fd);

    // 检查页是否存在
    if (!(entry & (1ULL << 63))) {
        printf("Page not present in memory\n");
        return -1;
    }

    // 提取页框号
    uint64_t pfn = entry & ((1ULL << 55) - 1);
    uint64_t page_offset = virtual_addr % page_size;

    *physical_addr = (pfn * page_size) + page_offset;
    return 0;
}

/*
 * 演示：获取虚拟地址对应的物理地址
 */
void show_address_translation(void) {
    int test_var = 0x12345678;
    uintptr_t virtual_addr = (uintptr_t)&test_var;
    uint64_t physical_addr;

    printf("\n=== 地址转换演示 ===\n");
    printf("虚拟地址: 0x%016lx\n", virtual_addr);

    if (get_physical_address(virtual_addr, &physical_addr) == 0) {
        printf("物理地址: 0x%016lx\n", physical_addr);
    }
}

/*
 * 模拟MMU地址转换过程 (概念性代码)
 */
typedef struct {
    uint64_t pml4_base;  // PML4物理基址
} mmu_state_t;

/*
 * 模拟四级页表遍历
 * 实际MMU在硬件中完成，这里仅用于理解
 */
uint64_t mmu_translate(mmu_state_t* mmu, uint64_t virtual_addr) {
    // 分解虚拟地址
    uint64_t pml4_index = (virtual_addr >> 39) & 0x1FF;
    uint64_t pdpt_index = (virtual_addr >> 30) & 0x1FF;
    uint64_t pd_index   = (virtual_addr >> 21) & 0x1FF;
    uint64_t pt_index   = (virtual_addr >> 12) & 0x1FF;
    uint64_t offset     = virtual_addr & 0xFFF;

    printf("\n=== MMU地址转换过程 (模拟) ===\n");
    printf("虚拟地址: 0x%016lx\n", virtual_addr);
    printf("  PML4 index: %3lu (0x%03lx)\n", pml4_index, pml4_index);
    printf("  PDPT index: %3lu (0x%03lx)\n", pdpt_index, pdpt_index);
    printf("  PD   index: %3lu (0x%03lx)\n", pd_index, pd_index);
    printf("  PT   index: %3lu (0x%03lx)\n", pt_index, pt_index);
    printf("  Page offset: 0x%03lx\n", offset);

    // 步骤1: 读取PML4项
    uint64_t pml4_entry_phys = mmu->pml4_base + (pml4_index * 8);
    printf("\n步骤1: 读取PML4项\n");
    printf("  PML4物理基址: 0x%lx\n", mmu->pml4_base);
    printf("  PML4项物理地址: 0x%lx\n", pml4_entry_phys);

    // 步骤2-4: 类似地遍历PDPT、PD、PT...
    // 这里省略具体的内存读取操作
```

### 5.2 特权级与内存访问权限

```c
/*
 * 特权级与内存访问权限的关系
 * 展示不同特权级如何使用MMU
 */

#include <stdio.h>
#include <stdint.h>

/*
 * x86-64页表项中的U/S位控制用户/超级用户访问
 */
#define PTE_PRESENT     (1ULL << 0)
#define PTE_RW          (1ULL << 1)   // 读写位
#define PTE_US          (1ULL << 2)   // 用户/超级用户位
#define PTE_ACCESSED    (1ULL << 5)
#define PTE_DIRTY       (1ULL << 6)
#define PTE_GLOBAL      (1ULL << 8)
#define PTE_NX          (1ULL << 63)  // 执行禁用

/*
 * 页表项权限组合
 */
typedef enum {
    // 内核页 (U/S=0): 只有Ring 0-2可以访问
    KERNEL_RW = PTE_PRESENT | PTE_RW,           // 内核读写
    KERNEL_RO = PTE_PRESENT,                     // 内核只读
    KERNEL_RX = PTE_PRESENT | PTE_RW,            // 内核可执行 (NX=0)

    // 用户页 (U/S=1): Ring 3可以访问
    USER_RW   = PTE_PRESENT | PTE_RW | PTE_US,  // 用户读写
    USER_RO   = PTE_PRESENT | PTE_US,            // 用户只读
    USER_RX   = PTE_PRESENT | PTE_US,            // 用户可执行 (NX=0)
    USER_NONE = PTE_PRESENT | PTE_US,            // 用户存在但不可访问

    // 不可执行页 (NX=1)
    KERNEL_DATA = PTE_PRESENT | PTE_RW | PTE_NX,       // 内核数据
    USER_DATA   = PTE_PRESENT | PTE_RW | PTE_US | PTE_NX, // 用户数据
} pte_permissions_t;

/*
 * 特权级检查伪代码
 */
int check_access_privilege(uint64_t pte, int cpl, int access_type) {
    /*
     * cpl: Current Privilege Level (0=内核, 3=用户)
     * access_type: 0=读, 1=写, 2=执行
     */

    int present = pte & PTE_PRESENT;
    int rw = pte & PTE_RW;
    int us = pte & PTE_US;
    int nx = pte & PTE_NX;

    // 检查存在位
    if (!present) {
        return -1;  // 页不存在异常
    }

    // 检查用户/超级用户权限
    if (cpl == 3) {
        // 用户模式访问
        if (!us) {
            return -2;  // 特权级违规 - 用户访问内核页
        }
    }
    // 内核模式可以访问所有页 (SMAP除外)

    // 检查读写权限
    if (access_type == 1 && !rw) {
        return -3;  // 写保护异常
    }

    // 检查执行权限
    if (access_type == 2 && nx) {
        return -4;  // 执行禁用异常
    }

    return 0;  // 访问允许
}

/*
 * C语言中的特权级概念
 * 虽然C代码运行在特定特权级，但可以通过系统调用切换
 */
void privilege_level_example(void) {
    int stack_var = 42;

    printf("在用户模式 (Ring 3 / EL0 / U-mode) 运行\n");
    printf("栈变量地址: %p\n", (void*)&stack_var);

    /*
     * 尝试访问内核地址会导致段错误
     * 因为MMU会阻止用户模式访问内核页
     */
    // int *kernel_ptr = (int*)0xFFFFFFFF80000000ULL;
    // *kernel_ptr = 0;  // 这将触发Page Fault!

    /*
     * 系统调用会切换到内核模式
     * 例如: write() 系统调用
     */
    write(1, "Hello\n", 6);  // 内部执行 syscall 指令

    /*
     * syscall指令执行流程:
     * 1. 用户程序在Ring 3执行syscall
     * 2. CPU切换到Ring 0 (内核模式)
     * 3. 进入内核syscall入口
     * 4. 内核执行write操作 (可以访问所有内存)
     * 5. sysret指令返回Ring 3
     */
}

/*
 * Linux内核中的特权级处理
 */
#ifdef __KERNEL__

/*
 * 内核中检查当前特权级
 */
static inline int current_privilege_level(void) {
    // 在Linux内核中，我们总是在内核模式运行
    // 但用户上下文通过thread_info区分
    return user_mode(current_pt_regs()) ? 3 : 0;
```

```c
int do_page_fault(struct pt_regs *regs, unsigned long error_code,
                  unsigned long address) {
    struct vm_area_struct *vma;
    struct mm_struct *mm;
    int fault;

    mm = current->mm;

    /*
     * error_code位定义:
     * bit 0: 0=页不存在, 1=保护错误
     * bit 1: 0=读访问, 1=写访问
     * bit 2: 0=内核模式, 1=用户模式
     * bit 3: 0=普通访问, 1=保留位违规
     * bit 4: 0=取指令, 1=数据访问
     */

    // 查找对应的VMA (虚拟内存区域)
    vma = find_vma(mm, address);
    if (!vma) {
        return SIGSEGV;  // 段错误 - 访问无效地址
    }

    // 检查访问权限
    if (!(vma->vm_flags & VM_READ) && (error_code & PF_WRITE)) {
        return SIGSEGV;  // 尝试写入只读区域
    }

    if (!(vma->vm_flags & VM_WRITE) && (error_code & PF_WRITE)) {
        return SIGSEGV;  // 写入无写权限区域
    }

    // 处理缺页
    fault = handle_mm_fault(vma, address,
                            error_code & PF_WRITE ? FAULT_FLAG_WRITE : 0);

    if (fault & VM_FAULT_OOM) {
        return SIGKILL;  // OOM killer
    }

    return 0;
}

/*
 * 处理缺页 (需求分页)
 */
int handle_mm_fault(struct vm_area_struct *vma, unsigned long address,
                    unsigned int flags) {
    pgd_t *pgd;
    p4d_t *p4d;
    pud_t *pud;
    pmd_t *pmd;
    pte_t *pte;

    pgd = pgd_offset(vma->vm_mm, address);
    p4d = p4d_alloc(vma->vm_mm, pgd, address);
    if (!p4d)
        return VM_FAULT_OOM;

    pud = pud_alloc(vma->vm_mm, p4d, address);
    if (!pud)
        return VM_FAULT_OOM;

    pmd = pmd_alloc(vma->vm_mm, pud, address);
    if (!pmd)
        return VM_FAULT_OOM;

    // 处理PTE级别的缺页
    return handle_pte_fault(vma, address, pte, pmd, flags);
}
```

```c
    void *addr = mmap(NULL, size, PROT_READ | PROT_WRITE,

    if (setjmp(jump_buffer) == 0) {
```

```verilog
    reg  [`VA_BITS-1:0]     tlb_update_vaddr;
    reg  [`PA_BITS-1:0]     tlb_update_paddr;
    reg  [7:0]              tlb_update_flags;
    reg                     tlb_update_cacheable;

    reg                     ptw_start;
    reg  [`VA_BITS-1:0]     ptw_vaddr;
    reg  [`PA_BITS-1:0]     ptw_pml4_base;

    wire                    ptw_fault;
    wire [3:0]              ptw_fault_code;
    wire [`PA_BITS-1:0]     ptw_paddr;
    wire [7:0]              ptw_flags;
    wire                    ptw_cacheable;

    //////////////////////////////////////////////////////////////////////////
    // 状态机
    //////////////////////////////////////////////////////////////////////////

    localparam STATE_IDLE       = 3'b000;
    localparam STATE_TLB_LOOKUP = 3'b001;
    localparam STATE_TLB_HIT    = 3'b010;
    localparam STATE_PTW_WAIT   = 3'b011;
    localparam STATE_PTW_DONE   = 3'b100;
    localparam STATE_RESP       = 3'b101;

    reg [2:0] state, next_state;

    // 状态寄存器
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            state <= STATE_IDLE;
        else
            state <= next_state;
    end

    // 次态逻辑
    always @(*) begin
        next_state = state;

        case (state)
            STATE_IDLE: begin
                if (req_valid && paging_enable)
                    next_state = STATE_TLB_LOOKUP;
                else if (req_valid && !paging_enable)
                    next_state = STATE_RESP;  // 分页禁用，直接透传
            end

            STATE_TLB_LOOKUP: begin
                // 一个周期完成TLB查找
                next_state = tlb_hit ? STATE_TLB_HIT : STATE_PTW_WAIT;
            end

            STATE_TLB_HIT: begin
                // 检查权限
                next_state = STATE_RESP;
            end

            STATE_PTW_WAIT: begin
                if (ptw_done)
                    next_state = STATE_PTW_DONE;
            end

            STATE_PTW_DONE: begin
                next_state = STATE_RESP;
            end

            STATE_RESP: begin
                if (resp_ready)
                    next_state = STATE_IDLE;
            end

            default: next_state = STATE_IDLE;
        endcase
    end

    //////////////////////////////////////////////////////////////////////////
    // 输出逻辑
    //////////////////////////////////////////////////////////////////////////

    // 请求就绪
    assign req_ready = (state == STATE_IDLE);

    // TLB查找控制
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            tlb_lookup_en <= 1'b0;
        end else begin
            tlb_lookup_en <= (state == STATE_IDLE && req_valid && paging_enable);
            if (tlb_lookup_en) begin
                tlb_lookup_vaddr <= req_vaddr;
                tlb_lookup_asid <= current_asid;
            end
        end
    end

    // PTW启动
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            ptw_start <= 1'b0;
        end else begin
            ptw_start <= (state == STATE_TLB_LOOKUP && !tlb_hit);
            if (ptw_start) begin
                ptw_vaddr <= tlb_lookup_vaddr;
                ptw_pml4_base <= cr3_pml4_base;
            end
        end
    end

    // 响应生成
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            resp_valid <= 1'b0;
            resp_page_fault <= 1'b0;
        end else begin
            case (state)
                STATE_TLB_HIT: begin
                    resp_valid <= 1'b1;
                    resp_paddr <= {tlb_paddr[`PA_BITS-1:`PAGE_SHIFT],
                                   tlb_lookup_vaddr[`PAGE_SHIFT-1:0]};
                    resp_cacheable <= tlb_cacheable;
                    // 权限检查
                    resp_page_fault <= check_permission(
                        tlb_flags, current_pl,
                        req_access, req_instruction, nx_enable
                    );
                end

                STATE_PTW_DONE: begin
                    resp_valid <= 1'b1;
                    if (ptw_fault) begin
                        resp_page_fault <= 1'b1;
                        resp_fault_code <= ptw_fault_code;
                    end else begin
                        resp_paddr <= {ptw_paddr[`PA_BITS-1:`PAGE_SHIFT],
                                       ptw_vaddr[`PAGE_SHIFT-1:0]};
                        resp_cacheable <= ptw_cacheable;
                        resp_page_fault <= check_permission(
                            ptw_flags, current_pl,
                            req_access, req_instruction, nx_enable
                        );
                        // 更新TLB
                        tlb_update_en <= !ptw_fault;
                        tlb_update_vaddr <= ptw_vaddr;
                        tlb_update_paddr <= ptw_paddr;
                        tlb_update_flags <= ptw_flags;
                        tlb_update_cacheable <= ptw_cacheable;
                    end
                end

                STATE_RESP: begin
                    if (resp_ready) begin
                        resp_valid <= 1'b0;
                        resp_page_fault <= 1'b0;
                        tlb_update_en <= 1'b0;
                    end
                end
            endcase
        end
    end

    //////////////////////////////////////////////////////////////////////////
    // 权限检查函数
    //////////////////////////////////////////////////////////////////////////

    function check_permission;
        input [7:0]  pte_flags;
        input [1:0]  cpl;
        input [1:0]  access_type;  // 00=读, 01=写, 10=执行
        input        is_instruction;
        input        nx_enabled;

        reg present;
        reg user_accessible;
        reg writable;
        reg no_execute;

        begin
            present = pte_flags[`PTE_P];
            user_accessible = pte_flags[`PTE_US];
            writable = pte_flags[`PTE_RW];
            no_execute = nx_enabled && pte_flags[`PTE_NX];

            // 默认允许
            check_permission = 1'b0;

            // 检查存在位
            if (!present)
                check_permission = 1'b1;

            // 用户模式只能访问用户页
            else if ((cpl == 2'b11) && !user_accessible)
                check_permission = 1'b1;

            // 检查写权限
            else if (access_type == 2'b01 && !writable)
                check_permission = 1'b1;

            // 检查执行权限 (NX)
            else if ((access_type == 2'b10 || is_instruction) && no_execute)
                check_permission = 1'b1;
        end
    endfunction

    //////////////////////////////////////////////////////////////////////////
    // 子模块实例化
    //////////////////////////////////////////////////////////////////////////

    // TLB模块
    tlb #(
        .ENTRIES(`TLB_ENTRIES),
        .ASID_BITS(`ASID_BITS)
    ) u_tlb (
        .clk(clk),
        .rst_n(rst_n),

        .lookup_en(tlb_lookup_en),
        .lookup_vaddr(tlb_lookup_vaddr),
        .lookup_asid(tlb_lookup_asid),
        .lookup_hit(tlb_hit),
        .lookup_paddr(tlb_paddr),
        .lookup_flags(tlb_flags),
        .lookup_cacheable(tlb_cacheable),

        .update_en(tlb_update_en),
        .update_vaddr(tlb_update_vaddr),
        .update_paddr(tlb_update_paddr),
        .update_flags(tlb_update_flags),
        .update_cacheable(tlb_update_cacheable),
        .update_asid(current_asid),

        .flush_all(tlb_flush_all),
        .flush_vaddr(tlb_flush_vaddr),
        .flush_addr(tlb_flush_addr),
        .flush_asid(tlb_flush_asid)
    );

    // 页表遍历器模块
    page_table_walker u_ptw (
        .clk(clk),
        .rst_n(rst_n),

        .ptw_idle(ptw_idle),
        .ptw_start(ptw_start),
        .ptw_vaddr(ptw_vaddr),
        .ptw_pml4_base(ptw_pml4_base),

        .ptw_done(ptw_done),
        .ptw_fault(ptw_fault),
        .ptw_fault_code(ptw_fault_code),
        .ptw_paddr(ptw_paddr),
        .ptw_flags(ptw_flags),
        .ptw_cacheable(ptw_cacheable),

        .mem_req_valid(mem_req_valid),
        .mem_req_ready(mem_req_ready),
        .mem_req_addr(mem_req_addr),
        .mem_req_size(mem_req_size),

        .mem_resp_valid(mem_resp_valid),
        .mem_resp_ready(mem_resp_ready),
        .mem_resp_data(mem_resp_data)
    );

endmodule

//////////////////////////////////////////////////////////////////////////////
// 模块: TLB (Translation Lookaside Buffer)
//////////////////////////////////////////////////////////////////////////////

module tlb #(
    parameter ENTRIES = 64,
    parameter ASID_BITS = 12
)(
    input  wire                    clk,
    input  wire                    rst_n,

    // 查找端口
    input  wire                    lookup_en,
    input  wire [47:0]             lookup_vaddr,
    input  wire [ASID_BITS-1:0]    lookup_asid,
    output reg                     lookup_hit,
    output reg  [51:0]             lookup_paddr,
    output reg  [7:0]              lookup_flags,
    output reg                     lookup_cacheable,

    // 更新端口
    input  wire                    update_en,
    input  wire [47:0]             update_vaddr,
    input  wire [51:0]             update_paddr,
    input  wire [7:0]              update_flags,
    input  wire                    update_cacheable,
    input  wire [ASID_BITS-1:0]    update_asid,

    // 刷新端口
    input  wire                    flush_all,
    input  wire                    flush_vaddr,
    input  wire [47:0]             flush_addr,
    input  wire [ASID_BITS-1:0]    flush_asid
);

    // TLB条目结构
    // VPN[47:12] + PFN[51:12] + FLAGS[7:0] + ASID + V
    localparam VPN_BITS = 36;  // 48-12
    localparam PFN_BITS = 40;  // 52-12

    reg [VPN_BITS-1:0]     tlb_vpn    [0:ENTRIES-1];
    reg [PFN_BITS-1:0]     tlb_pfn    [0:ENTRIES-1];
    reg [7:0]              tlb_flags  [0:ENTRIES-1];
    reg                    tlb_v      [0:ENTRIES-1];
    reg                    tlb_global [0:ENTRIES-1];
    reg [ASID_BITS-1:0]    tlb_asid   [0:ENTRIES-1];
    reg                    tlb_cache  [0:ENTRIES-1];

    // LRU替换指针
    reg [$clog2(ENTRIES)-1:0] lru_ptr;

    // 查找逻辑 (组合逻辑)
    integer i;
    reg hit;
    reg [$clog2(ENTRIES)-1:0] hit_idx;

    always @(*) begin
        hit = 1'b0;
        hit_idx = 0;

        for (i = 0; i < ENTRIES; i = i + 1) begin
            if (tlb_v[i] &&
                (tlb_vpn[i] == lookup_vaddr[47:12]) &&
                (tlb_global[i] || (tlb_asid[i] == lookup_asid))) begin
                hit = 1'b1;
                hit_idx = i;
            end
        end
    end

    // 查找输出
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            lookup_hit <= 1'b0;
        end else if (lookup_en) begin
            lookup_hit <= hit;
            if (hit) begin
                lookup_paddr <= {tlb_pfn[hit_idx], lookup_vaddr[11:0]};
                lookup_flags <= tlb_flags[hit_idx];
                lookup_cacheable <= tlb_cache[hit_idx];
            end
        end
    end

    // 更新逻辑
    integer j;
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (j = 0; j < ENTRIES; j = j + 1) begin
                tlb_v[j] <= 1'b0;
            end
            lru_ptr <= 0;
        end else begin
            if (update_en) begin
                // 查找空闲条目或使用LRU
                tlb_vpn[lru_ptr] <= update_vaddr[47:12];
                tlb_pfn[lru_ptr] <= update_paddr[51:12];
                tlb_flags[lru_ptr] <= update_flags;
                tlb_v[lru_ptr] <= 1'b1;
                tlb_global[lru_ptr] <= update_flags[`PTE_G];
                tlb_asid[lru_ptr] <= update_asid;
                tlb_cache[lru_ptr] <= update_cacheable;

                lru_ptr <= lru_ptr + 1;
            end

            if (flush_all) begin
                for (j = 0; j < ENTRIES; j = j + 1) begin
                    tlb_v[j] <= 1'b0;
                end
            end else if (flush_vaddr) begin
                for (j = 0; j < ENTRIES; j = j + 1) begin
                    if (tlb_v[j] &&
                        (tlb_vpn[j] == flush_addr[47:12]) &&
                        ((tlb_asid[j] == flush_asid) || tlb_global[j])) begin
                        tlb_v[j] <= 1'b0;
                    end
                end
            end
        end
    end

endmodule

//////////////////////////////////////////////////////////////////////////////
// 模块: Page Table Walker (页表遍历器)
//////////////////////////////////////////////////////////////////////////////

module page_table_walker (
    input  wire                    clk,
    input  wire                    rst_n,

    // 控制接口
    output wire                    ptw_idle,
    input  wire                    ptw_start,
    input  wire [47:0]             ptw_vaddr,
    input  wire [51:0]             ptw_pml4_base,

    // 结果接口
    output reg                     ptw_done,
    output reg                     ptw_fault,
    output reg  [3:0]              ptw_fault_code,
    output reg  [51:0]             ptw_paddr,
    output reg  [7:0]              ptw_flags,
    output reg                     ptw_cacheable,

    // 内存接口
    output reg                     mem_req_valid,
    input  wire                    mem_req_ready,
    output reg  [51:0]             mem_req_addr,
    output reg  [2:0]              mem_req_size,

    input  wire                    mem_resp_valid,
    output wire                    mem_resp_ready,
    input  wire [63:0]             mem_resp_data
);

    // 状态定义
    localparam PTW_IDLE     = 3'd0;
    localparam PTW_PML4     = 3'd1;
    localparam PTW_PDPT     = 3'd2;
    localparam PTW_PD       = 3'd3;
    localparam PTW_PT       = 3'd4;
    localparam PTW_DONE     = 3'd5;
    localparam PTW_FAULT    = 3'd6;

    reg [2:0] ptw_state;

    // 虚拟地址分解
    wire [8:0]  pml4_idx = ptw_vaddr[47:39];
    wire [8:0]  pdpt_idx = ptw_vaddr[38:30];
    wire [8:0]  pd_idx   = ptw_vaddr[29:21];
    wire [8:0]  pt_idx   = ptw_vaddr[20:12];

    // 页表项寄存器
    reg [63:0] pte_reg;

    // 计算下一级页表地址
    function [51:0] calc_next_table_addr;
        input [51:0] base;
        input [8:0]  index;
        begin
            calc_next_table_addr = base + (index << 3);  // ×8
        end
    endfunction

    // 状态机
    assign ptw_idle = (ptw_state == PTW_IDLE);
    assign mem_resp_ready = 1'b1;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            ptw_state <= PTW_IDLE;
            ptw_done <= 1'b0;
            ptw_fault <= 1'b0;
            mem_req_valid <= 1'b0;
        end else begin
            case (ptw_state)
                PTW_IDLE: begin
                    ptw_done <= 1'b0;
                    ptw_fault <= 1'b0;
                    if (ptw_start) begin
                        ptw_state <= PTW_PML4;
                        mem_req_addr <= calc_next_table_addr(ptw_pml4_base, pml4_idx);
                        mem_req_size <= 3'b011;  // 8字节
                        mem_req_valid <= 1'b1;
                    end
                end

                PTW_PML4: begin
                    if (mem_req_ready)
                        mem_req_valid <= 1'b0;

                    if (mem_resp_valid) begin
                        pte_reg <= mem_resp_data;
                        if (!mem_resp_data[`PTE_P]) begin
                            // 页不存在
                            ptw_fault <= 1'b1;
                            ptw_fault_code <= 4'b0000;  // P=0
                            ptw_state <= PTW_FAULT;
                        end else if (mem_resp_data[`PTE_PS]) begin
                            // 1GB大页
                            ptw_paddr <= {mem_resp_data[51:30], ptw_vaddr[29:0]};
                            ptw_flags <= mem_resp_data[7:0];
                            ptw_state <= PTW_DONE;
                        end else begin
                            // 继续遍历
                            ptw_state <= PTW_PDPT;
                            mem_req_addr <= calc_next_table_addr(
                                mem_resp_data[51:12], pdpt_idx);
                            mem_req_valid <= 1'b1;
                        end
                    end
                end

                PTW_PDPT: begin
                    if (mem_req_ready)
                        mem_req_valid <= 1'b0;

                    if (mem_resp_valid) begin
                        pte_reg <= mem_resp_data;
                        if (!mem_resp_data[`PTE_P]) begin
                            ptw_fault <= 1'b1;
                            ptw_fault_code <= 4'b0000;
                            ptw_state <= PTW_FAULT;
                        end else if (mem_resp_data[`PTE_PS]) begin
                            // 2MB大页
                            ptw_paddr <= {mem_resp_data[51:21], ptw_vaddr[20:0]};
                            ptw_flags <= mem_resp_data[7:0];
                            ptw_state <= PTW_DONE;
                        end else begin
                            ptw_state <= PTW_PD;
                            mem_req_addr <= calc_next_table_addr(
                                mem_resp_data[51:12], pd_idx);
                            mem_req_valid <= 1'b1;
                        end
                    end
                end

                PTW_PD: begin
                    if (mem_req_ready)
                        mem_req_valid <= 1'b0;

                    if (mem_resp_valid) begin
                        pte_reg <= mem_resp_data;
                        if (!mem_resp_data[`PTE_P]) begin
                            ptw_fault <= 1'b1;
                            ptw_fault_code <= 4'b0000;
                            ptw_state <= PTW_FAULT;
                        end else if (mem_resp_data[`PTE_PS]) begin
                            // 2MB大页 (在某些架构中)
                            ptw_paddr <= {mem_resp_data[51:21], ptw_vaddr[20:0]};
                            ptw_flags <= mem_resp_data[7:0];
                            ptw_state <= PTW_DONE;
                        end else begin
                            ptw_state <= PTW_PT;
                            mem_req_addr <= calc_next_table_addr(
                                mem_resp_data[51:12], pt_idx);
                            mem_req_valid <= 1'b1;
                        end
                    end
                end

                PTW_PT: begin
                    if (mem_req_ready)
                        mem_req_valid <= 1'b0;

                    if (mem_resp_valid) begin
                        pte_reg <= mem_resp_data;
                        if (!mem_resp_data[`PTE_P]) begin
                            ptw_fault <= 1'b1;
                            ptw_fault_code <= 4'b0000;
                            ptw_state <= PTW_FAULT;
                        end else begin
                            // 4KB页
                            ptw_paddr <= {mem_resp_data[51:12], ptw_vaddr[11:0]};
                            ptw_flags <= mem_resp_data[7:0];
                            ptw_state <= PTW_DONE;
                        end
                    end
                end

                PTW_DONE: begin
                    ptw_done <= 1'b1;
                    // 检查并设置缓存属性
                    ptw_cacheable <= !(pte_reg[`PTE_PCD] || pte_reg[`PTE_PWT]);
                    ptw_state <= PTW_IDLE;
                end

                PTW_FAULT: begin
                    ptw_done <= 1'b1;
                    ptw_fault <= 1'b1;
                    ptw_state <= PTW_IDLE;
                end

                default: ptw_state <= PTW_IDLE;
            endcase
        end
    end

endmodule

`endif // MMU_V
```
