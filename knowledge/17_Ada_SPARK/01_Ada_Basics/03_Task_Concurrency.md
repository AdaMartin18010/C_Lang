﻿# Ada任务与并发编程

## 1. Ada并发模型概述

### 1.1 与C语言的对比

| 特性 | C (pthreads) | Ada任务 |
|------|-------------|---------|
| 线程创建 | `pthread_create` | `task`类型声明 |
| 同步 | 互斥锁/条件变量 | `protected`对象 |
| 通信 | 共享内存+显式锁 | 会合(Rendezvous) |
| 类型安全 | 无 | 强类型检查 |
| 死锁检测 | 无 | 部分支持 |
| 实时支持 | 需额外库 | 原生Ravenscar Profile |

### 1.2 核心优势

```ada
-- Ada并发是语言原生特性，不是库附加
-- 编译器可检查并发错误

-- 任务声明即创建，类型安全
type Worker_Task is task;  -- 任务类型声明
task Server;               -- 单任务声明

-- 受保护对象：内置互斥，自动同步
protected Shared_Data is
   procedure Write (Value : Integer);
   function Read return Integer;
private
   Data : Integer := 0;
end Shared_Data;
```

## 2. 任务（Task）基础

### 2.1 任务声明与体

```ada
-- 任务规范（接口声明）
task type Worker_Task (ID : Integer);  -- 判别式参数

-- 任务体（实现）
task body Worker_Task is
   Local_ID : Integer := ID;
begin
   Put_Line ("Worker " & Integer'Image (Local_ID) & " started");

   -- 任务主循环
   loop
      -- 执行工作
      delay 1.0;  -- 延迟1秒
   end loop;

exception
   when others =>
      Put_Line ("Worker " & Integer'Image (Local_ID) & " error");
end Worker_Task;
```

### 2.2 任务创建与生命周期

```ada
procedure Task_Lifecycle_Demo is
   -- 方式1: 声明即创建
   task Background_Task;

   -- 方式2: 任务类型实例化
   task type Worker (ID : Integer);
   type Worker_Access is access Worker;

   W1 : Worker (1);           -- 栈分配
   W2 : Worker_Access;        -- 堆分配（稍后创建）

   task body Background_Task is
   begin
      Put_Line ("Background running");
   end Background_Task;

   task body Worker is
   begin
      Put_Line ("Worker " & Integer'Image (ID));
   end Worker;
begin
   -- W1自动运行
   W2 := new Worker (2);  -- 动态创建

   -- 等待任务完成（隐式等待）
   -- 作用域结束时，主程序等待所有任务完成
end Task_Lifecycle_Demo;
```

## 3. 会合（Rendezvous）通信

### 3.1 会合机制

Ada独特的同步通信方式：调用者等待被调用者准备好，两者"会合"交换数据。

```ada
-- 服务端任务
task Server_Task is
   entry Get_Data (Value : out Integer);  -- 入口声明
   entry Put_Data (Value : in Integer);
end Server_Task;

task body Server_Task is
   Internal_Data : Integer := 0;
begin
   loop
      select
         accept Get_Data (Value : out Integer) do
            Value := Internal_Data;
         end Get_Data;
      or
         accept Put_Data (Value : in Integer) do
            Internal_Data := Value;
         end Put_Data;
      or
         terminate;  -- 当主程序结束时终止
      end select;
   end loop;
end Server_Task;

-- 客户端调用
procedure Client is
   Data : Integer;
begin
   Server_Task.Put_Data (42);     -- 会合调用
   Server_Task.Get_Data (Data);   -- 接收数据
end Client;
```

### 3.2 选择性接受

```ada
task Selective_Server is
   entry Urgent_Request (Data : Integer);
   entry Normal_Request (Data : Integer);
   entry Shutdown;
end Selective_Server;

task body Selective_Server is
   Running : Boolean := True;
begin
   while Running loop
      select
         -- 优先处理紧急请求
         accept Urgent_Request (Data : Integer) do
            Process_Urgent (Data);
         end Urgent_Request;
      or
         -- 延迟接受普通请求
         delay until Clock + Seconds (5);
         accept Normal_Request (Data : Integer) do
            Process_Normal (Data);
         end Normal_Request;
      or
         accept Shutdown do
            Running := False;
         end Shutdown;
      else
         -- 如果没有等待的调用，执行其他工作
         Do_Background_Work;
      end select;
   end loop;
end Selective_Server;
```

## 4. 受保护对象（Protected Objects）

### 4.1 基本用法

```ada
-- 线程安全的数据封装
protected Counter is
   procedure Increment;
   procedure Decrement;
   function Get return Integer;

private
   Value : Integer := 0;
end Counter;

protected body Counter is
   procedure Increment is
   begin
      Value := Value + 1;  -- 互斥访问
   end Increment;

   procedure Decrement is
   begin
      Value := Value - 1;
   end Decrement;

   function Get return Integer is
   begin
      return Value;  -- 允许多个读者
   end Get;
end Counter;

-- 使用
procedure Use_Counter is
begin
   Counter.Increment;  -- 自动互斥
   Put_Line (Integer'Image (Counter.Get));
end Use_Counter;
```

### 4.2 条件变量模式

```ada
-- 生产者-消费者缓冲区
protected Bounded_Buffer is
   entry Put (Item : Integer);  -- 条件入口
   entry Get (Item : out Integer);

private
   Buffer : array (1 .. 10) of Integer;
   Count  : Natural := 0;
   Head, Tail : Positive := 1;
end Bounded_Buffer;

protected body Bounded_Buffer is
   entry Put (Item : Integer) when Count < Buffer'Length is
      -- 仅在缓冲区不满时接受
   begin
      Buffer (Tail) := Item;
      Tail := (Tail mod Buffer'Length) + 1;
      Count := Count + 1;
   end Put;

   entry Get (Item : out Integer) when Count > 0 is
      -- 仅在缓冲区非空时接受
   begin
      Item := Buffer (Head);
      Head := (Head mod Buffer'Length) + 1;
      Count := Count - 1;
   end Get;
end Bounded_Buffer;
```

## 5. 实时扩展（Real-Time Annex）

### 5.1 Ravenscar Profile

为安全关键系统设计的确定性子集：

```ada
-- Ravenscar限制（确保可分析性）
pragma Profile (Ravenscar);

-- 特性限制：
-- - 无动态任务创建
-- - 无选择性接受（except terminate）
-- - 无 entries上的delay
-- - 无 requeue
-- - 受保护对象入口最多1个

-- 适合DO-178C、ISO 26262认证
```

### 5.2 优先级与调度

```ada
-- 定义优先级类型
subtype Priority is System.Priority range 1 .. 10;

-- 周期性任务（实时系统常见模式）
task type Periodic_Task (Period_MS : Integer; Pri : Priority);

task body Periodic_Task is
   Next_Release : Time := Clock;
   Period_Duration : constant Time_Span := Milliseconds (Period_MS);
begin
   -- 设置优先级
   Set_Priority (Pri);

   loop
      -- 执行周期性工作
      Do_Periodic_Work;

      -- 等待下一个周期
      Next_Release := Next_Release + Period_Duration;
      delay until Next_Release;
   end loop;
end Periodic_Task;

-- 创建不同优先级的任务
Controller : Periodic_Task (10, 10);  -- 高优先级，10ms周期
Monitor    : Periodic_Task (100, 5);  -- 低优先级，100ms周期
```

## 6. 与C pthread对比示例

### 6.1 生产者-消费者

**C (pthreads)**:

```c
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t not_full = PTHREAD_COND_INITIALIZER;
pthread_cond_t not_empty = PTHREAD_COND_INITIALIZER;
int buffer[10], count = 0;

void* producer(void* arg) {
    pthread_mutex_lock(&mutex);
    while (count == 10)
        pthread_cond_wait(&not_full, &mutex);
    buffer[count++] = *(int*)arg;
    pthread_cond_signal(&not_empty);
    pthread_mutex_unlock(&mutex);
}
```

**Ada**:

```ada
protected Buffer is
   entry Put (Item : Integer);
   entry Get (Item : out Integer);
private
   Data : array (1..10) of Integer;
   Count : Natural := 0;
end Buffer;

protected body Buffer is
   entry Put (Item : Integer) when Count < 10 is
   begin
      Data (Count + 1) := Item;
      Count := Count + 1;
   end Put;
   -- Get类似...
end Buffer;
```

**Ada优势**:

- 无需显式锁/条件变量
- 条件在入口处声明，更清晰
- 编译器保证互斥

## 7. 常见模式

### 7.1 线程池

```ada
task type Worker (ID : Integer) is
   entry Assign_Job (Job : Work_Item);
end Worker;

task body Worker is
   Current_Job : Work_Item;
begin
   loop
      accept Assign_Job (Job : Work_Item) do
         Current_Job := Job;
      end Assign_Job;
      Execute (Current_Job);
   end loop;
end Worker;
```

```ada
select
```
