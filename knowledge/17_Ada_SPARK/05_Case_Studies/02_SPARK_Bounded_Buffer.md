﻿# SPARK案例研究：有界缓冲区（生产者-消费者）

## 1. 项目概述

### 1.1 目标

实现经典的生产者-消费者问题，使用SPARK证明：

- 无缓冲区溢出/下溢
- 线程安全
- 功能正确性

### 1.2 问题描述

```
生产者 ──► [有界缓冲区] ◄── 消费者

约束:
- 缓冲区大小有限
- 生产者不能向满缓冲区写入
- 消费者不能从空缓冲区读取
- 多线程安全
```

## 2. 规范设计

### 2.1 接口定义

```ada
-- bounded_buffer.ads
package Bounded_Buffer with SPARK_Mode is

   -- 缓冲区容量
   Capacity : constant := 10;

   -- 元素类型
   subtype Element_Type is Integer;

   -- 缓冲区类型
   type Buffer is private;

   -- 缓冲区状态查询
   function Is_Empty (B : Buffer) return Boolean;
   function Is_Full (B : Buffer) return Boolean;
   function Count (B : Buffer) return Natural;

   -- 初始化
   procedure Initialize (B : out Buffer)
      with Post => Is_Empty (B) and Count (B) = 0;

   -- 生产者：向缓冲区放入元素
   procedure Put (B : in out Buffer; Item : Element_Type)
      with
         Pre  => not Is_Full (B),
         Post => Count (B) = Count (B)'Old + 1 and
                 not Is_Empty (B) and
                 Get_Last (B) = Item;  -- 最后放入的是Item

   -- 消费者：从缓冲区取出元素
   procedure Get (B : in out Buffer; Item : out Element_Type)
      with
         Pre  => not Is_Empty (B),
         Post => Count (B) = Count (B)'Old - 1 and
                 not Is_Full (B) and
                 Item = Get_First (B)'Old;  -- 取出的是第一个

private

   subtype Index_Type is Natural range 0 .. Capacity - 1;
   type Array_Type is array (Index_Type) of Element_Type;

   type Buffer is record
      Data   : Array_Type;
      Head   : Index_Type;     -- 消费者位置
      Tail   : Index_Type;     -- 生产者位置
      Count  : Natural range 0 .. Capacity;
   end record
   with Type_Invariant => Count <= Capacity;

   -- 私有辅助函数
   function Get_First (B : Buffer) return Element_Type is
      (B.Data (B.Head));

   function Get_Last (B : Buffer) return Element_Type is
      (if B.Tail = 0 then B.Data (Capacity - 1)
       else B.Data (B.Tail - 1));

   function Is_Empty (B : Buffer) return Boolean is (B.Count = 0);
   function Is_Full (B : Buffer) return Boolean is (B.Count = Capacity);

end Bounded_Buffer;
```

### 2.2 实现

```ada
-- bounded_buffer.adb
package body Bounded_Buffer with SPARK_Mode is

   procedure Initialize (B : out Buffer) is
   begin
      B.Head := 0;
      B.Tail := 0;
      B.Count := 0;
   end Initialize;

   procedure Put (B : in out Buffer; Item : Element_Type) is
   begin
      B.Data (B.Tail) := Item;
      B.Tail := (B.Tail + 1) mod Capacity;
      B.Count := B.Count + 1;
   end Put;

   procedure Get (B : in out Buffer; Item : out Element_Type) is
   begin
      Item := B.Data (B.Head);
      B.Head := (B.Head + 1) mod Capacity;
      B.Count := B.Count - 1;
   end Get;

end Bounded_Buffer;
```

## 4. 并发保护版本

### 4.1 使用受保护对象

```ada
-- concurrent_buffer.ads
package Concurrent_Buffer is

   protected type Buffer is
      entry Put (Item : Integer);
      entry Get (Item : out Integer);
      function Is_Empty return Boolean;
      function Is_Full return Boolean;
   private
      -- 内部使用Bounded_Buffer
      Internal : Bounded_Buffer.Buffer;
      Initialized : Boolean := False;
   end Buffer;

end Concurrent_Buffer;

-- concurrent_buffer.adb
package body Concurrent_Buffer is

   protected body Buffer is

      entry Put (Item : Integer) when not Bounded_Buffer.Is_Full (Internal) is
      begin
         if not Initialized then
            Bounded_Buffer.Initialize (Internal);
            Initialized := True;
         end if;
         Bounded_Buffer.Put (Internal, Item);
      end Put;

      entry Get (Item : out Integer) when not Bounded_Buffer.Is_Empty (Internal) is
      begin
         Bounded_Buffer.Get (Internal, Item);
      end Get;

      function Is_Empty return Boolean is
      begin
         return Bounded_Buffer.Is_Empty (Internal);
      end Is_Empty;

      function Is_Full return Boolean is
      begin
         return Bounded_Buffer.Is_Full (Internal);
      end Is_Full;

   end Buffer;

end Concurrent_Buffer;
```

## 5. 测试程序

```ada
-- test_buffer.adb
with Bounded_Buffer;
with Ada.Text_IO;

procedure Test_Buffer with SPARK_Mode is
   use Bounded_Buffer;
   use Ada.Text_IO;

   B : Buffer;
   V : Element_Type;
begin
   -- 测试1: 初始化
   Initialize (B);
   pragma Assert (Is_Empty (B));
   pragma Assert (Count (B) = 0);

   -- 测试2: Put操作
   for I in 1 .. 5 loop
      Put (B, I * 10);
   end loop;

   pragma Assert (Count (B) = 5);
   pragma Assert (not Is_Empty (B));
   pragma Assert (not Is_Full (B));

   -- 测试3: Get操作
   Get (B, V);
   pragma Assert (V = 10);  -- 先进先出
   pragma Assert (Count (B) = 4);

   Get (B, V);
   pragma Assert (V = 20);

   -- 测试4: 满缓冲区
   for I in 6 .. 10 loop
      Put (B, I * 10);
   end loop;

   pragma Assert (Is_Full (B));
   pragma Assert (Count (B) = Capacity);

   -- 测试5: 空缓冲区
   for I in 1 .. 10 loop
      Get (B, V);
   end loop;

   pragma Assert (Is_Empty (B));

   Put_Line ("All tests passed!");
end Test_Buffer;
```

## 6. 关键证明点

### 6.1 循环缓冲区索引

```ada
-- 使用mod运算确保索引在范围内
B.Tail := (B.Tail + 1) mod Capacity;

-- 证明: B.Tail始终在0..Capacity-1范围内
-- mod运算保证不会超过Capacity-1
```

### 6.2 计数器正确性

```ada
-- Put: Count增加1
-- Pre => Count < Capacity 保证不会溢出
-- Post => Count = Count'Old + 1

-- Get: Count减少1
-- Pre => Count > 0 保证不会下溢
-- Post => Count = Count'Old - 1
```
