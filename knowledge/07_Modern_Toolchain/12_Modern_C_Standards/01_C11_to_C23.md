## C11新特性

```c
/* _Generic: 类型安全的泛型 */
#define max(a, b) _Generic((a) + (b), \
    int: max_int, \
    double: max_double, \
    default: max_generic \
)(a, b)

int max_int(int a, int b) { return a > b ? a : b; }
double max_double(double a, double b) { return a > b ? a : b; }

/* 使用 */
int mi = max(1, 2);           /* 调用max_int */
double md = max(1.0, 2.0);    /* 调用max_double */

/* _Static_assert: 编译时断言 */
_Static_assert(sizeof(int) == 4, "int must be 32-bit");

/* 匿名结构体和联合体 */
struct {
    union {
        struct { int x, y; };  /* 匿名 */
        int coords[2];
    };
} point;

point.x = 10;  /* 直接访问 */
point.coords[1] = 20;

/* 原子操作 */
#include <stdatomic.h>
_Atomic(int) counter = 0;
atomic_fetch_add(&counter, 1);

/* 线程支持 */
#include <threads.h>
thrd_t thread;
thrd_create(&thread, func, arg);

/* 复数支持 */
#include <complex.h>
double complex z = 1.0 + 2.0*I;

/* alignas/alignof */
alignas(64) char cache_line[64];
_Static_assert(alignof(max_align_t) >= 8, "");
```

### 1. C11特性详解与最佳实践

#### _Generic 泛型选择

```c
#include <stdio.h>
#include <math.h>

/* 基础用法 */
#define abs(x) _Generic((x), \
    int: abs_int, \
    long: abs_long, \
    float: fabsf, \
    double: fabs, \
    long double: fabsl, \
    default: abs_int \
)(x)

int abs_int(int x) { return x < 0 ? -x : x; }
long abs_long(long x) { return x < 0 ? -x : x; }

/* 高级用法 - 类型特征 */
#define type_name(x) _Generic((x), \
    char: "char", \
    short: "short", \
    int: "int", \
    long: "long", \
    float: "float", \
    double: "double", \
    char*: "string", \
    void*: "pointer", \
    default: "unknown" \
)

/* 容器泛型 */
typedef struct { int* data; size_t len; } int_vec;
typedef struct { double* data; size_t len; } double_vec;

#define vec_sum(v) _Generic((v), \
    int_vec: vec_sum_int, \
    double_vec: vec_sum_double \
)(v)

int vec_sum_int(int_vec v) {
    int sum = 0;
    for (size_t i = 0; i < v.len; i++) sum += v.data[i];
    return sum;
}

double vec_sum_double(double_vec v) {
    double sum = 0;
    for (size_t i = 0; i < v.len; i++) sum += v.data[i];
    return sum;
}

/* 使用示例 */
int main(void) {
    int i = -42;
    double d = -3.14;

    printf("abs(%d) = %d\n", i, abs(i));       /* 使用abs_int */
    printf("abs(%f) = %f\n", d, abs(d));       /* 使用fabs */

    printf("type of 42 is %s\n", type_name(42));
    printf("type of 3.14 is %s\n", type_name(3.14));

    return 0;
}
```

#### 原子操作与内存序

```c
#include <stdatomic.h>
#include <stdbool.h>
#include <threads.h>

/* 原子变量声明 */
_Atomic(int) counter = 0;
_Atomic(uint64_t) shared_timestamp = 0;

/* 无锁计数器 */
void increment_counter(void) {
    /* memory_order_relaxed: 最宽松，只保证原子性 */
    atomic_fetch_add_explicit(&counter, 1, memory_order_relaxed);
}

/* 发布-订阅模式 */
_Atomic(struct data*) shared_data = NULL;

void producer(void) {
    struct data* d = malloc(sizeof(*d));
    d->value = 42;
    d->ready = true;

    /* memory_order_release: 确保之前的写入对订阅者可见 */
    atomic_store_explicit(&shared_data, d, memory_order_release);
}

void consumer(void) {
    struct data* d;

    /* memory_order_acquire: 确保看到发布者的所有写入 */
    while (!(d = atomic_load_explicit(&shared_data, memory_order_acquire))) {
        thrd_yield();
    }

    /* 现在可以安全地访问d的内容 */
    printf("Got value: %d\n", d->value);
}

/* 无锁栈实现 */
typedef struct node {
    int value;
    struct node* next;
} node_t;

typedef struct {
    _Atomic(node_t*) head;
} lock_free_stack_t;

void stack_init(lock_free_stack_t* s) {
    atomic_init(&s->head, NULL);
}

void stack_push(lock_free_stack_t* s, int value) {
    node_t* new_node = malloc(sizeof(node_t));
    new_node->value = value;

    node_t* old_head;
    do {
        old_head = atomic_load(&s->head);
        new_node->next = old_head;
        /* memory_order_release确保new_node完全初始化 */
    } while (!atomic_compare_exchange_weak_explicit(
        &s->head, &old_head, new_node,
        memory_order_release, memory_order_relaxed));
}

bool stack_pop(lock_free_stack_t* s, int* value) {
    node_t* old_head;
    do {
        old_head = atomic_load_explicit(&s->head, memory_order_acquire);
        if (old_head == NULL) return false;
        /* memory_order_acquire确保看到old_head的内容 */
    } while (!atomic_compare_exchange_weak_explicit(
        &s->head, &old_head, old_head->next,
        memory_order_acquire, memory_order_relaxed));

    *value = old_head->value;
    free(old_head);
    return true;
}

/* 顺序一致性（最强约束） */
_Atomic(int) flag = 0;
_Atomic(int) data = 0;

void thread1(void) {
    atomic_store(&data, 100);       /* 顺序一致性 */
    atomic_store(&flag, 1);         /* 顺序一致性 */
}

void thread2(void) {
    while (atomic_load(&flag) != 1) {
        /* 等待 */
    }
    /* 保证看到data = 100 */
    printf("data = %d\n", atomic_load(&data));
}
```

#### 线程支持

```c
#include <threads.h>
#include <stdio.h>

/* 线程本地存储 */
_Thread_local int thread_id = 0;

/* 互斥锁和条件变量 */
mtx_t mutex;
cnd_t cond;
int shared_count = 0;

int worker_thread(void* arg) {
    int id = *(int*)arg;
    thread_id = id;

    printf("Thread %d starting\n", id);

    /* 锁定 */
    mtx_lock(&mutex);

    shared_count++;
    printf("Thread %d: count = %d\n", id, shared_count);

    /* 条件变量等待 */
    while (shared_count < 3) {
        cnd_wait(&cond, &mutex);
    }

    /* 广播 */
    cnd_broadcast(&cond);

    mtx_unlock(&mutex);

    return id * 10;
}

int main(void) {
    thrd_t threads[3];
    int ids[3] = {1, 2, 3};

    /* 初始化 */
    mtx_init(&mutex, mtx_plain);
    cnd_init(&cond);

    /* 创建线程 */
    for (int i = 0; i < 3; i++) {
        thrd_create(&threads[i], worker_thread, &ids[i]);
    }

    /* 等待完成 */
    int results[3];
    for (int i = 0; i < 3; i++) {
        thrd_join(threads[i], &results[i]);
        printf("Thread %d returned %d\n", i + 1, results[i]);
    }

    /* 清理 */
    mtx_destroy(&mutex);
    cnd_destroy(&cond);

    return 0;
}
```

### 2. C23新特性详解

#### typeof 运算符

```c
#include <stdio.h>
#include <stdlib.h>

/* 基本用法 */
typeof(int) x = 42;          /* x是int类型 */
typeof(x) y;                  /* y也是int类型 */

/* 泛型宏改进 */
#define max(a, b) ({ \
    typeof(a) _a = (a); \
    typeof(b) _b = (b); \
    _a > _b ? _a : _b; \
})

/* 安全的swap */
#define swap(a, b) do { \
    typeof(a) _tmp = (a); \
    (a) = (b); \
    (b) = _tmp; \
} while(0)

/* 泛型容器 */
#define declare_stack(type, name) \
    typedef struct { \
        type* data; \
        size_t size; \
        size_t capacity; \
    } name; \
    \
    static inline void name##_init(name* s) { \
        s->data = NULL; \
        s->size = 0; \
        s->capacity = 0; \
    } \
    \
    static inline void name##_push(name* s, type val) { \
        if (s->size >= s->capacity) { \
            s->capacity = s->capacity ? s->capacity * 2 : 8; \
            s->data = realloc(s->data, s->capacity * sizeof(type)); \
        } \
        s->data[s->size++] = val; \
    } \
    \
    static inline type name##_pop(name* s) { \
        return s->data[--s->size]; \
    }

/* 实例化 */
declare_stack(int, int_stack)
declare_stack(double, double_stack)

/* 使用 */
int main(void) {
    int_stack s;
    int_stack_init(&s);
    int_stack_push(&s, 42);
    int_stack_push(&s, 100);
    printf("%d\n", int_stack_pop(&s));

    return 0;
}
```

/* 安全字符串操作 */
void safe_string_ops(void) {
    char dest[100];
    const char* src = "Hello, World!";

    /* 安全拷贝 */
    errno_t err = strncpy_s(dest, sizeof(dest), src, sizeof(dest) - 1);
    if (err != 0) {
        /* 处理错误 */
    }

    /* 安全拼接 */
    strcat_s(dest, sizeof(dest), "!!!");

    /* 安全格式化 */
    sprintf_s(dest, sizeof(dest), "Value: %d", 42);
}
