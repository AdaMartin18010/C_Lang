## 概述

C11标准引入了标准化的内存模型和原子操作，为C语言提供了可移植的并发编程基础。

### 为什么需要内存模型

```
问题: 编译器和CPU可能重排指令
int x = 0, y = 0;

Thread 1:           Thread 2:
x = 1;              y = 1;
y = 2;              x = 2;

可能的结果: x=2, y=2 (符合直觉)
但也可能: x=1, y=1 (由于重排!)

内存模型: 定义什么重排是合法的，如何防止错误重排
```

### C11原子头文件

```c
#include <stdatomic.h>
#include <stdbool.h>
#include <stdint.h>
#include <stdio.h>
#include <threads.h>

/* 原子类型定义 */
_Atomic int counter;           /* 原子int */
atomic_int counter2;           /* 标准原子类型 */
_Atomic(uint64_t) big_counter; /* 原子64位 */
```

## 内存序 (Memory Order)

### 六种内存序

```c
/* C11标准定义的内存序 */
typedef enum {
    memory_order_relaxed = 0,  /* 无同步保证，仅原子性 */
    memory_order_consume = 1,  /* 消费序 (少用) */
    memory_order_acquire = 2,  /* 获取序 */
    memory_order_release = 3,  /* 释放序 */
    memory_order_acq_rel = 4,  /* 获取-释放 */
    memory_order_seq_cst = 5   /* 顺序一致性 (默认) */
} memory_order;
```

### 1. Relaxed (宽松序)

```c
/* 仅保证操作的原子性，不提供内存序保证 */
_Atomic(int) counter = 0;

void increment_relaxed(void) {
    /* 纯计数场景，不需要同步 */
    atomic_fetch_add_explicit(&counter, 1, memory_order_relaxed);
}

/* 使用场景:
 * - 纯计数器 (统计、ID生成)
 * - 不需要与其他变量同步的场景
 * - 追求极致性能
 */

/* 示例: 统计访问次数 */
_Atomic(uint64_t) page_hits = 0;

void on_page_visit(void) {
    atomic_fetch_add_explicit(&page_hits, 1, memory_order_relaxed);
}

uint64_t get_page_hits(void) {
    return atomic_load_explicit(&page_hits, memory_order_relaxed);
}
```

### 2. Acquire-Release (获取-释放)

```c
/* 最常用的内存序组合，用于线程间传递数据 */

_Atomic(int*) shared_ptr = NULL;
int data = 0;  /* 非原子数据 */

/* Thread 1: 生产者 */
void produce(void) {
    data = 42;  /* 准备数据 */

    /* Release: 保证之前的写入对获取方可见 */
    atomic_store_explicit(&shared_ptr, &data, memory_order_release);
}

/* Thread 2: 消费者 */
void consume(void) {
    int* ptr;

    /* Acquire: 保证看到发布方的所有写入 */
    while ((ptr = atomic_load_explicit(&shared_ptr,
                                        memory_order_acquire)) == NULL) {
        /* 自旋等待 */
    }

    /* 这里可以安全访问*ptr，保证看到data=42 */
    printf("Got: %d\n", *ptr);
}

/* Acquire-Release 语义:
 * Release: 之前的所有读写操作不会重排到后面
 * Acquire: 之后的所有读写操作不会重排到前面
 * 形成同步点
 */
```

### 3. Sequential Consistency (顺序一致性)

```c
/* 默认、最强的内存序，所有线程看到一致的操作顺序 */
_Atomic(int) x = 0;
_Atomic(int) y = 0;

void thread1(void) {
    atomic_store(&x, 1);  /* seq_cst */
    atomic_store(&y, 1);  /* seq_cst */
}

void thread2(void) {
    int a = atomic_load(&y);  /* seq_cst */
    int b = atomic_load(&x);  /* seq_cst */

    /* 不可能出现: a=1, b=0 的情况
     * 因为所有seq_cst操作有全局顺序
     */
}

/* 使用场景:
 * - 默认选择，最安全
 * - 锁的实现
 * - 需要全局顺序保证的场景
 *
 * 性能: 比acquire-release慢，因为需要额外的内存屏障
 */
```

### 4. Acquire-Release 双向同步

```c
/* 双向同步 - 用于读写锁、引用计数等 */
_Atomic(int) refcount = 1;

void retain(void) {
    /* 增加引用计数 */
    atomic_fetch_add_explicit(&refcount, 1, memory_order_relaxed);
}

bool release(void) {
    /* Acq-Rel: 获取之前的操作，释放后续操作 */
    int old = atomic_fetch_sub_explicit(&refcount, 1,
                                        memory_order_acq_rel);
    if (old == 1) {
        /* 最后一个引用，可以安全释放 */
        return true;
    }
    return false;
}

/* 使用场景:
 * - 引用计数
 * - 读写锁
 * - 需要读写双向同步的场景
 */
```

### 基本原子操作

```c
/* 初始化 */
_Atomic(int) counter = ATOMIC_VAR_INIT(0);
atomic_int counter2 = ATOMIC_VAR_INIT(0);

/* C23中更简单 */
_Atomic int counter3 = 0;  /* 直接初始化 */

/* 加载和存储 */
int value = atomic_load(&counter);
atomic_store(&counter, 42);

/* 显式内存序 */
int val = atomic_load_explicit(&counter, memory_order_acquire);
atomic_store_explicit(&counter, 42, memory_order_release);

/* 交换 */
int old = atomic_exchange(&counter, 100);

/* 比较交换 (CAS) - 无锁编程核心 */
int expected = 0;
bool success = atomic_compare_exchange_strong(
    &counter,           /* 原子变量 */
    &expected,          /* 预期值 (失败时更新为当前值) */
    1                   /* 新值 */
);
/* 返回true表示交换成功，false表示失败 */

/* 弱CAS (可能伪失败，但更高效) */
while (!atomic_compare_exchange_weak(&counter, &expected, 1)) {
    expected = 0;  /* 重新设置预期值 */
}
```

### 算术和位操作

```c
_Atomic(int) counter = 0;

/* 加减 */
int old = atomic_fetch_add(&counter, 5);   /* counter += 5, 返回旧值 */
int old2 = atomic_fetch_sub(&counter, 3);  /* counter -= 3 */

/* 位操作 */
atomic_fetch_or(&counter, 0x1);   /* 设置位 */
atomic_fetch_and(&counter, ~0x1); /* 清除位 */
atomic_fetch_xor(&counter, 0x1);  /* 翻转位 */

/* 自增自减 */
atomic_fetch_add(&counter, 1);  /* ++counter */
atomic_fetch_sub(&counter, 1);  /* --counter */

/* 显式内存序版本 */
atomic_fetch_add_explicit(&counter, 1, memory_order_relaxed);
```

## Happens-Before 关系

### 概念

```
Happens-Before 是内存模型的核心概念:
- 如果 A happens-before B，那么 A 的效果对 B 可见
- 用于判断多线程程序的行为是否正确

建立 Happens-Before 的方式:
1. 同一线程内的顺序
2. 线程创建/join
3. 原子操作的 release-acquire 对
4. 锁的解锁-加锁
```

### 线程同步

```c
#include <threads.h>

/* 线程创建 happens-before 线程入口 */
int worker(void* arg) {
    /* 这里看到创建线程之前的所有写入 */
    return 0;
}

void create_worker(void) {
    int data = 42;
    thrd_t t;
    thrd_create(&t, worker, &data);
    /* 注意: data的生命周期! */
}

/* 线程 join happens-before join之后的代码 */
void wait_worker(void) {
    thrd_t t;
    thrd_create(&t, worker, NULL);

    thrd_join(t, NULL);
    /* 这里看到线程的所有写入 */
}
```

### Release-Acquire 同步

```c
_Atomic(int) flag = 0;
int data = 0;

void thread_a(void) {
    data = 42;  /* A */
    atomic_store_explicit(&flag, 1, memory_order_release);  /* B */
}

void thread_b(void) {
    while (atomic_load_explicit(&flag, memory_order_acquire) != 1) {
        /* 自旋 */
    }  /* C */

    /* D: 这里看到 A 的写入，因为 B release -> C acquire */
    printf("%d\n", data);  /* 保证输出 42 */
}

/* Happens-Before 链:
 * A -> B (同一线程)
 * B -> C (release-acquire 同步)
 * C -> D (同一线程)
 * 因此: A -> D，data的写入对读取可见
 */
```

## 同步机制

### 互斥锁 (Mtx)

```c
#include <threads.h>

/* C11标准互斥锁 */
mtx_t mutex;
int shared_data = 0;

/* 初始化 */
void init(void) {
    mtx_init(&mutex, mtx_plain);        /* 普通锁 */
    mtx_init(&mutex, mtx_recursive);    /* 递归锁 */
    mtx_init(&mutex, mtx_timed);        /* 支持超时 */
}

/* 基本使用 */
void update_data(int value) {
    mtx_lock(&mutex);       /* 获取锁 */
    shared_data = value;    /* 临界区 */
    mtx_unlock(&mutex);     /* 释放锁 */
}

/* 非阻塞尝试 */
bool try_update(int value) {
    if (mtx_trylock(&mutex) == thrd_success) {
        shared_data = value;
        mtx_unlock(&mutex);
        return true;
    }
    return false;
}

/* 带超时 */
void timed_update(int value) {
    struct timespec ts;
    timespec_get(&ts, TIME_UTC);
    ts.tv_sec += 1;  /* 1秒超时 */

    if (mtx_timedlock(&mutex, &ts) == thrd_success) {
        shared_data = value;
        mtx_unlock(&mutex);
    }
}

/* 清理 */
void cleanup(void) {
    mtx_destroy(&mutex);
}
```

### 条件变量 (Cnd)

```c
#include <threads.h>

mtx_t mutex;
cnd_t cond;
int ready = 0;

/* 条件变量基本用法 */
void waiter(void) {
    mtx_lock(&mutex);

    while (!ready) {  /* 必须用循环检查，防止虚假唤醒 */
        cnd_wait(&cond, &mutex);  /* 原子: 解锁->等待->加锁 */
    }

    /* 条件满足，处理 */
    printf("Ready!\n");

    mtx_unlock(&mutex);
}

void signaler(void) {
    mtx_lock(&mutex);

    ready = 1;
    cnd_signal(&cond);  /* 唤醒一个等待者 */
    /* cnd_broadcast(&cond); */  /* 唤醒所有等待者 */

    mtx_unlock(&mutex);
}

/* 带超时的等待 */
bool timed_waiter(void) {
    mtx_lock(&mutex);

    struct timespec ts;
    timespec_get(&ts, TIME_UTC);
    ts.tv_sec += 5;

    while (!ready) {
        if (cnd_timedwait(&cond, &mutex, &ts) == thrd_timeout) {
            mtx_unlock(&mutex);
            return false;  /* 超时 */
        }
    }

    mtx_unlock(&mutex);
    return true;
}
```

## 无锁数据结构基础

### 无锁栈 (Treiber Stack)

```c
#include <stdatomic.h>
#include <stdlib.h>

typedef struct node {
    int data;
    struct node* next;
} node_t;

typedef struct {
    _Atomic(node_t*) top;
} lock_free_stack_t;

void stack_init(lock_free_stack_t* stack) {
    atomic_init(&stack->top, NULL);
}

void stack_push(lock_free_stack_t* stack, int value) {
    node_t* new_node = malloc(sizeof(node_t));
    new_node->data = value;

    node_t* old_top;
    do {
        old_top = atomic_load_explicit(&stack->top, memory_order_relaxed);
        new_node->next = old_top;
        /* release保证node初始化对其他线程可见 */
    } while (!atomic_compare_exchange_weak_explicit(
        &stack->top, &old_top, new_node,
        memory_order_release, memory_order_relaxed));
}

bool stack_pop(lock_free_stack_t* stack, int* value) {
    node_t* old_top;
    do {
        old_top = atomic_load_explicit(&stack->top, memory_order_acquire);
        if (old_top == NULL) {
            return false;  /* 空栈 */
        }
        /* acquire保证看到完整的node数据 */
    } while (!atomic_compare_exchange_weak_explicit(
        &stack->top, &old_top, old_top->next,
        memory_order_release, memory_order_relaxed));

    *value = old_top->data;
    free(old_top);  /* 注意: ABA问题! */
    return true;
}
```

### 单生产者单消费者队列 (SPSC)

```c
#include <stdatomic.h>
#include <stdbool.h>

#define QUEUE_SIZE 1024

typedef struct {
    int buffer[QUEUE_SIZE];
    _Atomic(size_t) head;  /* 只有生产者写入 */
    _Atomic(size_t) tail;  /* 只有消费者写入 */
} spsc_queue_t;

void spsc_init(spsc_queue_t* q) {
    atomic_init(&q->head, 0);
    atomic_init(&q->tail, 0);
}

bool spsc_enqueue(spsc_queue_t* q, int value) {
    const size_t head = atomic_load_explicit(&q->head, memory_order_relaxed);
    const size_t next = (head + 1) % QUEUE_SIZE;

    /* 检查满 */
    if (next == atomic_load_explicit(&q->tail, memory_order_acquire)) {
        return false;  /* 队列满 */
    }

    q->buffer[head] = value;
    atomic_store_explicit(&q->head, next, memory_order_release);
    return true;
}

bool spsc_dequeue(spsc_queue_t* q, int* value) {
    const size_t tail = atomic_load_explicit(&q->tail, memory_order_relaxed);

    /* 检查空 */
    if (tail == atomic_load_explicit(&q->head, memory_order_acquire)) {
        return false;  /* 队列空 */
    }

    *value = q->buffer[tail];
    atomic_store_explicit(&q->tail, (tail + 1) % QUEUE_SIZE, memory_order_release);
    return true;
}

/*
 * 注意: SPSC队列中，head和tail由不同线程独占写入
 * 只需要acquire-release保证数据的可见性
 * 不需要昂贵的CAS操作
 */
```

## 常见陷阱与最佳实践

### 1. 数据竞争 (Data Race)

```c
/* 错误: 非原子变量多线程访问 */
int counter = 0;

void thread_func(void) {
    for (int i = 0; i < 1000000; i++) {
        counter++;  /* 数据竞争! 未定义行为 */
    }
}

/* 正确: 使用原子变量 */
_Atomic(int) counter = 0;

void thread_func_correct(void) {
    for (int i = 0; i < 1000000; i++) {
        atomic_fetch_add(&counter, 1);
    }
}
```

### 2. ABA问题

```c
/* ABA问题示例: 栈顶从A->B->A，CAS检测不到 */
/* 解决方案: 使用带版本的指针 (Tagged Pointer) */

typedef struct {
    node_t* ptr;
    uint64_t version;
} tagged_ptr_t;

typedef struct {
    _Atomic(tagged_ptr_t) top;
} safe_stack_t;

bool stack_pop_safe(safe_stack_t* stack, int* value) {
    tagged_ptr_t old, new;
    do {
        old = atomic_load_explicit(&stack->top, memory_order_acquire);
        if (old.ptr == NULL) return false;

        new.ptr = old.ptr->next;
        new.version = old.version + 1;  /* 版本递增 */
    } while (!atomic_compare_exchange_weak_explicit(
        &stack->top, &old, new,
        memory_order_release, memory_order_relaxed));

    *value = old.ptr->data;
    /* 延迟释放或使用Hazard Pointer */
    return true;
}
```

### 3. 内存序选择

```c
/* 决策树 */
void memory_order_guide(void) {
    /*
     * 问题1: 是否只需要原子性，不需要同步?
     *   是 -> memory_order_relaxed
     *
     * 问题2: 是否发布数据给另一个线程?
     *   是 -> memory_order_release (写) + memory_order_acquire (读)
     *
     * 问题3: 是否需要全局顺序?
     *   是 -> memory_order_seq_cst
     *
     * 默认选择: seq_cst (正确性优先)
     * 性能优化: 根据场景降级到 acquire/release/relaxed
     */
}

/* 最佳实践 */
/* 1. 默认使用 seq_cst，性能瓶颈时再优化 */
/* 2. 明确文档注释内存序选择的原因 */
/* 3. 配对使用 release-acquire */
/* 4. 谨慎使用 relaxed */
```

### 4. 伪共享 (False Sharing)

```c
/* 错误: 原子变量紧邻，导致缓存行竞争 */
typedef struct {
    _Atomic(int) counter1;  /* 同一线程写入 */
    _Atomic(int) counter2;  /* 另一线程写入 */

/* 正确: 缓存行填充 */
#define CACHE_LINE_SIZE 64

typedef struct {
    _Atomic(int) counter;

    padded_counter_t counter1;

```

### 5. 无锁 vs 有锁选择

```c
/* 选择指南 */
void lock_vs_lockfree(void) {
    /*
     * 使用锁当:
     * - 竞争不激烈
     * - 需要复杂的数据结构不变式
     * - 代码可读性优先
     * - 需要条件变量等阻塞语义
     *
     * 使用无锁当:
     * - 竞争激烈（高并发）
     * - 需要确定性的延迟（无阻塞）
     * - 实现简单的数据结构（栈、队列、计数器）
     * - 专家级性能优化
     *
     * 注意: 无锁代码更难编写和验证，谨慎使用!
     */
}
```

## 总结

C11内存模型提供了：

1. **标准化的原子操作** - 跨平台可移植
2. **灵活的内存序** - 在性能和正确性间权衡
3. **清晰的同步语义** - happens-before关系
4. **现代化的线程API** - threads.h、互斥锁、条件变量

**关键要点**:

- 默认使用 `memory_order_seq_cst`
- 需要性能时，使用 `release-acquire` 对
- 纯计数器使用 `relaxed`
- 无锁编程先考虑是否真的需要
