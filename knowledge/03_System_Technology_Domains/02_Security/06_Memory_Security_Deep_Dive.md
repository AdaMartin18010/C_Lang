### 1.1 内存安全的形式化定义


### 2.2 堆溢出与元数据破坏

```c
/* ============================================================
 * 堆溢出与fastbin攻击原理
 * ============================================================ */

// ptmalloc2堆块结构 (glibc)
struct malloc_chunk {
    size_t      mchunk_prev_size;  // 物理相邻前一个块大小 (如果前一个块空闲)
    size_t      mchunk_size;       // 当前块大小 + 标志位
    
    struct malloc_chunk* fd;       // 下一个空闲块 (仅当空闲时)
    struct malloc_chunk* bk;       // 上一个空闲块 (仅当空闲时)
    
    // 仅large bins使用:
    struct malloc_chunk* fd_nextsize;
    struct malloc_chunk* bk_nextsize;
};

// size字段标志位:
// - bit 0: PREV_INUSE (前一个块是否在使用)
// - bit 1: IS_MMAPPED (是否通过mmap分配)
// - bit 2: NON_MAIN_ARENA (是否非主arena)

/* 堆溢出攻击向量：
 * 
 * 1. 元数据覆盖:
 *    溢出覆盖下一个块的size字段，可能导致:
 *    - 向后合并时的非法unlink操作
 *    - 大小分类错误导致的错误chunk重用
 * 
 * 2. Fastbin Dup:
 *    利用double-free在fastbin链表中创建环
 *    
 * 3. House of Spirit:
 *    伪造chunk头，释放栈上的内存
 *    
 * 4. Unsorted Bin Attack:
 *    修改unsorted bin的bk指针，实现任意地址写入
 */
```

### 2.3 格式化字符串漏洞

```c
/* ============================================================
 * 格式化字符串漏洞原理与利用
 * ============================================================ */

// 漏洞代码示例:
void vulnerable(char* user_input) {
    printf(user_input);  // ❌ 危险: 用户控制格式字符串
}

// 安全代码:
void safe(char* user_input) {
    printf("%s", user_input);  // ✅ 正确: 格式字符串固定
}

/* 攻击原理:
 * 
 * printf函数通过解析格式字符串，从栈上读取参数。
 * 如果攻击者控制格式字符串，可以:
 * 
 * 1. 内存读取: %x %p %s 读取栈/任意地址内容
 *    printf("AAAA%p%p%p%p");  // 泄露栈内容
 *    printf("\x10\x01\x48\x00%p%p%s");  // 读取指定地址
 * 
 * 2. 内存写入: %n 系列向地址写入值
 *    %n:  写入已输出字符数到int*
 *    %hn: 写入short
 *    %hhn: 写入char
 *    
 *    利用:
 *    printf("AAAA%10$n\x10\x01\x48\x00");  // 向0x4801110写入4
 *    
 *    任意地址写入需要构造:
 *    - 地址放置在栈上 (通过之前调用残留或可控数据)
 *    - 使用%<offset>$n直接访问第offset个参数
 */

// 现代缓解:
// - FORTIFY_SOURCE: 编译时检测简单格式字符串
// - _FORTIFY_SOURCE=2: 更严格的检查
// - 静态分析器警告
```

```
│  ┌─────────┐    ┌─────────┐    ┌─────────┐                 │
```

### 3.2 堆风水 (Heap Feng Shui)

```c
/* ============================================================
 * 堆风水技术：精确控制堆布局
 * ============================================================ */

/* 原理：
 * 通过精心设计的分配/释放序列，控制堆上对象的相对位置，
 * 使得UAF后可预测地分配到攻击者控制的对象。
 */

// 堆风水步骤:
void heap_feng_shui() {
    // 步骤1: 堆排整 (Heap Grooming)
    // 分配填充chunk，将堆整理到已知状态
    void* groom[10];
    for (int i = 0; i < 10; i++) {
        groom[i] = malloc(0x100);
    }
    for (int i = 0; i < 10; i += 2) {
        free(groom[i]);  // 创建洞
    }
    
    // 步骤2: 放置目标对象
    // 目标对象会落入之前创建的洞中
    void* target = malloc(0x100);  // 使用groom[0]的位置
    
    // 步骤3: 释放目标，创建UAF
    free(target);  // target现在是悬垂指针
    
    // 步骤4: 喷射控制对象
    // 控制对象会重用target的内存
    void* spray[100];
    for (int i = 0; i < 100; i++) {
        spray[i] = malloc(0x100);
        // 填充控制数据，模拟victim对象的结构
        memset(spray[i], 'A', 0x100);
    }
    
    // 步骤5: 触发UAF
    // 通过悬垂指针访问，实际访问的是控制对象
    // *(target->function_ptr)() 将执行攻击者控制的地址
}

/* 现代防护：
 * - ASLR: 随机化基址，但堆相对位置仍可控
 * - Safe-Unlink: 双向链表完整性检查
 * - Fastbin Consolidation: 合并相邻空闲块
 * - TCache: 线程缓存，影响堆风水可预测性
 * - Memory Tagging (MTE): ARM硬件特性，标记内存
 */
```

## 4. 内存安全检测技术

### 4.1 影子内存 (Shadow Memory)

```
┌─────────────────────────────────────────────────────────────┐
│            AddressSanitizer 影子内存机制                     │
├─────────────────────────────────────────────────────────────┤
│                                                              │
│  影子内存映射：                                               │
│                                                              │
│  应用程序内存 (App Mem)        影子内存 (Shadow Mem)        │
│  ┌─────────────────────┐      ┌─────────────────────┐       │
│  │ 0x0000 - 0x7FFF     │      │ 0x2000 - 0x3FFF     │       │
│  │ (用户空间 128TB)    │─────►│ (影子映射 32TB)     │       │
│  └─────────────────────┘      └─────────────────────┘       │
│                                                              │
│  映射公式 (64-bit)：                                          │
│  shadow_addr = (app_addr >> 3) + 0x0000100000000000         │
│  (每8字节应用内存对应1字节影子内存)                           │
│                                                              │
│  影子内存编码：                                               │
│  ┌────────┬─────────────────────────────────────────┐       │
│  │ 编码   │ 含义                                    │       │
│  ├────────┼─────────────────────────────────────────┤       │
│  │ 0x00   │ 8字节全部可访问                         │       │
│  │ 0x01   │ 只有第1字节可访问                       │       │
│  │ ...    │ ...                                     │       │
│  │ 0x07   │ 只有第7字节可访问                       │       │
│  │ 0xFA   │ 左边红区 (heap-left-redzone)            │       │
│  │ 0xFB   │ 右边红区 (heap-right-redzone)           │       │
│  │ 0xFC   │ 栈红区 (stack-redzone)                  │       │
│  │ 0xFD   │ 全局红区 (global-redzone)               │       │
│  │ 0xFE   │ 已释放内存 (freed)                      │       │
│  └────────┴─────────────────────────────────────────┘       │
│                                                              │
│  运行时检查：                                                 │
│  每次内存访问前：                                             │
│  1. 计算影子地址: shadow = (addr >> 3) + offset             │
│  2. 读取影子字节: shadow_value = *shadow                    │
│  3. 检查合法性:  if (shadow_value != 0) error()             │
│                                                              │
│  开销: 约2-3x CPU, 2-4x 内存                                 │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

### 4.2 标记内存 (Memory Tagging)

```
┌─────────────────────────────────────────────────────────────┐
│          ARM Memory Tagging Extension (MTE)                  │
├─────────────────────────────────────────────────────────────┤
│                                                              │
│  内存标记原理：                                               │
│                                                              │
│  物理地址布局 (56-bit)：                                      │
│  ┌───────────────────────────────────────────────────────┐   │
│  │  物理地址 (52-bit)  │  标签 (4-bit)  │                │   │
│  └───────────────────────────────────────────────────────┘   │
│                                                              │
│  分配时：                                                     │
│  1. 随机生成4-bit标签 (0-15)                                 │
│  2. 标签存入专用内存 (与数据分离)                            │
│  3. 将标签嵌入指针高4位 (TBI: Top Byte Ignore)               │
│                                                              │
│  访问时：                                                     │
│  硬件自动比较指针标签与内存标签                               │
│  不匹配 → 同步异常 (SIGSEGV)                                 │
│                                                              │
│  优势：                                                       │
│  • 低开销 (~5% 性能影响)                                     │
│  • 无需插桩，硬件实现                                        │
│  • 可检测UAF、缓冲区溢出、无效释放                           │
│                                                              │
│  局限：                                                       │
│  • 仅16种标签，存在碰撞概率 (1/16)                           │
│  • 需要硬件支持 (ARMv8.5+)                                   │
│  • 不检测读取未初始化内存                                    │
│                                                              │
│  软件替代方案：                                               │
│  • SPARC ADI (Application Data Integrity)                    │
│  • LLVM HWASAN (软件模拟MTE)                                 │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

### 5.2 现代C安全编程

```c
/* ============================================================
 * 现代C安全编程最佳实践
 * ============================================================ */

#include <stddef.h>
#include <stdint.h>
#include <stdbool.h>

// 1. 使用显式大小的整数类型
void safe_types() {
    uint32_t count;      // 明确32位无符号
    int64_t offset;      // 明确64位有符号
    size_t size;         // 与平台相关的尺寸
}

// 2. 安全的字符串处理
#include <string.h>

// ❌ 危险
void unsafe_string(char* src) {
    char buf[64];
    strcpy(buf, src);  // 无边界检查
    strcat(buf, ".txt");  // 可能溢出
}

// ✅ 安全
#include <bsd/string.h>  // strlcpy, strlcat

void safe_string(const char* src) {
    char buf[64];
    strlcpy(buf, src, sizeof(buf));  // 总是null终止
    strlcat(buf, ".txt", sizeof(buf));
}

// 3. 安全的内存分配
#include <stdlib.h>

// ✅ 检查溢出，清零内存
void* safe_malloc(size_t nmemb, size_t size) {
    if (nmemb == 0 || size == 0) return NULL;
    
    // 检查乘法溢出
    if (nmemb > SIZE_MAX / size) {
        return NULL;  // 溢出
    }
    
    void* p = calloc(nmemb, size);  // calloc检查溢出并清零
    return p;
}

// 4. 边界检查接口 (C11 Annex K, 可选支持)
#define __STDC_WANT_LIB_EXT1__ 1
#include <string.h>

    char dest[64];

// 5. 智能指针模式 (C语言模拟)
typedef struct {
    void* ptr;
    void (*destructor)(void*);
} unique_ptr;

unique_ptr make_unique(size_t size, void (*dtor)(void*)) {
    unique_ptr p = {calloc(1, size), dtor};
    return p;
}

void unique_ptr_free(unique_ptr* p) {
    if (p && p->ptr) {
        if (p->destructor) p->destructor(p->ptr);
        free(p->ptr);
        p->ptr = NULL;
    }
}

#define UNIQUE_PTR(name, type, dtor) \
    unique_ptr name __attribute__((cleanup(unique_ptr_free))) = \
        make_unique(sizeof(type), dtor)

// 使用示例
void example() {
    UNIQUE_PTR(buf, char[1024], NULL);
    // buf.ptr 自动释放
}

// 6. 编译时安全属性 (GCC/Clang)
struct buffer {
    size_t len;
    char data[] __attribute__((counted_by(len)));
};

void process(const struct buffer* buf 
             __attribute__((pass_dynamic_object_size(0))))
{
    // 编译器知道buf->data的大小是buf->len
    memcpy(buf->data, "test", buf->len);  // 自动边界检查
```
