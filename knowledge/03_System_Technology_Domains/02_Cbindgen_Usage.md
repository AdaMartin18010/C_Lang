## 一、Cbindgen工具介绍

### 1.1 什么是Cbindgen

Cbindgen是一个从Rust代码生成C/C++头文件的工具，由Mozilla开发，主要用于将Rust库暴露给C/C++调用。
它是实现Rust FFI（Foreign Function Interface）的重要工具。

```
┌─────────────────────────────────────────────────────────────┐
│                    Cbindgen工作流程                          │
│                                                             │
│   ┌──────────┐     ┌──────────┐     ┌──────────┐          │
│   │ Rust源码  │ ──▶ │ Cbindgen │ ──▶ │ C头文件  │          │
│   │  .rs     │     │  解析    │     │  .h      │          │
│   └──────────┘     └──────────┘     └──────────┘          │
│         │                                    │              │
│         │                                    ▼              │
│         │                           ┌──────────┐           │
│         │                           │ C/C++    │           │
│         └──────────────────────────▶│ 代码调用 │           │
│                                     └──────────┘           │
└─────────────────────────────────────────────────────────────┘
```

## 二、配置.toml

### 2.1 基础配置文件

创建 `cbindgen.toml` 配置文件：

```toml
# cbindgen.toml - 基础配置示例

# 生成的头文件名
header = "/* Auto-generated by cbindgen */"

# 包含守卫（防止重复包含）
include_guard = "MY_LIBRARY_H"

# 语言模式: c 或 cxx
language = "c"

# 命名空间（仅C++有效）
namespace = "my_lib"

# 前缀（添加到所有函数和类型）
prefix = "my_"

# 括号风格
braces = "SameLine"  # 或 "NextLine"

# 缩进风格
line_length = 100

tab_width = 4

# 文档注释风格
documentation = true
documentation_style = "c"

# 是否包含未使用的类型
export.include = []
export.exclude = []

# 预处理指令
[defines]
"feature = serde" = "DEFINE_SERDE"
"cfg(target_os = \"windows\")" = "_WIN32"

# 枚举配置
[enum]
add_sentinel = false
prefix_with_name = false
derive_helper_methods = true
derive_const_casts = true

# 函数配置
[fn]
rename_args = "CamelCase"  # 参数命名风格
no_return = "void"

# 结构体配置
[struct]
rename_fields = "SnakeCase"

# 解析配置
[parse]
parse_deps = false
include = []
exclude = []

# 宏展开
[macro_expansion]
bitflags = false
```

## 三、生成C头文件

### 3.1 命令行方式

```bash
# 基础生成
cbindgen --crate my_crate --output my_crate.h

# 指定配置文件
cbindgen --crate my_crate --config cbindgen.toml --output my_crate.h

# 从stdin读取Rust代码
echo "pub fn add(a: i32, b: i32) -> i32 { a + b }" | cbindgen --lang c -

# 验证配置
cbindgen --validate cbindgen.toml

# 生成C++头文件
cbindgen --crate my_crate --lang c++ --output my_crate.hpp

# 导出所有类型（包括未使用的）
cbindgen --crate my_crate --export all --output my_crate.h
```

### 3.2 构建脚本方式（build.rs）

```rust
// build.rs - 自动在编译时生成头文件
use std::env;
use std::path::PathBuf;

fn main() {
    let crate_dir = env::var("CARGO_MANIFEST_DIR").unwrap();
    let out_dir = env::var("OUT_DIR").unwrap();

    // 配置cbindgen
    let config = cbindgen::Config {
        language: cbindgen::Language::C,
        header: Some(String::from("/* Auto-generated header */")),
        ..Default::default()
    };

    // 生成头文件
    cbindgen::Builder::new()
        .with_crate(&crate_dir)
        .with_config(config)
        .generate()
        .expect("Unable to generate bindings")
        .write_to_file(PathBuf::from(&out_dir).join("my_lib.h"));

    // 或从配置文件生成
    cbindgen::Builder::new()
        .with_crate(&crate_dir)
        .with_config(cbindgen::Config::from_file("cbindgen.toml")
            .expect("Unable to read cbindgen.toml"))
        .generate()
        .expect("Unable to generate bindings")
        .write_to_file("include/my_lib.h");
}
```

### 3.3 完整Rust库示例

```rust
// src/lib.rs

//! My Rust Library - C Bindings
//!
//! This library provides safe Rust implementations
//! with C FFI bindings.

use std::ffi::{CStr, CString};
use std::os::raw::{c_char, c_int, c_void};
use std::ptr;

// ============================================
// 基本类型示例
// ============================================

/// Adds two integers
///
/// # Safety
/// This function is safe and doesn't require unsafe
#[no_mangle]
pub extern "C" fn rl_add(a: c_int, b: c_int) -> c_int {
    a + b
}

/// Divides two floats
///
/// Returns 0.0 if b is 0
#[no_mangle]
pub extern "C" fn rl_divide(a: f64, b: f64) -> f64 {
    if b == 0.0 {
        0.0
    } else {
        a / b
    }
}

// ============================================
// 结构体示例
// ============================================

/// A 2D point
#[repr(C)]
#[derive(Debug, Clone, Copy)]
pub struct RlPoint {
    pub x: f64,
    pub y: f64,
}

/// Creates a new point
#[no_mangle]
pub extern "C" fn rl_point_new(x: f64, y: f64) -> RlPoint {
    RlPoint { x, y }
}

/// Calculates distance between two points
#[no_mangle]
pub extern "C" fn rl_point_distance(a: &RlPoint, b: &RlPoint) -> f64 {
    ((b.x - a.x).powi(2) + (b.y - a.y).powi(2)).sqrt()
}

// ============================================
// 枚举示例
// ============================================

/// Status codes for operations
#[repr(C)]
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum RlStatus {
    Ok = 0,
    Error = 1,
    InvalidInput = 2,
    OutOfMemory = 3,
    NotFound = 4,
}

/// Returns status as string
#[no_mangle]
pub extern "C" fn rl_status_string(status: RlStatus) -> *const c_char {
    match status {
        RlStatus::Ok => "ok\0".as_ptr() as *const c_char,
        RlStatus::Error => "error\0".as_ptr() as *const c_char,
        RlStatus::InvalidInput => "invalid_input\0".as_ptr() as *const c_char,
        RlStatus::OutOfMemory => "out_of_memory\0".as_ptr() as *const c_char,
        RlStatus::NotFound => "not_found\0".as_ptr() as *const c_char,
    }
}

// ============================================
// 不透明类型（Opaque Types）
// ============================================

/// Opaque handle to a database connection
pub struct RlDatabase {
    connection_string: String,
    is_connected: bool,
}

/// Creates a new database handle
///
/// # Safety
/// conn_str must be a valid null-terminated UTF-8 string
#[no_mangle]
pub unsafe extern "C" fn rl_db_create(conn_str: *const c_char) -> *mut RlDatabase {
    if conn_str.is_null() {
        return ptr::null_mut();
    }

    let c_str = CStr::from_ptr(conn_str);
    match c_str.to_str() {
        Ok(s) => {
            let db = Box::new(RlDatabase {
                connection_string: s.to_string(),
                is_connected: false,
            });
            Box::into_raw(db)
        }
        Err(_) => ptr::null_mut(),
    }
}

/// Destroys a database handle
///
/// # Safety
/// db must be a valid pointer returned by rl_db_create
/// After calling, db becomes invalid and must not be used
#[no_mangle]
pub unsafe extern "C" fn rl_db_destroy(db: *mut RlDatabase) {
    if !db.is_null() {
        drop(Box::from_raw(db));
    }
}

/// Connects to the database
///
/// # Safety
/// db must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn rl_db_connect(db: *mut RlDatabase) -> RlStatus {
    if db.is_null() {
        return RlStatus::InvalidInput;
    }

    let db = &mut *db;
    // 模拟连接逻辑
    db.is_connected = true;
    RlStatus::Ok
}

// ============================================
// 字符串处理
// ============================================

/// Returns a greeting message
///
/// # Safety
/// Caller must free the returned string using rl_string_free
#[no_mangle]
pub unsafe extern "C" fn rl_greet(name: *const c_char) -> *mut c_char {
    if name.is_null() {
        return ptr::null_mut();
    }

    let c_str = CStr::from_ptr(name);
    match c_str.to_str() {
        Ok(s) => {
            let greeting = format!("Hello, {}!", s);
            match CString::new(greeting) {
                Ok(c_greeting) => c_greeting.into_raw(),
                Err(_) => ptr::null_mut(),
            }
        }
        Err(_) => ptr::null_mut(),
    }
}

/// Frees a string allocated by the library
///
/// # Safety
/// s must be a valid pointer returned by library functions
#[no_mangle]
pub unsafe extern "C" fn rl_string_free(s: *mut c_char) {
    if !s.is_null() {
        drop(CString::from_raw(s));
    }
}

// ============================================
// 回调函数示例
// ============================================

/// Callback type for iteration
pub type RlIterCallback = extern "C" fn(item: *const c_char, user_data: *mut c_void);

/// Iterates over items and calls callback for each
///
/// # Safety
/// callback must be a valid function pointer
#[no_mangle]
pub unsafe extern "C" fn rl_iterate(callback: RlIterCallback, user_data: *mut c_void) {
    let items = vec!["item1", "item2", "item3"];

    for item in items {
        let c_item = CString::new(item).unwrap();
        callback(c_item.as_ptr(), user_data);
    }
}

// ============================================
// 数组和切片
// ============================================

/// Sum of integer array
///
/// # Safety
/// data must point to at least len valid integers
#[no_mangle]
pub unsafe extern "C" fn rl_array_sum(data: *const c_int, len: usize) -> c_int {
    if data.is_null() || len == 0 {
        return 0;
    }

    let slice = std::slice::from_raw_parts(data, len);
    slice.iter().sum()
}

/// Sorts an array in place
///
/// # Safety
/// data must point to at least len valid integers
#[no_mangle]
pub unsafe extern "C" fn rl_array_sort(data: *mut c_int, len: usize) {
    if data.is_null() || len == 0 {
        return;
    }

    let slice = std::slice::from_raw_parts_mut(data, len);
    slice.sort_unstable();
}

// ============================================
// 线程安全类型
// ============================================

use std::sync::Arc;
use std::sync::atomic::{AtomicI64, Ordering};

/// Thread-safe counter
pub struct RlCounter {
    value: AtomicI64,
}

/// Creates a new counter
#[no_mangle]
pub extern "C" fn rl_counter_new() -> *mut RlCounter {
    let counter = Box::new(RlCounter {
        value: AtomicI64::new(0),
    });
    Box::into_raw(counter)
}

/// Destroys a counter
///
/// # Safety
/// counter must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn rl_counter_destroy(counter: *mut RlCounter) {
    if !counter.is_null() {
        drop(Box::from_raw(counter));
    }
}

/// Increments the counter
///
/// # Safety
/// counter must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn rl_counter_increment(counter: *mut RlCounter) -> i64 {
    if counter.is_null() {
        return 0;
    }
    (*counter).value.fetch_add(1, Ordering::SeqCst) + 1
}

/// Gets the current value
///
/// # Safety
/// counter must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn rl_counter_get(counter: *mut RlCounter) -> i64 {
    if counter.is_null() {
        return 0;
    }
    (*counter).value.load(Ordering::SeqCst)
}
```

```rust
// Cbindgen对泛型的支持有限，通常需要具体化

        pub struct $name {
            data: Vec<$type>,
        }

        impl $name {
            #[no_mangle]
            pub extern "C" fn new() -> *mut $name {
                Box::into_raw(Box::new($name { data: Vec::new() }))
            }

            #[no_mangle]
            pub extern "C" fn push(vec: *mut $name, value: $type) {
                unsafe { (*vec).data.push(value); }
            }

            #[no_mangle]
            pub extern "C" fn len(vec: *const $name) -> usize {
                unsafe { (*vec).data.len() }
            }
        }
    };
}
```

### 5.1 内存管理规范

```rust
use std::ffi::{CStr, CString};
use std::os::raw::c_char;
use std::ptr;

/// 内存管理黄金法则：
/// 1. 谁分配，谁释放
/// 2. 跨边界传递的数据必须明确定义所有权
/// 3. 使用Box::into_raw和Box::from_raw配对

/// 创建字符串（Rust分配，C释放）
///
/// # Safety
/// Caller must call mylib_string_free when done
#[no_mangle]
pub unsafe extern "C" fn mylib_string_create(s: *const c_char) -> *mut c_char {
    if s.is_null() {
        return ptr::null_mut();
    }

    // 复制输入字符串
    let input = CStr::from_ptr(s);
    match CString::new(input.to_bytes()) {
        Ok(copied) => copied.into_raw(),
        Err(_) => ptr::null_mut(),
    }
}

/// 释放字符串（对应mylib_string_create）
///
/// # Safety
/// s must be a valid pointer returned by mylib_string_create or mylib functions
#[no_mangle]
pub unsafe extern "C" fn mylib_string_free(s: *mut c_char) {
    if !s.is_null() {
        drop(CString::from_raw(s));
    }
}

/// 不透明类型的生命周期管理
pub struct Context {
    data: Vec<u8>,
    config: Config,
}

/// 创建上下文
#[no_mangle]
pub extern "C" fn mylib_context_create() -> *mut Context {
    let ctx = Box::new(Context {
        data: Vec::new(),
        config: Config::default(),
    });
    Box::into_raw(ctx)
}

/// 销毁上下文
///
/// # Safety
/// ctx must be a valid pointer from mylib_context_create
/// After this call, ctx is invalid and must not be used
#[no_mangle]
pub unsafe extern "C" fn mylib_context_destroy(ctx: *mut Context) {
    if !ctx.is_null() {
        drop(Box::from_raw(ctx));
    }
}

/// 重置上下文（重新初始化）
///
/// # Safety
/// ctx must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn mylib_context_reset(ctx: *mut Context) {
    if !ctx.is_null() {
        // 重新初始化而不释放内存
        ptr::drop_in_place(ctx);
        ptr::write(ctx, Context {
            data: Vec::new(),
            config: Config::default(),
        });
    }
}
```

### 5.2 错误处理模式

```rust
use std::ffi::CStr;
use std::os::raw::{c_char, c_int};
use std::ptr;

/// 错误码定义
#[repr(C)]
#[derive(Debug, Copy, Clone, PartialEq, Eq)]
pub enum ErrorCode {
    Success = 0,
    NullPointer = -1,
    InvalidArgument = -2,
    OutOfMemory = -3,
    IoError = -4,
    Utf8Error = -5,
    // ...
}

/// 线程本地错误信息存储
thread_local! {
    static LAST_ERROR: std::cell::RefCell<Option<Box<dyn std::error::Error>>> =
        std::cell::RefCell::new(None);
}

/// 设置最后错误
fn set_last_error<E: std::error::Error + 'static>(err: E) {
    LAST_ERROR.with(|e| {
        *e.borrow_mut() = Some(Box::new(err));
    });
}

/// 获取错误码（C接口）
#[no_mangle]
pub extern "C" fn mylib_last_error_code() -> c_int {
    LAST_ERROR.with(|e| {
        match *e.borrow() {
            None => ErrorCode::Success as c_int,
            Some(ref err) => {
                // 根据错误类型映射错误码
                if err.downcast_ref::<std::io::Error>().is_some() {
                    ErrorCode::IoError as c_int
                } else if err.downcast_ref::<std::str::Utf8Error>().is_some() {
                    ErrorCode::Utf8Error as c_int
                } else {
                    ErrorCode::InvalidArgument as c_int
                }
            }
        }
    })
}

/// 获取错误信息（C接口）
///
/// # Safety
/// Caller must free the returned string using mylib_string_free
#[no_mangle]
pub unsafe extern "C" fn mylib_last_error_message() -> *mut c_char {
    LAST_ERROR.with(|e| {
        match *e.borrow() {
            None => {
                let ok = std::ffi::CString::new("Success").unwrap();
                ok.into_raw()
            }
            Some(ref err) => {
                let msg = format!("{}", err);
                match std::ffi::CString::new(msg) {
                    Ok(c_msg) => c_msg.into_raw(),
                    Err(_) => ptr::null_mut(),
                }
            }
        }
    })
}

/// 清除错误状态
#[no_mangle]
pub extern "C" fn mylib_clear_error() {
    LAST_ERROR.with(|e| {
        *e.borrow_mut() = None;
    });
}

/// 使用示例函数
///
/// # Safety
/// input must be a valid null-terminated string
#[no_mangle]
pub unsafe extern "C" fn mylib_process(input: *const c_char) -> c_int {
    if input.is_null() {
        set_last_error(std::io::Error::new(
            std::io::ErrorKind::InvalidInput,
            "null pointer"
        ));
        return ErrorCode::NullPointer as c_int;
    }

    match CStr::from_ptr(input).to_str() {
        Ok(s) => {
            // 处理逻辑...
            println!("Processing: {}", s);
            ErrorCode::Success as c_int
        }
        Err(e) => {
            set_last_error(e);
            ErrorCode::Utf8Error as c_int
        }
    }
}
```

### 5.3 线程安全

```rust
use std::sync::{Arc, Mutex, RwLock};
use std::sync::atomic::{AtomicU64, Ordering};
use std::os::raw::c_void;

/// 线程安全的计数器
pub struct ThreadSafeCounter {
    value: AtomicU64,
}

impl ThreadSafeCounter {
    pub fn new() -> Self {
        ThreadSafeCounter {
            value: AtomicU64::new(0),
        }
    }
}

/// 创建计数器（线程安全）
#[no_mangle]
pub extern "C" fn mylib_counter_create() -> *mut ThreadSafeCounter {
    let counter = Box::new(ThreadSafeCounter::new());
    Box::into_raw(counter)
}

/// 销毁计数器
///
/// # Safety
/// counter must be a valid pointer and not accessed by other threads
#[no_mangle]
pub unsafe extern "C" fn mylib_counter_destroy(counter: *mut ThreadSafeCounter) {
    if !counter.is_null() {
        drop(Box::from_raw(counter));
    }
}

/// 递增计数器（线程安全）
///
/// # Safety
/// counter must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn mylib_counter_increment(counter: *mut ThreadSafeCounter) -> u64 {
    if counter.is_null() {
        return 0;
    }
    (*counter).value.fetch_add(1, Ordering::SeqCst) + 1
}

/// 获取值（线程安全）
///
/// # Safety
/// counter must be a valid pointer
#[no_mangle]
pub unsafe extern "C" fn mylib_counter_get(counter: *mut ThreadSafeCounter) -> u64 {
    if counter.is_null() {
        return 0;
    }
    (*counter).value.load(Ordering::SeqCst)
}

// ============================================
// 互斥锁保护的数据
// ============================================

pub struct SharedState {
    data: Mutex<Vec<u8>>,
    config: RwLock<Config>,
}

/// 创建共享状态
#[no_mangle]
pub extern "C" fn mylib_state_create() -> *mut SharedState {
    let state = Box::new(SharedState {
        data: Mutex::new(Vec::new()),
        config: RwLock::new(Config::default()),
    });
    Box::into_raw(state)
}

/// 向共享状态添加数据
///
/// # Safety
/// state must be a valid pointer, data must point to at least len bytes
#[no_mangle]
pub unsafe extern "C" fn mylib_state_append(
    state: *mut SharedState,
    data: *const u8,
    len: usize,
) -> c_int {
    if state.is_null() || data.is_null() {
        return -1;
    }

    let slice = std::slice::from_raw_parts(data, len);

    match (*state).data.lock() {
        Ok(mut guard) => {
            guard.extend_from_slice(slice);
            0
        }
        Err(_) => -2, // 锁被污染
    }
}

// ============================================
// Send + Sync约束
// ============================================

/// 确保类型可以安全地跨线程边界传递
pub struct SafeHandle {
    inner: Arc<dyn Send + Sync>,
}

/// 回调函数上下文（必须是Send + Sync）
pub struct CallbackContext {
    callback: extern "C" fn(*const u8, usize, *mut c_void),
    user_data: *mut c_void,
}

// 明确标记为Send + Sync（前提是调用者保证线程安全）
unsafe impl Send for CallbackContext {}
unsafe impl Sync for CallbackContext {}
```
