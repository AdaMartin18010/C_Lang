### 2.1 数据类型理解

#### 题目 DT-001 【理解】

**sizeof 与数据类型大小**

预测以下代码的输出（假设在 64 位 Linux 系统上）：

```c
#include <stdio.h>

int main() {
    printf("sizeof(char) = %zu\n", sizeof(char));
    printf("sizeof(short) = %zu\n", sizeof(short));
    printf("sizeof(int) = %zu\n", sizeof(int));
    printf("sizeof(long) = %zu\n", sizeof(long));
    printf("sizeof(long long) = %zu\n", sizeof(long long));
    printf("sizeof(float) = %zu\n", sizeof(float));
    printf("sizeof(double) = %zu\n", sizeof(double));
    printf("sizeof(void*) = %zu\n", sizeof(void*));
    return 0;
}
```

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌱 L1 |
| **考察点** | 基本数据类型大小、sizeof 运算符 |
| **答案** | 1, 2, 4, 8, 8, 4, 8, 8（64位Linux）|
| **解析** | C标准只规定相对大小：sizeof(char) ≤ sizeof(short) ≤ sizeof(int) ≤ sizeof(long) ≤ sizeof(long long)。具体大小取决于平台和编译器。 |

#### 题目 DT-002 【应用】

**整数溢出检测**

编写程序检测两个正整数相加是否会导致溢出。

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌿 L2 |
| **考察点** | 整数溢出、无符号整数运算 |
| **核心思路** | 对于无符号整数：`if (a > UINT_MAX - b)` 则溢出；对于有符号整数：`if (b > 0 && a > INT_MAX - b)` 或 `if (b < 0 && a < INT_MIN - b)` |

**参考答案：**

```c
#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

// 检测无符号整数加法溢出
bool unsigned_add_overflow(unsigned int a, unsigned int b) {
    return a > UINT_MAX - b;
}

// 检测有符号整数加法溢出
bool signed_add_overflow(int a, int b) {
    if (b > 0 && a > INT_MAX - b) return true;
    if (b < 0 && a < INT_MIN - b) return true;
    return false;
}

int main() {
    unsigned int ua = UINT_MAX - 1, ub = 2;
    int sa = INT_MAX - 1, sb = 3;

    printf("无符号溢出检测: %s\n",
           unsigned_add_overflow(ua, ub) ? "会溢出" : "不会溢出");
    printf("有符号溢出检测: %s\n",
           signed_add_overflow(sa, sb) ? "会溢出" : "不会溢出");

    return 0;
}
```

#### 题目 DT-003 【理解】

**浮点数精度问题**

以下代码的输出是什么？为什么？

```c
#include <stdio.h>

int main() {
    double a = 0.1;
    double b = 0.2;
    double sum = a + b;

    if (sum == 0.3) {
        printf("Equal\n");
    } else {
        printf("Not equal: %.17f\n", sum);
    }
    return 0;
}
```

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌿 L2 |
| **考察点** | IEEE 754浮点数表示、精度损失 |
| **答案** | 输出 "Not equal: 0.30000000000000004" |
| **解析** | 0.1和0.2在二进制浮点数中无法精确表示，相加后产生微小误差。比较浮点数应使用误差范围：`fabs(a - b) < epsilon` |

#### 题目 DT-004 【分析】

**结构体内存对齐**

以下结构体在64位系统上的大小是多少？成员偏移量分别是多少？

```c
struct Example {
    char a;
    int b;
    char c;
    double d;
    char e;
};
```

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌲 L3 |
| **考察点** | 内存对齐、结构体填充、#pragma pack |
| **答案** | 大小为32字节（默认对齐）|
| **布局分析** | a(0-0, 1B), 填充(1-3, 3B), b(4-7, 4B), c(8-8, 1B), 填充(9-15, 7B), d(16-23, 8B), e(24-24, 1B), 填充(25-31, 7B) |
| **优化建议** | 按大小降序排列成员可减少内存占用：double(8), int(4), char(3个共3B) → 总共16字节 |

### 2.2 运算符优先级练习

#### 题目 OP-001 【理解】

**运算符优先级分析**

预测以下代码的输出：

```c
#include <stdio.h>

int main() {
    int a = 5, b = 3, c = 2;
    int r1 = a + b * c;
    int r2 = a * b + c;
    int r3 = a & b == c;
    int r4 = a == b & c;
    int r5 = a << 1 + 2;
    int r6 = a + ++a;

    printf("r1=%d, r2=%d, r3=%d, r4=%d, r5=%d, r6=%d\n",
           r1, r2, r3, r4, r5, r6);
    return 0;
}
```

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌲 L3 |
| **考察点** | 运算符优先级、未定义行为 |
| **答案** | r1=11, r2=17, r3=0, r4=0, r5=40, r6=未定义行为 |
| **解析** | 优先级顺序：算术 > 移位 > 关系 > 位运算 > 逻辑。注意：`a + ++a` 是未定义行为（同一变量在序列点间被修改和访问）。r3: `a & (b == c)` → `5 & 0` → 0。r5: `a << (1 + 2)` → `5 << 3` → 40。 |

#### 题目 OP-002 【应用】

**位运算技巧**

使用位运算实现以下功能：

1. 判断一个整数是否为2的幂
2. 计算一个整数的二进制中1的个数
3. 交换两个整数不使用临时变量
4. 清零最低位的1

| 属性 | 内容 |
|:---|:---|
| **难度** | 🌲 L3 |
| **考察点** | 位运算、位操作技巧 |

**参考答案：**

```c
#include <stdio.h>
#include <stdbool.h>

// 1. 判断是否为2的幂：二进制只有一位是1
bool is_power_of_2(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}

// 2. 计算1的个数（Brian Kernighan算法）
int count_bits(int n) {
    int count = 0;
    while (n) {
        n &= n - 1;  // 清零最低位的1
        count++;
    }
    return count;
}

// 3. 交换两个整数
void swap(int *a, int *b) {
    *a ^= *b;
    *b ^= *a;
    *a ^= *b;
}

// 4. 清零最低位的1

    swap(&x, &y);
    printf("交换后: x=%d, y=%d\n", x, y);

    printf("12(1100)清零最低位1: %d\n", clear_lowest_bit(12));  // 8(1000)
```

**参考答案：**

```c
int binary_search(int arr[], int n, int target) {
    int left = 0, right = n - 1;

    while (left <= right) {
        int mid = left + (right - left) / 2;  // 防止溢出

        if (arr[mid] == target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}
```

```c
void create_matrix(int n) {

        matrix[i] = malloc(n * sizeof(int));
```

```c
    // 先释放每行
    for (int i = 0; i < n; i++) {
        free(matrix[i]);
    }
    // 再释放指针数组
    free(matrix);
}
```

```c
#include <stdio.h>
```

```c
bool is_prime(int n) {
    if (n < 2) return false;
    if (n == 2) return true;
```

#### L2-010 二分查找 详细解析

**为什么用 `left + (right - left) / 2` 而不是 `(left + right) / 2`？**

防止整数溢出！当left和right都很大时，left + right可能溢出。

**常见错误：**

- 循环条件写成 `left < right`，可能漏查元素
- 更新边界时写成 `left = mid` 或 `right = mid`，可能导致死循环
- 查找边界条件混淆（找第一个/最后一个）

**变体：查找第一个等于target的位置**

```c
int lower_bound(int arr[], int n, int target) {
    int left = 0, right = n;  // 注意right=n
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    return left;  // 返回第一个>=target的位置
}
```

```c
bool is_palindrome(const char *str) {
    const char *left = str;
    const char *right = str;

    // 找到字符串末尾
    while (*right) right++;
    right--;

    while (left < right) {

        while (left < right && !isalpha(*left)) left++;
        while (left < right && !isalpha(*right)) right--;

        if (tolower(*left) != tolower(*right)) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}
```

### 6.3 L3 进阶练习答案

#### L3-001 快速幂 参考答案

```c
#include <stdio.h>

// 递归版本
long long fast_pow(long long base, int exp) {
    if (exp == 0) return 1;
    if (exp == 1) return base;

    long long half = fast_pow(base, exp / 2);
    if (exp % 2 == 0) {
        return half * half;
    } else {
        return half * half * base;
    }
}

// 迭代版本（更推荐）
long long fast_pow_iter(long long base, int exp) {
    long long result = 1;
    while (exp > 0) {
        if (exp & 1) {  // exp是奇数
            result *= base;
        }
        base *= base;
        exp >>= 1;
    }
    return result;
}

int main() {
    printf("2^10 = %lld\n", fast_pow(2, 10));
    printf("3^20 = %lld\n", fast_pow_iter(3, 20));
    return 0;
}
```

**复杂度分析：**

- 时间复杂度：O(log exp)，每次指数减半
- 空间复杂度：递归O(log exp)，迭代O(1)

#### L3-014 单向链表实现

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

// 创建新节点
Node* create_node(int data) {
    Node *new_node = malloc(sizeof(Node));
    if (new_node == NULL) {
        fprintf(stderr, "内存分配失败\n");
        exit(1);
    }
    new_node->data = data;
    new_node->next = NULL;
    return new_node;
}

// 头部插入
void insert_head(Node **head, int data) {
    Node *new_node = create_node(data);
    new_node->next = *head;
    *head = new_node;
}

// 尾部插入
void insert_tail(Node **head, int data) {
    Node *new_node = create_node(data);

    if (*head == NULL) {
        *head = new_node;
        return;
    }

    Node *current = *head;
    while (current->next != NULL) {
        current = current->next;
    }
    current->next = new_node;
}

// 删除节点（删除第一个值为data的节点）
void delete_node(Node **head, int data) {
    if (*head == NULL) return;

    // 头节点就是要删除的节点
    if ((*head)->data == data) {
        Node *temp = *head;
        *head = (*head)->next;
        free(temp);
        return;
    }

    Node *current = *head;
    while (current->next != NULL && current->next->data != data) {
        current = current->next;
    }

    if (current->next != NULL) {
        Node *temp = current->next;
        current->next = current->next->next;
        free(temp);
    }
}

// 查找节点
Node* find(Node *head, int data) {
    Node *current = head;
    while (current != NULL) {
        if (current->data == data) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}

// 打印链表
void print_list(Node *head) {
    Node *current = head;
    printf("[");
    while (current != NULL) {
        printf("%d", current->data);
        if (current->next != NULL) {
            printf(" -> ");
        }
        current = current->next;
    }
    printf("]\n");
}

// 释放链表
void free_list(Node **head) {
    Node *current = *head;
    while (current != NULL) {
        Node *temp = current;
        current = current->next;
        free(temp);
    }
    *head = NULL;
}

// 测试
int main() {
    Node *list = NULL;

    insert_tail(&list, 1);
    insert_tail(&list, 2);
    insert_tail(&list, 3);
    insert_head(&list, 0);

    print_list(list);  // [0 -> 1 -> 2 -> 3]

    delete_node(&list, 2);
    print_list(list);  // [0 -> 1 -> 3]

    free_list(&list);
    return 0;
}
```

**常见错误：**

- 忘记检查malloc返回值
- 删除节点时忘记处理头节点特殊情况
- 释放链表后没有将head置NULL
- 忘记free导致内存泄漏

### 6.4 L4 挑战练习答案

#### L4-001 表达式求值 参考答案

```c
#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>

const char *expr;
int pos = 0;

// 前向声明
double parse_expression();
double parse_term();
double parse_factor();

// 解析数字
double parse_number() {
    double result = 0;
    while (isdigit(expr[pos])) {
        result = result * 10 + (expr[pos] - '0');
        pos++;
    }
    if (expr[pos] == '.') {
        pos++;
        double fraction = 0.1;
        while (isdigit(expr[pos])) {
            result += (expr[pos] - '0') * fraction;
            fraction *= 0.1;
            pos++;
        }
    }
    return result;
}

// factor := number | '(' expression ')'
double parse_factor() {
    if (expr[pos] == '(') {
        pos++;
        double result = parse_expression();
        if (expr[pos] == ')') {
            pos++;
        }
        return result;
    }
    return parse_number();
}

// term := factor { ('*'|'/') factor }
double parse_term() {
    double result = parse_factor();
    while (expr[pos] == '*' || expr[pos] == '/') {
        char op = expr[pos++];
        double right = parse_factor();
        if (op == '*') {
            result *= right;
        } else {
            if (right == 0) {
                fprintf(stderr, "错误: 除零\n");
                exit(1);
            }
            result /= right;
        }
    }
    return result;
}

// expression := term { ('+'|'-') term }
double parse_expression() {
    double result = parse_term();
    while (expr[pos] == '+' || expr[pos] == '-') {
        char op = expr[pos++];
        double right = parse_term();
        if (op == '+') {
            result += right;
        } else {
            result -= right;
        }
    }
    return result;
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "用法: %s \"表达式\"\n", argv[0]);
        return 1;
    }

    expr = argv[1];
    double result = parse_expression();
    printf("结果: %f\n", result);

    return 0;
}
```

**文法说明：**

- expression: 处理加减法（最低优先级）
- term: 处理乘除法（较高优先级）
- factor: 处理数字和括号（最高优先级）
