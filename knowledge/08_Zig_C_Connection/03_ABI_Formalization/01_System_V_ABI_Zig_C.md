### 1.2 核心设计原则

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                   System V AMD64 ABI 设计原则                                │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  原则 1: 最大化寄存器使用                                                    │
│  ─────────────────────────────────────────                                   │
│  • 前6个整数/指针参数 → %rdi, %rsi, %rdx, %rcx, %r8, %r9                     │
│  • 前8个浮点/向量参数 → %xmm0-%xmm7                                          │
│  • 减少内存访问，提升性能                                                      │
│                                                                              │
│  原则 2: 保持栈对齐                                                          │
│  ─────────────────────────────────────────                                   │
│  • 栈必须始终保持 16 字节对齐                                                 │
│  • CALL 指令隐式将 8 字节返回地址压栈                                        │
│  • 因此 RSP+8 必须是 16 的倍数                                               │
│                                                                              │
│  原则 3: 简单类型的简单传递                                                  │
│  ─────────────────────────────────────────                                   │
│  • 整数、指针直接放入通用寄存器                                                │
│  • 浮点数直接放入 SSE 寄存器                                                   │
│  • 复合类型通过分类系统决定传递方式                                            │
│                                                                              │
│  原则 4: 与 x86-32 ABI 的兼容性                                              │
│  ─────────────────────────────────────────                                   │
│  • 保持与 legacy 代码的互操作能力                                             │
│  • 支持混合 32/64 位进程 (WoW64)                                              │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 2.2 调用者保存 vs 被调用者保存

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    寄存器保存责任分配                                         │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  被调用者保存寄存器 (Callee-Saved):                                           │
│  ─────────────────────────────────────────────────────────────────────────  │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ %rbx, %rbp, %r12, %r13, %r14, %r15                                  │   │
│  │ %xmm8-%xmm15 (如果目标支持 AVX, %ymm8-%ymm15)                       │   │
│  ├─────────────────────────────────────────────────────────────────────┤   │
│  │ 规则: 如果被调用者修改了这些寄存器，必须恢复原始值                   │   │
│  │ 用途: 长期存储，跨函数调用保持                                       │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                              │
│  调用者保存寄存器 (Caller-Saved / Volatile):                                  │
│  ─────────────────────────────────────────────────────────────────────────  │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ %rax, %rcx, %rdx, %rsi, %rdi, %r8-%r11                              │   │
│  │ %xmm0-%xmm7                                                         │   │
│  │ %rflags (标志寄存器)                                                │   │
│  │ %st(0)-%st(7) (x87 FPU 栈)                                          │   │
│  ├─────────────────────────────────────────────────────────────────────┤   │
│  │ 规则: 调用者假设这些寄存器在函数调用后可能被修改                     │   │
│  │ 用途: 临时计算，参数传递，返回值                                     │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                              │
│  特殊寄存器:                                                                 │
│  ─────────────────────────────────────────────────────────────────────────  │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ %rsp    │ 栈指针 - 必须始终保持 16 字节对齐，被调用者保存             │   │
│  │ %rip    │ 指令指针 - 程序计数器                                       │   │
│  │ %rflags │ 标志寄存器 - 调用者保存                                     │   │
│  │ %st(0)  │ x87 栈顶 - 用于 long double 返回值                         │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

## 三、参数传递分类系统

### 3.1 八个参数类别

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    System V AMD64 ABI 参数类别                               │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  类别          │ 描述                          │ 传递方式                  │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  INTEGER      │ 整数类型 (char, short, int,   │ 通用寄存器 (rdi, rsi,    │
│               │ long, pointer, __int128)      │ rdx, rcx, r8, r9)        │
│               │                               │ 或栈                      │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  SSE          │ 浮点类型 (float, double)      │ SSE 寄存器 (xmm0-xmm7)   │

│  ─────────────┼───────────────────────────────┼─────────────────────────  │

│               │                               │                           │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  X87UP        │ X87 类型的上部                │ 与 X87 配合               │
│               │                               │                           │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  COMPLEX_X87  │ 复数类型 (complex long double)│ 通过 x87 栈               │
│               │                               │                           │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  NO_CLASS     │ 无数据/填充                    │ 不传递                    │
│               │                               │                           │
│  ─────────────┼───────────────────────────────┼─────────────────────────  │
│  MEMORY       │ 必须通过内存传递               │ 栈上的参数区              │
│               │ (大结构体/联合体)              │ 或通过隐藏指针            │
│               │                               │                           │
│                                                                              │
│  寄存器限制:                                                                 │
│  ─────────────────                                                           │
│  • 最多 6 个 INTEGER 类参数可使用寄存器                                      │
│  • 最多 8 个 SSE 类参数可使用寄存器                                          │
│  • 超过限制或类型为 MEMORY 的参数使用栈                                       │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 3.2 分类算法

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    类型分类算法 (伪代码)                                      │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  function classify_type(type, offset):                                       │
│      if type is integer or pointer:                                          │
│          return INTEGER                                                      │
│                                                                              │
│      if type is float or double:                                             │
│          return SSE                                                          │
│                                                                              │
│      if type is long double:                                                 │
│          return X87                                                          │
│                                                                              │
│      if type is complex long double:                                         │
│          return COMPLEX_X87                                                  │
│                                                                              │
│      if type is struct or union:                                             │
│          size = sizeof(type)                                                 │
│          if size > 16 bytes or has unaligned fields:                         │
│              return MEMORY                                                   │
│                                                                              │
│          // 将结构体划分为 8 字节字                                          │
│          words = ceil(size / 8)                                              │
│          classes = array of NO_CLASS[words]                                  │
│                                                                              │
│          for each field f in type:                                           │
│              field_offset = offset of f in type                              │
│              field_class = classify_type(f, field_offset)                    │
│              word_idx = field_offset / 8                                     │
│              classes[word_idx] = merge_classes(classes[word_idx],            │
│                                               field_class)                   │
│                                                                              │
│              if field spans multiple words:                                  │
│                  classes[word_idx + 1] = field_class (or SSEUP/X87UP)        │
│                                                                              │
│          // 后处理                                                           │
│          if any class is MEMORY:                                             │
│              return MEMORY                                                   │
│                                                                              │
│          if classes[0] is X87 and words == 2:                                │
│              classes[1] = X87UP                                              │
│                                                                              │
│          if classes[0] is SSE and words == 2:                                │
│              classes[1] = SSEUP                                              │
│                                                                              │
│          return classes                                                      │
│                                                                              │
│  function merge_classes(class1, class2):                                     │
│      if class1 == class2: return class1                                      │
│      if class1 == NO_CLASS: return class2                                    │
│      if class2 == NO_CLASS: return class1                                    │
│      if class1 == MEMORY or class2 == MEMORY: return MEMORY                  │
│      if class1 == INTEGER or class2 == INTEGER: return INTEGER               │
│      if class1 == X87 or class2 == X87: return MEMORY                        │
│      if class1 == X87UP or class2 == X87UP: return MEMORY                    │
│      if class1 == COMPLEX_X87 or class2 == COMPLEX_X87: return MEMORY        │
│      return SSE                                                              │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 3.3 分类示例

```c
// C 代码: 参数分类示例

// 单个整数 - INTEGER 类
void f1(int x);                    // %rdi

// 多个整数 - 依次使用寄存器
void f2(int a, int b, int c,       // %rdi, %rsi, %rdx
        int d, int e, int f);      // %rcx, %r8, %r9

// 第7个整数 - 使用栈
void f3(int a, int b, int c,       // %rdi, %rsi, %rdx
        int d, int e, int f,       // %rcx, %r8, %r9
        int g);                    // 8(%rsp)

// 混合类型
void f4(int x, double y,           // %rdi, %xmm0
        long z, float w);          // %rsi, %xmm1

// 结构体 (16字节, 两个整数) - 两个 INTEGER 类

// 结构体 (含浮点) - SSE 类
struct Vec2 { double x, y; };      // %xmm0, %xmm1 (或 ymm0)
void f6(struct Vec2 v);
```

### 6.2 内存传递规则

```c
struct BigStruct {
    long a, b, c;    // 24 bytes
};
void big_struct(struct BigStruct s) {

struct Unaligned {

}

}
```

## 七、栈帧布局

### 7.1 栈帧结构

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    System V AMD64 栈帧布局                                   │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  高地址                                                                      │
│  ─────────────────────────────────────────────────────────────────────────  │
│  │  ...                                                                    │
│  │  调用者的栈帧                                                            │
│  │  ...                                                                    │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  返回地址                    ← %rsp (函数入口)                           │
│  │  [8 bytes]                  ← 由 CALL 指令压入                           │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  被保存的寄存器                ← push %rbp; push %rbx ...                 │
│  │  (%rbp, %rbx, %r12-%r15)                                                │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  局部变量区                                                              │
│  │  (编译器分配)                                                            │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  临时栈空间 / 动态分配                                                    │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  参数区 (用于传递额外参数)                                                │
│  │  8(%rsp)   - 第7个整数参数 (如果存在)                                    │
│  │  16(%rsp)  - 第8个整数参数 或 第9个 SSE 参数                             │
│  │  ...                                                                     │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  红区 (Red Zone)           ← -128(%rsp) 到 (%rsp)                        │
│  │  [128 bytes below %rsp]   ← 叶子函数可用，不会被信号处理破坏             │
│  │                           ← 注意: 变参函数不能使用红区                   │
│  ├──────────────────────────────────────────────────────────────────────   │
│  │  被调用者的栈帧 (当前函数调用其他函数时)                                  │
│  │  ...                                                                     │
│  ─────────────────────────────────────────────────────────────────────────  │
│  低地址                                                                      │
│                                                                              │
│  栈对齐规则:                                                                 │
│  ─────────────────                                                           │
│  • %rsp 必须在函数入口时 16 字节对齐                                          │
│  • CALL 压入返回地址 (8 bytes)，所以 RSP+8 是 16 的倍数                       │
│  • 分配栈空间时必须保持 16 字节对齐                                            │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 7.2 参数区布局

```text

│                    栈参数区布局                                              │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  栈参数区 (%rsp + 8 起始，因为返回地址占 8 bytes):                            │
│  ─────────────────────────────────────────────────────────────────────────  │
│                                                                              │

│  ────────────────────┼───────────────────────────────┼─────────────────────  │
│  0(%rsp)            │  返回地址 (CALL 压入)          │  -                     │

│  16(%rsp)           │  第8个 INTEGER 参数            │  8-byte                │
│  24(%rsp)           │  ...                          │  8-byte                │
│  ...                │  第9+个 INTEGER 参数           │  8-byte                │
│  ────────────────────┼───────────────────────────────┼─────────────────────  │
│  8(%rsp)            │  第9个 SSE 参数                │  16-byte               │
```

```asm
; 生成的汇编代码

_stack_frame_example:
    pushq   %rbp                ; 保存旧帧指针
    movq    %rsp, %rbp          ; 设置新帧指针
    subq    $32, %rsp           ; 分配 32 字节栈空间 (16 字节对齐)

    ; 局部变量
    ; local1: -8(%rbp)
    ; local2: -12(%rbp)

    leal    (%rdi,%rsi), %eax   ; x + y
    movl    %eax, -8(%rbp)      ; local1 = %eax

    movl    %edi, %eax
    subl    %esi, %eax          ; x - y
    movl    %eax, -12(%rbp)     ; local2 = %eax

    ; 准备调用 printf
    movl    -12(%rbp), %edx     ; 第3个参数
    movl    -8(%rbp), %esi      ; 第2个参数
    leaq    .LC0(%rip), %rdi    ; 第1个参数 (格式字符串)
    xorl    %eax, %eax          ; 清除 AL (无向量寄存器用于变参)
    call    printf@PLT

    leave                       ; 恢复栈帧
    ret
```

### 7.4 栈对齐要求

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    栈对齐要求详解                                            │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  对齐要求:                                                                   │
│  ─────────────────────────────────────────────────────────────────────────  │
│  • 函数入口: %rsp 必须 16 字节对齐                                            │
│  • 任何时刻: %rsp 必须 16 字节对齐                                            │
│  • CALL 指令: 自动压入 8 字节返回地址，破坏对齐                                │
│  • 因此: 进入函数后，%rsp + 8 是 16 的倍数                                    │
│                                                                              │
│  对齐示例:                                                                   │
│  ─────────────────                                                           │
│  进入函数前:    %rsp = 0x...FF0 (16 字节对齐)                                 │
│  CALL 之后:     %rsp = 0x...FE8 (16 字节对齐 - 8)                             │
│  push %rbp:     %rsp = 0x...FE0 (16 字节对齐)                                 │
│  subq $32,%rsp: %rsp = 0x...FC0 (16 字节对齐)                                 │
│                                                                              │
│  常见错误:                                                                   │
│  ─────────────────                                                           │
│  1. 手工汇编时忘记对齐                                                        │
│     pushq %rax      ; 错误！破坏对齐                                         │
│     call some_func  ; 调用时 %rsp 未对齐                                     │
│                                                                              │
│  2. 正确的做法:                                                               │
│     pushq %rax                                                      │
│     pushq %rbx      ; 再压一个，恢复对齐                                     │
│     call some_func                                                          │
│     popq %rbx                                                               │
│     popq %rax                                                               │
│                                                                              │
│  3. 或使用 subq 分配空间:                                                     │
│     subq $16, %rsp  ; 分配 16 字节，保持对齐                                 │
│     movq %rax, 8(%rsp)                                                      │
│     call some_func                                                          │
│     addq $16, %rsp                                                          │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

```c
typedef struct {
    unsigned int gp_offset;      // 下一个 GP 参数偏移 (字节)
    unsigned int fp_offset;      // 下一个 FP 参数偏移 (字节)
    void* overflow_arg_area;     // 栈参数区指针

int sum_variadic(int count, ...) {

    return sum;
}
```

## 十、形式化 ABI 兼容性证明

### 10.1 类型分类等价性

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    定理 1: 类型分类等价性                                     │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  定理陈述:                                                                   │
│  ─────────────────────────────────────────────────────────────────────────  │
│                                                                              │
│  对于任意 C 类型 T_C 和对应的 Zig extern 类型 T_Z，                          │
│  若 T_C ≅ T_Z (布局等价)，则:                                                │
│                                                                              │
│      classify_C(T_C) = classify_Zig(T_Z)                                     │
│                                                                              │
│  其中 ≅ 表示布局等价关系，classify 表示 System V AMD64 ABI 分类函数。        │
│                                                                              │
│  证明:                                                                       │
│  ─────────────────                                                           │
│                                                                              │
│  基础情形:                                                                   │
│  ─────────────────                                                           │
│                                                                              │
│  1. 整数类型:                                                                │
│     • C int{n}_t  ≅ Zig i{n}  (n ∈ {8,16,32,64})                             │
│     • sizeof 相等 → 分类算法判定为 INTEGER                                   │
│     ∴ classify_C(int32_t) = classify_Zig(i32) = INTEGER                      │
│                                                                              │
│  2. 浮点类型:                                                                │
│     • C float  ≅ Zig f32                                                     │
│     • C double ≅ Zig f64                                                     │
│     • 都满足 IEEE 754 标准，sizeof 相等                                       │
│     ∴ classify_C(float) = classify_Zig(f32) = SSE                            │
│     ∴ classify_C(double) = classify_Zig(f64) = SSE                           │
│                                                                              │
│  3. 指针类型:                                                                │
│     • C T* ≅ Zig *T (或 ?*T 对于可为空的指针)                                 │
│     • 都是 64位地址                                                          │
│     ∴ classify_C(T*) = classify_Zig(*T) = INTEGER                            │
│                                                                              │
│  归纳情形:                                                                   │
│  ─────────────────                                                           │
│                                                                              │
│  4. 结构体类型:                                                              │
│     假设: 对于结构体的每个字段 f_i，                                         │
│           classify_C(f_i^C) = classify_Zig(f_i^Z)                            │
│                                                                              │
│     设 S_C = struct { f_1^C, ..., f_n^C }                                    │
│     设 S_Z = extern struct { f_1^Z, ..., f_n^Z }                             │
│                                                                              │
│     若 S_C ≅ S_Z (相同字段顺序、相同偏移、相同填充)，                         │
│     则分类算法处理每个字的字段分类相同，                                       │
│     合并操作结果相同。                                                        │
│     ∴ classify_C(S_C) = classify_Zig(S_Z)                                    │
│                                                                              │
│  5. 联合体类型:                                                              │
│     类似结构体，考虑最大字段的分类。                                          │
│     ∴ classify_C(U_C) = classify_Zig(U_Z)                                    │
│                                                                              │
│  ∎                                                                           │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

```text
│  ─────────────────────────────────────────────────────────────────────────  │
│                                                                              │
│  对于函数 F_C (C 实现) 和 F_Z (Zig 实现)，若其参数类型列表布局等价，          │
│  则参数到寄存器的映射相同。                                                   │
│                                                                              │
│  形式化:                                                                     │
│  ─────────────────                                                           │
│                                                                              │
│  设 params_C = [p_1^C, ..., p_n^C]                                           │
│  设 params_Z = [p_1^Z, ..., p_n^Z]                                           │
│                                                                              │
│  若 ∀i. p_i^C ≅ p_i^Z，则:                                                   │
│      ∀i. reg_assignment_C(p_i^C) = reg_assignment_Zig(p_i^Z)                 │
│                                                                              │
│  其中 reg_assignment 根据 System V AMD64 ABI 规则分配寄存器或栈槽。          │
│                                                                              │
│  证明概要:                                                                   │
│  ─────────────────                                                           │
│                                                                              │
│  1. 独立计数器引理:                                                          │
│     INTEGER 类和 SSE 类的寄存器分配使用独立计数器。                           │
│     • int_reg_idx = 0, sse_reg_idx = 0 初始化                                │
│     • 遇到 INTEGER 类参数: 使用 int_regs[int_reg_idx++]                      │
│     • 遇到 SSE 类参数: 使用 sse_regs[sse_reg_idx++]                          │
│                                                                              │
│  2. 分类决定寄存器:                                                          │
│     由定理 1，classify_C(p_i^C) = classify_Zig(p_i^Z)                        │
│     ∴ 两个计数器在 C 和 Zig 中同步递增                                       │
│                                                                              │
│  3. 寄存器耗尽处理:                                                          │
│     当 int_reg_idx > 6 或 sse_reg_idx > 8 时，                               │
│     参数通过栈传递。                                                          │
│     由于分类相同，耗尽点相同，栈槽分配相同。                                  │
│                                                                              │
│  ∎                                                                           │
│                                                                              │
│  推论:                                                                       │
│  ─────────────────                                                           │
│  C 函数和等价的 Zig 函数生成的调用序列可以互相调用，无需适配层。               │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 10.3 语义保持定理

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    定理 3: ABI 语义保持定理                                   │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  定理陈述:                                                                   │
│  ─────────────────────────────────────────────────────────────────────────  │
│                                                                              │
│  设 P_C 是一个 C 程序，P_Z 是对应的 Zig 程序，满足:                            │
│  1. 所有数据类型布局等价 (T_C ≅ T_Z)                                         │
│  2. 所有函数签名 ABI 等价                                                    │
│  3. P_Z 使用 extern 和 callconv(.C) 声明与 C 互操作的部分                    │
│                                                                              │
│  则: P_C 和 P_Z 在 System V AMD64 ABI 层面语义等价，                          │
│      即它们产生相同的寄存器状态和内存状态。                                    │
│                                                                              │
│  形式化定义:                                                                 │
│  ─────────────────                                                           │
│                                                                              │
│  定义 ABI 状态 σ = (GPRs, XMMs, FPU, Stack, Flags)                           │
│  其中:                                                                       │
│    GPRs = {%rax, %rbx, ..., %r15}                                            │
│    XMMs = {%xmm0, ..., %xmm15}                                               │
│    FPU  = x87 栈状态                                                          │
│    Stack = 栈内存内容                                                         │
│    Flags = 标志寄存器                                                         │
│                                                                              │
│  定理:                                                                       │
│  ∀输入 I. exec_C(P_C, I) = exec_Zig(P_Z, I)                                  │
│                                                                              │
│  其中 = 表示 ABI 状态等价 (忽略非 ABI 可见的内部状态差异)。                  │
│                                                                              │
│  证明要点:                                                                   │
│  ─────────────────                                                           │
│                                                                              │
│  1. 调用前状态:                                                              │
│     • 参数按定理 2 分配到相同寄存器/栈位置                                     │
│     • 栈指针对齐要求相同 (16字节)                                             │
│     • 被调用者保存寄存器内容相同                                              │
│                                                                              │
│  2. 调用过程:                                                                │
│     • CALL 指令行为相同 (压入返回地址)                                        │
│     • 控制权转移机制相同                                                      │
│                                                                              │
│  3. 被调用者执行:                                                            │
│     • 栈帧建立 (保存 %rbp, 分配空间) 遵循相同模式                             │
│     • 参数访问使用相同的寄存器/栈偏移                                         │
│                                                                              │
│  4. 返回过程:                                                                │
│     • 返回值按相同规则放入 %rax/%xmm0                                         │
│     • 栈帧恢复顺序相同                                                        │
│     • RET 指令行为相同                                                        │
│                                                                              │
│  5. 调用后状态:                                                              │
│     • 调用者保存寄存器按约定可能被修改                                        │
│     • 返回值在相同位置                                                        │
│     • 栈状态恢复                                                              │
│                                                                              │
│  ∎                                                                           │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 11.2 类型对齐控制

```c
// C 代码: 对齐控制

#include <stdalign.h>
#include <stdint.h>

// 默认对齐的结构体
struct DefaultAlign {
    char c;
    int i;
};
// sizeof = 8, alignof = 4

// 显式对齐
struct ExplicitAlign {
    char c;
    alignas(16) int i;
};
// alignof = 16

// packed 结构体 (无填充)
struct __attribute__((packed)) PackedStruct {
    char c;
    int i;
};
// sizeof = 5, alignof = 1

// 对齐的缓冲区
alignas(64) char cache_line_buffer[64];
// 适合 SIMD 操作，避免伪共享
```

```zig
// Zig 代码: 对齐控制

// 默认对齐的结构体
const DefaultAlign = extern struct {
    c: u8,
    i: i32,
};
// Zig 会自动计算对齐，与 C 默认行为相同

// 显式对齐
const ExplicitAlign = extern struct {
    c: u8,
    // Zig 使用 align 关键字
    i: i32 align(16),
};

// packed 结构体
const PackedStruct = packed struct {
    c: u8,
    i: i32,
};
// 注意: Zig 的 packed struct 位序与 C 可能不同

// 对齐的缓冲区
var cache_line_buffer: [64]u8 align(64) = undefined;

// 在 Zig 中检查和确保对齐
fn ensureAlignment() void {
    comptime {
        // 编译时检查
        if (@alignOf(DefaultAlign) != 4) {
            @compileError("对齐不匹配");
        }
    }

    // 运行时检查
    const ptr: *u8 = &cache_line_buffer[0];
    std.debug.assert(@intFromPtr(ptr) % 64 == 0);
}

// C 兼容的对齐结构体
const AlignedForC = extern struct {
    data: [16]u8 align(16),
};

// 确保与 C 的 alignas(16) char[16] 兼容
comptime {
    assert(@sizeOf(AlignedForC) == 16);
    assert(@alignOf(AlignedForC) == 16);
}
```

### 11.3 常见陷阱与解决方案

```text
┌─────────────────────────────────────────────────────────────────────────────┐
│                    C/Zig 互操作常见陷阱                                      │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  陷阱 1: 布尔类型大小不匹配                                                   │
│  ─────────────────────────────────────────────────────────────────────────  │
│  C 代码:                                                                     │
│      #include <stdbool.h>                                                    │
│      void process(bool flag);    // _Bool, 通常 1 byte                       │
│                                                                              │
│  Zig 代码:                                                                   │
│      extern fn process(flag: bool) void;    // ⚠️ 危险！                      │
│      // Zig bool 是 1 bit 语义，但在 extern fn 中应与 C _Bool 兼容           │
│                                                                              │
│  解决方案:                                                                   │
│      extern fn process(flag: c_bool) void;  // 使用 std.c 中的别名           │
│      // 或                                                                   │
│      extern fn process(flag: u8) void;      // 显式使用 1 byte               │
│                                                                              │
│  陷阱 2: 变长数组 (VLA)                                                       │
│  ─────────────────────────────────────────────────────────────────────────  │
│  C 代码:                                                                     │
│      void process(int n, int arr[n]);  // VLA                               │
│                                                                              │
│  Zig 代码:                                                                   │
│      // Zig 不支持 VLA                                                       │
│      extern fn process(n: c_int, arr: [*]c_int) void;  // 使用多元素指针     │
│                                                                              │
│  陷阱 3: 位域布局差异                                                         │
│  ─────────────────────────────────────────────────────────────────────────  │
│  C 代码:                                                                     │
│      struct Flags {                                                          │
│          unsigned int a : 1;                                                 │
│          unsigned int b : 2;                                                 │
│      };                                                                      │
│      // C 位域布局是实现定义的                                               │
│                                                                              │
│  Zig 代码:                                                                   │
│      const Flags = packed struct {                                           │
│          a: u1,                                                              │
│          b: u2,                                                              │
│      };                                                                      │
│      // Zig packed struct 从 LSB 开始                                        │
│      // 可能与 C 不同！                                                      │
│                                                                              │
│  解决方案: 避免在 C/Zig 边界使用位域                                         │
│      // 使用显式掩码操作替代                                                 │
│      const Flags = extern struct {                                           │
│          bits: u32,                                                          │
│      };                                                                      │
│      const FLAG_A = 1 << 0;                                                  │
│      const FLAG_B = 3 << 1;                                                  │
│                                                                              │
│  陷阱 4: 空指针可空性                                                         │
│  ─────────────────────────────────────────────────────────────────────────  │
│  C 代码:                                                                     │
│      void* may_be_null(void);                                                │
│                                                                              │
│  Zig 代码:                                                                   │
│      extern fn may_be_null() *anyopaque;    // ⚠️ 危险！可能返回 null        │
│                                                                              │
│  正确做法:                                                                   │
│      extern fn may_be_null() ?*anyopaque;   // 可选指针                      │
│      // 使用时必须检查:                                                      │
│      if (may_be_null()) |ptr| { /* ... */ }                                  │

│  陷阱 5: 调用约定遗漏                                                         │
│  ─────────────────────────────────────────────────────────────────────────  │

│      const Callback = *const fn (i32) i32;  // ⚠️ 默认 Zig 调用约定          │
│                                                                              │
│  正确做法:                                                                   │
```

```c
// C 代码: AVX-512 调用约定

#include <immintrin.h>

// AVX-512 函数使用 ZMM 寄存器
__m512 add_avx512(__m512 a, __m512 b, __m512 c, __m512 d,
                  __m512 e, __m512 f, __m512 g, __m512 h) {
    // a-h: %zmm0-%zmm7
    __m512 sum1 = _mm512_add_ps(a, b);
    __m512 sum2 = _mm512_add_ps(c, d);

                   __m512 e, __m512 f, __m512 g, __m512 h,

    return i;
}
```

```zig
// Zig 代码: 向量调用约定

const std = @import("std");

// 检查目标特性
const has_avx512 = std.Target.x86.featureSetHas(
    std.Target.current.cpu.features, .avx512f
);

// AVX-512 向量类型
const Vec16f = @Vector(16, f32);  // 512-bit
const Vec8d = @Vector(8, f64);    // 512-bit

// 条件编译
export fn addAvx512(a: Vec16f, b: Vec16f, c: Vec16f, d: Vec16f,
                    e: Vec16f, f: Vec16f, g: Vec16f, h: Vec16f) callconv(.C) Vec16f {
    if (!has_avx512) {
        @panic("AVX-512 not supported");
    }
    // 生成: a-h 使用 %zmm0-%zmm7
    return a + b + c + d + e + f + g + h;
}

// 运行时 CPU 特性检测
pub fn detectAndCall(v: Vec16f) Vec16f {
    if (comptime has_avx512) {
        return addAvx512(v, v, v, v, v, v, v, v);
    } else {
        // 回退到 SSE/AVX 实现
        return fallbackAdd(v);
    }
}

fn fallbackAdd(v: Vec16f) Vec16f {
    // 标量或 AVX 实现
    return v;
}
```
